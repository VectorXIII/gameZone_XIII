27         private function normalizeOptions(array $options = array())
28         {
29             $options += array(
30                 'allowedOrigins' => array(),
31                 'allowedOriginsPatterns' => array(),
32                 'supportsCredentials' => false,
33                 'allowedHeaders' => array(),
34                 'exposedHeaders' => array(),
35                 'allowedMethods' => array(),
36                 'maxAge' => 0,
37             );
38     
39             // normalize array('*') to true
40             if (in_array('*', $options['allowedOrigins'])) {
41                 $options['allowedOrigins'] = true;
42             }
43             if (in_array('*', $options['allowedHeaders'])) {
44                 $options['allowedHeaders'] = true;
45             } else {
46                 $options['allowedHeaders'] = array_map('strtolower', $options['allowedHeaders']);
47             }
48     
49             if (in_array('*', $options['allowedMethods'])) {
50                 $options['allowedMethods'] = true;
51             } else {
52                 $options['allowedMethods'] = array_map('strtoupper', $options['allowedMethods']);
53             }
54     
55             return $options;
56         }

60             return $this->checkOrigin($request);

63         public function isCorsRequest(Request $request)
64         {
65             return $request->headers->has('Origin') && !$this->isSameHost($request);
66         }

68         public function isPreflightRequest(Request $request)
69         {
70             return $this->isCorsRequest($request)
71                 && $request->getMethod() === 'OPTIONS'
72                 && $request->headers->has('Access-Control-Request-Method');
73         }

75         public function addActualRequestHeaders(Response $response, Request $request)
76         {
77             if (!$this->checkOrigin($request)) {
78                 return $response;
79             }
80     
81             $response->headers->set('Access-Control-Allow-Origin', $request->headers->get('Origin'));
82     
83             if (!$response->headers->has('Vary')) {
84                 $response->headers->set('Vary', 'Origin');
85             } else {
86                 $response->headers->set('Vary', $response->headers->get('Vary') . ', Origin');
87             }
88     
89             if ($this->options['supportsCredentials']) {
90                 $response->headers->set('Access-Control-Allow-Credentials', 'true');
91             }
92     
93             if ($this->options['exposedHeaders']) {
94                 $response->headers->set('Access-Control-Expose-Headers', implode(', ', $this->options['exposedHeaders']));
95             }
96     
97             return $response;
98         }

100         public function handlePreflightRequest(Request $request)
101         {
102             if (true !== $check = $this->checkPreflightRequestConditions($request)) {
103                 return $check;
104             }
105     
106             return $this->buildPreflightCheckResponse($request);
107         }

109         private function buildPreflightCheckResponse(Request $request)
110         {
111             $response = new Response();
112     
113             if ($this->options['supportsCredentials']) {
114                 $response->headers->set('Access-Control-Allow-Credentials', 'true');
115             }
116     
117             $response->headers->set('Access-Control-Allow-Origin', $request->headers->get('Origin'));
118     
119             if ($this->options['maxAge']) {
120                 $response->headers->set('Access-Control-Max-Age', $this->options['maxAge']);
121             }
122     
123             $allowMethods = $this->options['allowedMethods'] === true
124                 ? strtoupper($request->headers->get('Access-Control-Request-Method'))
125                 : implode(', ', $this->options['allowedMethods']);
126             $response->headers->set('Access-Control-Allow-Methods', $allowMethods);
127     
128             $allowHeaders = $this->options['allowedHeaders'] === true
129                 ? strtoupper($request->headers->get('Access-Control-Request-Headers'))
130                 : implode(', ', $this->options['allowedHeaders']);
131             $response->headers->set('Access-Control-Allow-Headers', $allowHeaders);
132     
133             $response->setStatusCode(204);
134     
135             return $response;
136         }

138         private function checkPreflightRequestConditions(Request $request)
139         {
140             if (!$this->checkOrigin($request)) {
141                 return $this->createBadRequestResponse(403, 'Origin not allowed');
142             }
143     
144             if (!$this->checkMethod($request)) {
145                 return $this->createBadRequestResponse(405, 'Method not allowed');
146             }
147     
148             $requestHeaders = array();
149             // if allowedHeaders has been set to true ('*' allow all flag) just skip this check
150             if ($this->options['allowedHeaders'] !== true && $request->headers->has('Access-Control-Request-Headers')) {
151                 $headers        = strtolower($request->headers->get('Access-Control-Request-Headers'));
152                 $requestHeaders = array_filter(explode(',', $headers));
153     
154                 foreach ($requestHeaders as $header) {
155                     if (!in_array(trim($header), $this->options['allowedHeaders'])) {
156                         return $this->createBadRequestResponse(403, 'Header not allowed');
157                     }
158                 }
159             }
160     
161             return true;
162         }

164         private function createBadRequestResponse($code, $reason = '')
165         {
166             return new Response($reason, $code);
167         }

169         private function isSameHost(Request $request)
170         {
171             return $request->headers->get('Origin') === $request->getSchemeAndHttpHost();
172         }
173     
174         private function checkOrigin(Request $request)
175         {
176             if ($this->options['allowedOrigins'] === true) {
177                 // allow all '*' flag
178                 return true;
179             }
180             $origin = $request->headers->get('Origin');
181     
182             if (in_array($origin, $this->options['allowedOrigins'])) {
183                 return true;
184             }
185     
186             foreach ($this->options['allowedOriginsPatterns'] as $pattern) {
187                 if (preg_match($pattern, $origin)) {
188                     return true;
189                 }
190             }
191     
192             return false;
193         }
194     
195         private function checkMethod(Request $request)
196         {
197             if ($this->options['allowedMethods'] === true) {
198                 // allow all '*' flag
199                 return true;
200             }
201     
202             $requestMethod = strtoupper($request->headers->get('Access-Control-Request-Method'));
203             return in_array($requestMethod, $this->options['allowedMethods']);
204         }
