48         /**
49          * Registers a new ErrorHandler for a given Logger
50          *
51          * By default it will handle errors, exceptions and fatal errors
52          *
53          * @param  LoggerInterface   $logger
54          * @param  array|false       $errorLevelMap     an array of E_* constant to LogLevel::* constant mapping, or false to disable error handling
55          * @param  array|false       $exceptionLevelMap an array of class name to LogLevel::* constant mapping, or false to disable exception handling
56          * @param  string|null|false $fatalLevel        a LogLevel::* constant, null to use the default LogLevel::ALERT or false to disable fatal error handling
57          * @return ErrorHandler
58          */
59         public static function register(LoggerInterface $logger, $errorLevelMap = [], $exceptionLevelMap = [], $fatalLevel = null): self
60         {
61             $handler = new static($logger);
62             if ($errorLevelMap !== false) {
63                 $handler->registerErrorHandler($errorLevelMap);
64             }
65             if ($exceptionLevelMap !== false) {
66                 $handler->registerExceptionHandler($exceptionLevelMap);
67             }
68             if ($fatalLevel !== false) {
69                 $handler->registerFatalHandler($fatalLevel);
70             }
71     
72             return $handler;
73         }

75         public function registerExceptionHandler($levelMap = [], $callPrevious = true): self
76         {
77             $prev = set_exception_handler([$this, 'handleException']);
78             $this->uncaughtExceptionLevelMap = $levelMap;
79             foreach ($this->defaultExceptionLevelMap() as $class => $level) {
80                 if (!isset($this->uncaughtExceptionLevelMap[$class])) {
81                     $this->uncaughtExceptionLevelMap[$class] = $level;
82                 }
83             }
84             if ($callPrevious && $prev) {
85                 $this->previousExceptionHandler = $prev;
86             }
87     
88             return $this;
89         }

91         public function registerErrorHandler(array $levelMap = [], $callPrevious = true, $errorTypes = -1, $handleOnlyReportedErrors = true): self
92         {
93             $prev = set_error_handler([$this, 'handleError'], $errorTypes);
94             $this->errorLevelMap = array_replace($this->defaultErrorLevelMap(), $levelMap);
95             if ($callPrevious) {
96                 $this->previousErrorHandler = $prev ?: true;
97             }
98     
99             $this->handleOnlyReportedErrors = $handleOnlyReportedErrors;
100     
101             return $this;
102         }

104         /**
105          * @param string|null $level              a LogLevel::* constant, null to use the default LogLevel::ALERT or false to disable fatal error handling
106          * @param int         $reservedMemorySize Amount of KBs to reserve in memory so that it can be freed when handling fatal errors giving Monolog some room in memory to get its job done
107          */
108         public function registerFatalHandler($level = null, int $reservedMemorySize = 20): self
109         {
110             register_shutdown_function([$this, 'handleFatalError']);
111     
112             $this->reservedMemory = str_repeat(' ', 1024 * $reservedMemorySize);
113             $this->fatalLevel = $level;
114             $this->hasFatalErrorHandler = true;
115     
116             return $this;
117         }

148         /**
149          * @private
150          * @param \Exception $e
151          */
152         public function handleException($e)
153         {
154             $level = LogLevel::ERROR;
155             foreach ($this->uncaughtExceptionLevelMap as $class => $candidate) {
156                 if ($e instanceof $class) {
157                     $level = $candidate;
158                     break;
159                 }
160             }
161     
162             $this->logger->log(
163                 $level,
164                 sprintf('Uncaught Exception %s: "%s" at %s line %s', Utils::getClass($e), $e->getMessage(), $e->getFile(), $e->getLine()),
165                 ['exception' => $e]
166             );
167     
168             if ($this->previousExceptionHandler) {
169                 call_user_func($this->previousExceptionHandler, $e);
170             }
171     
172             if (!headers_sent() && !ini_get('display_errors')) {
173                 http_response_code(500);
174             }
175     
176             exit(255);
177         }

179         /**
180          * @private
181          */
182         public function handleError($code, $message, $file = '', $line = 0, $context = [])
183         {
184             if ($this->handleOnlyReportedErrors && !(error_reporting() & $code)) {
185                 return;
186             }
187     
188             // fatal error codes are ignored if a fatal error handler is present as well to avoid duplicate log entries
189             if (!$this->hasFatalErrorHandler || !in_array($code, self::$fatalErrors, true)) {
190                 $level = $this->errorLevelMap[$code] ?? LogLevel::CRITICAL;
191                 $this->logger->log($level, self::codeToString($code).': '.$message, ['code' => $code, 'message' => $message, 'file' => $file, 'line' => $line]);
192             } else {
193                 $trace = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS);
194                 array_shift($trace); // Exclude handleError from trace
195                 $this->lastFatalTrace = $trace;
196             }
197     
198             if ($this->previousErrorHandler === true) {
199                 return false;
200             } elseif ($this->previousErrorHandler) {
201                 return call_user_func($this->previousErrorHandler, $code, $message, $file, $line, $context);
202             }
203     
204             return true;
205         }

207         /**
208          * @private
209          */
210         public function handleFatalError()
211         {
212             $this->reservedMemory = '';
213     
214             $lastError = error_get_last();
215             if ($lastError && in_array($lastError['type'], self::$fatalErrors, true)) {
216                 $this->logger->log(
217                     $this->fatalLevel === null ? LogLevel::ALERT : $this->fatalLevel,
218                     'Fatal Error ('.self::codeToString($lastError['type']).'): '.$lastError['message'],
219                     ['code' => $lastError['type'], 'message' => $lastError['message'], 'file' => $lastError['file'], 'line' => $lastError['line'], 'trace' => $this->lastFatalTrace]
220                 );
221     
222                 if ($this->logger instanceof Logger) {
223                     foreach ($this->logger->getHandlers() as $handler) {
224                         $handler->close();
225                     }
226                 }
227             }
228         }
