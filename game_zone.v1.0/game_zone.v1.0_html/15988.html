135         /**
136          * Returns the cookie as a string.
137          *
138          * @return string The cookie
139          */
140         public function __toString()
141         {
142             if ($this->isRaw()) {
143                 $str = $this->getName();
144             } else {
145                 $str = str_replace(self::$reservedCharsFrom, self::$reservedCharsTo, $this->getName());
146             }
147     
148             $str .= '=';
149     
150             if ('' === (string) $this->getValue()) {
151                 $str .= 'deleted; expires='.gmdate('D, d-M-Y H:i:s T', time() - 31536001).'; Max-Age=0';
152             } else {
153                 $str .= $this->isRaw() ? $this->getValue() : rawurlencode($this->getValue());
154     
155                 if (0 !== $this->getExpiresTime()) {
156                     $str .= '; expires='.gmdate('D, d-M-Y H:i:s T', $this->getExpiresTime()).'; Max-Age='.$this->getMaxAge();
157                 }
158             }
159     
160             if ($this->getPath()) {
161                 $str .= '; path='.$this->getPath();
162             }
163     
164             if ($this->getDomain()) {
165                 $str .= '; domain='.$this->getDomain();
166             }
167     
168             if (true === $this->isSecure()) {
169                 $str .= '; secure';
170             }
171     
172             if (true === $this->isHttpOnly()) {
173                 $str .= '; httponly';
174             }
175     
176             if (null !== $this->getSameSite()) {
177                 $str .= '; samesite='.$this->getSameSite();
178             }
179     
180             return $str;
181         }
