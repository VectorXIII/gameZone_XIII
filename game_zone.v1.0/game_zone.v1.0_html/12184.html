21     namespace Mockery;
22     
23     class Configuration
24     {
25         /**
26          * Boolean assertion of whether we can mock methods which do not actually
27          * exist for the given class or object (ignored for unreal mocks)
28          *
29          * @var bool
30          */
31         protected $_allowMockingNonExistentMethod = true;
32     
33         /**
34          * Boolean assertion of whether we ignore unnecessary mocking of methods,
35          * i.e. when method expectations are made, set using a zeroOrMoreTimes()
36          * constraint, and then never called. Essentially such expectations are
37          * not required and are just taking up test space.
38          *
39          * @var bool
40          */
41         protected $_allowMockingMethodsUnnecessarily = true;
42     
43         /**
44          * Parameter map for use with PHP internal classes.
45          *
46          * @var array
47          */
48         protected $_internalClassParamMap = array();
49     
50         protected $_constantsMap = array();
51     
52         /**
53          * Boolean assertion is reflection caching enabled or not. It should be
54          * always enabled, except when using PHPUnit's --static-backup option.
55          *
56          * @see https://github.com/mockery/mockery/issues/268
57          */
58         protected $_reflectionCacheEnabled = true;
59     
60         /**
61          * Set boolean to allow/prevent mocking of non-existent methods
62          *
63          * @param bool $flag
64          */
65         public function allowMockingNonExistentMethods($flag = true)
66         {
67             $this->_allowMockingNonExistentMethod = (bool) $flag;
68         }
69     
70         /**
71          * Return flag indicating whether mocking non-existent methods allowed
72          *
73          * @return bool
74          */
75         public function mockingNonExistentMethodsAllowed()
76         {
77             return $this->_allowMockingNonExistentMethod;
78         }
79     
80         /**
81          * @deprecated
82          *
83          * Set boolean to allow/prevent unnecessary mocking of methods
84          *
85          * @param bool $flag
86          */
87         public function allowMockingMethodsUnnecessarily($flag = true)
88         {
89             trigger_error(sprintf("The %s method is deprecated and will be removed in a future version of Mockery", __METHOD__), E_USER_DEPRECATED);
90     
91             $this->_allowMockingMethodsUnnecessarily = (bool) $flag;
92         }
93     
94         /**
95          * Return flag indicating whether mocking non-existent methods allowed
96          *
97          * @return bool
98          */
99         public function mockingMethodsUnnecessarilyAllowed()
100         {
101             trigger_error(sprintf("The %s method is deprecated and will be removed in a future version of Mockery", __METHOD__), E_USER_DEPRECATED);
102     
103             return $this->_allowMockingMethodsUnnecessarily;
104         }
105     
106         /**
107          * Set a parameter map (array of param signature strings) for the method
108          * of an internal PHP class.
109          *
110          * @param string $class
111          * @param string $method
112          * @param array $map
113          */
114         public function setInternalClassMethodParamMap($class, $method, array $map)
115         {
116             if (!isset($this->_internalClassParamMap[strtolower($class)])) {
117                 $this->_internalClassParamMap[strtolower($class)] = array();
118             }
119             $this->_internalClassParamMap[strtolower($class)][strtolower($method)] = $map;
120         }
121     
122         /**
123          * Remove all overriden parameter maps from internal PHP classes.
124          */
125         public function resetInternalClassMethodParamMaps()
126         {
127             $this->_internalClassParamMap = array();
128         }
129     
130         /**
131          * Get the parameter map of an internal PHP class method
132          *
133          * @return array
134          */
135         public function getInternalClassMethodParamMap($class, $method)
136         {
137             if (isset($this->_internalClassParamMap[strtolower($class)][strtolower($method)])) {
138                 return $this->_internalClassParamMap[strtolower($class)][strtolower($method)];
139             }
140         }
141     
142         public function getInternalClassMethodParamMaps()
143         {
144             return $this->_internalClassParamMap;
145         }
146     
147         public function setConstantsMap(array $map)
148         {
149             $this->_constantsMap = $map;
150         }
151     
152         public function getConstantsMap()
153         {
154             return $this->_constantsMap;
155         }
156     
157         /**
158          * Disable reflection caching
159          *
160          * It should be always enabled, except when using
161          * PHPUnit's --static-backup option.
162          *
163          * @see https://github.com/mockery/mockery/issues/268
164          */
165         public function disableReflectionCache()
166         {
167             $this->_reflectionCacheEnabled = false;
168         }
169     
170         /**
171          * Enable reflection caching
172          *
173          * It should be always enabled, except when using
174          * PHPUnit's --static-backup option.
175          *
176          * @see https://github.com/mockery/mockery/issues/268
177          */
178         public function enableReflectionCache()
179         {
180             $this->_reflectionCacheEnabled = true;
181         }
182     
183         /**
184          * Is reflection cache enabled?
185          */
186         public function reflectionCacheEnabled()
187         {
188             return $this->_reflectionCacheEnabled;
189         }
