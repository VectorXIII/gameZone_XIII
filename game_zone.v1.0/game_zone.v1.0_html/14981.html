19     /**
20      * This class contains the basic implementation of a collection that does not
21      * allow duplicated values (a set), to minimize the effort required to implement
22      * this specific type of collection.
23      */
24     abstract class AbstractSet extends AbstractCollection
25     {
26         /**
27          * Adds the specified element to this set, if it is not already present.
28          *
29          * @param mixed $element The element to add to the set.
30          *
31          * @return bool `true` if this set did not already contain the specified
32          *     element.
33          *
34          * @throws InvalidArgumentException when the element does not match the
35          *     specified type for this set.
36          */
37         public function add($element): bool
38         {
39             if ($this->contains($element)) {
40                 return false;
41             }
42     
43             return parent::add($element);
44         }
45     
46         /**
47          * Sets the given value to the given offset in this set, if it is not
48          * already present.
49          *
50          * @param mixed|null $offset The offset is ignored and is treated as `null`.
51          * @param mixed $value The value to set at the given offset.
52          *
53          * @throws InvalidArgumentException when the value does not match the
54          *     specified type for this set.
55          */
56         public function offsetSet($offset, $value): void
57         {
58             if ($this->contains($value)) {
59                 return;
60             }
61
