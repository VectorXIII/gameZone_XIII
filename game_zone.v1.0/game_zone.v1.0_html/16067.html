46         /**
47          * Warms up the cache.
48          */
49         public function warmUp(string $cacheDir)
50         {
51             if ($collectDeprecations = $this->debug && !\defined('PHPUNIT_COMPOSER_INSTALL')) {
52                 $collectedLogs = [];
53                 $previousHandler = set_error_handler(function ($type, $message, $file, $line) use (&$collectedLogs, &$previousHandler) {
54                     if (E_USER_DEPRECATED !== $type && E_DEPRECATED !== $type) {
55                         return $previousHandler ? $previousHandler($type, $message, $file, $line) : false;
56                     }
57     
58                     if (isset($collectedLogs[$message])) {
59                         ++$collectedLogs[$message]['count'];
60     
61                         return null;
62                     }
63     
64                     $backtrace = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 3);
65                     // Clean the trace by removing first frames added by the error handler itself.
66                     for ($i = 0; isset($backtrace[$i]); ++$i) {
67                         if (isset($backtrace[$i]['file'], $backtrace[$i]['line']) && $backtrace[$i]['line'] === $line && $backtrace[$i]['file'] === $file) {
68                             $backtrace = \array_slice($backtrace, 1 + $i);
69                             break;
70                         }
71                     }
72     
73                     $collectedLogs[$message] = [
74                         'type' => $type,
75                         'message' => $message,
76                         'file' => $file,
77                         'line' => $line,
78                         'trace' => $backtrace,
79                         'count' => 1,
80                     ];
81     
82                     return null;
83                 });
84             }
85     
86             try {
87                 foreach ($this->warmers as $warmer) {
88                     if (!$this->optionalsEnabled && $warmer->isOptional()) {
89                         continue;
90                     }
91                     if ($this->onlyOptionalsEnabled && !$warmer->isOptional()) {
92                         continue;
93                     }
94     
95                     $warmer->warmUp($cacheDir);
96                 }
97             } finally {
98                 if ($collectDeprecations) {
99                     restore_error_handler();
100     
101                     if (file_exists($this->deprecationLogsFilepath)) {
102                         $previousLogs = unserialize(file_get_contents($this->deprecationLogsFilepath));
103                         $collectedLogs = array_merge($previousLogs, $collectedLogs);
104                     }
105     
106                     file_put_contents($this->deprecationLogsFilepath, serialize(array_values($collectedLogs)));
107                 }
108             }
109         }
