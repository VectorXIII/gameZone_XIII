194         /**
195          * Fire an event and call the listeners.
196          *
197          * @param  string|object  $event
198          * @param  mixed  $payload
199          * @param  bool  $halt
200          * @return array|null
201          */
202         public function dispatch($event, $payload = [], $halt = false)
203         {
204             // When the given "event" is actually an object we will assume it is an event
205             // object and use the class as the event name and this event itself as the
206             // payload to the handler, which makes object based events quite simple.
207             [$event, $payload] = $this->parseEventAndPayload(
208                 $event, $payload
209             );
210     
211             if ($this->shouldBroadcast($payload)) {
212                 $this->broadcastEvent($payload[0]);
213             }
214     
215             $responses = [];
216     
217             foreach ($this->getListeners($event) as $listener) {
218                 $response = $listener($event, $payload);
219     
220                 // If a response is returned from the listener and event halting is enabled
221                 // we will just return this response, and not call the rest of the event
222                 // listeners. Otherwise we will add the response on the response list.
223                 if ($halt && ! is_null($response)) {
224                     return $response;
225                 }
226     
227                 // If a boolean false is returned from a listener, we will stop propagating
228                 // the event to any further listeners down in the chain, else we keep on
229                 // looping through the listeners and firing every one in our sequence.
230                 if ($response === false) {
231                     break;
232                 }
233     
234                 $responses[] = $response;
235             }
236     
237             return $halt ? null : $responses;
238         }

477          * Queue the handler class.
478          *
479          * @param  string  $class
480          * @param  string  $method
481          * @param  array  $arguments
482          * @return void
483          */
484         protected function queueHandler($class, $method, $arguments)
485         {
486             [$listener, $job] = $this->createListenerAndJob($class, $method, $arguments);
487     
488             $connection = $this->resolveQueue()->connection(
489                 $listener->connection ?? null
490             );
491     
492             $queue = $listener->queue ?? null;
493     
494             isset($listener->delay)
495                         ? $connection->laterOn($queue, $listener->delay, $job)
496                         : $connection->pushOn($queue, $job);
497         }
