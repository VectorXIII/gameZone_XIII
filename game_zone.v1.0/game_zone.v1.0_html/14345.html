156         protected function writeTestResult(array $prevResult, array $result): void
157         {
158             // spacer line for new suite headers and after verbose messages
159             if ($prevResult['testName'] !== '' &&
160                 (!empty($prevResult['message']) || $prevResult['className'] !== $result['className'])) {
161                 $this->write(\PHP_EOL);
162             }
163     
164             // suite header
165             if ($prevResult['className'] !== $result['className']) {
166                 $this->write($this->colorizeTextBox('underlined', $result['className']) . \PHP_EOL);
167             }
168     
169             // test result line
170             if ($this->colors && $result['className'] === PhptTestCase::class) {
171                 $testName = Color::colorizePath($result['testName'], $prevResult['testName'], true);
172             } else {
173                 $testName = $result['testMethod'];
174             }
175     
176             $style = self::STATUS_STYLES[$result['status']];
177             $line  = \sprintf(
178                 ' %s %s%s' . \PHP_EOL,
179                 $this->colorizeTextBox($style['color'], $style['symbol']),
180                 $testName,
181                 $this->verbose ? ' ' . $this->formatRuntime($result['time'], $style['color']) : ''
182             );
183     
184             $this->write($line);
185     
186             // additional information when verbose
187             $this->write($result['message']);
188         }

195         protected function colorizeMessageAndDiff(string $style, string $buffer): array
196         {
197             $lines      = $buffer ? \array_map('\rtrim', \explode(\PHP_EOL, $buffer)) : [];
198             $message    = [];
199             $diff       = [];
200             $insideDiff = false;
201     
202             foreach ($lines as $line) {
203                 if ($line === '--- Expected') {
204                     $insideDiff = true;
205                 }
206     
207                 if (!$insideDiff) {
208                     $message[] = $line;
209                 } else {
210                     if (\strpos($line, '-') === 0) {
211                         $line = Color::colorize('fg-red', Color::visualizeWhitespace($line, true));
212                     } elseif (\strpos($line, '+') === 0) {
213                         $line = Color::colorize('fg-green', Color::visualizeWhitespace($line, true));
214                     } elseif ($line === '@@ @@') {
215                         $line = Color::colorize('fg-cyan', $line);
216                     }
217                     $diff[] = $line;
218                 }
219             }
220             $diff = \implode(\PHP_EOL, $diff);
221     
222             if (!empty($message)) {
223                 $message = $this->colorizeTextBox($style, \implode(\PHP_EOL, $message));
224             }
225     
226             return [$message, $diff];
227         }

229         protected function formatStacktrace(\Throwable $t): string
230         {
231             $trace = \PHPUnit\Util\Filter::getFilteredStacktrace($t);
232     
233             if (!$this->colors) {
234                 return $trace;
235             }
236     
237             $lines    = [];
238             $prevPath = '';
239     
240             foreach (\explode(\PHP_EOL, $trace) as $line) {
241                 if (\preg_match('/^(.*):(\d+)$/', $line, $matches)) {
242                     $lines[] = Color::colorizePath($matches[1], $prevPath) .
243                         Color::dim(':') .
244                         Color::colorize('fg-blue', $matches[2]) .
245                         "\n";
246                     $prevPath = $matches[1];
247                 } else {
248                     $lines[]  = $line;
249                     $prevPath = '';
250                 }
251             }
252     
253             return \implode('', $lines);
254         }

256         protected function formatTestResultMessage(\Throwable $t, array $result, ?string $prefix = null): string
257         {
258             $message = $this->formatThrowable($t, $result['status']);
259             $diff    = '';
260     
261             if (!($this->verbose || $result['verbose'])) {
262                 return '';
263             }
264     
265             if ($message && $this->colors) {
266                 $style            = self::STATUS_STYLES[$result['status']]['message'] ?? '';
267                 [$message, $diff] = $this->colorizeMessageAndDiff($style, $message);
268             }
269     
270             if ($prefix === null || !$this->colors) {
271                 $prefix = self::PREFIX_SIMPLE;
272             }
273     
274             if ($this->colors) {
275                 $color  = self::STATUS_STYLES[$result['status']]['color'] ?? '';
276                 $prefix = \array_map(static function ($p) use ($color) {
277                     return Color::colorize($color, $p);
278                 }, self::PREFIX_DECORATED);
279             }
280     
281             $trace = $this->formatStacktrace($t);
282             $out   = $this->prefixLines($prefix['start'], \PHP_EOL) . \PHP_EOL;
283     
284             if ($message) {
285                 $out .= $this->prefixLines($prefix['message'], $message . \PHP_EOL) . \PHP_EOL;
286             }
287     
288             if ($diff) {
289                 $out .= $this->prefixLines($prefix['diff'], $diff . \PHP_EOL) . \PHP_EOL;
290             }
291     
292             if ($trace) {
293                 if ($message || $diff) {
294                     $out .= $this->prefixLines($prefix['default'], \PHP_EOL) . \PHP_EOL;
295                 }
296                 $out .= $this->prefixLines($prefix['trace'], $trace . \PHP_EOL) . \PHP_EOL;
297             }
298             $out .= $this->prefixLines($prefix['last'], \PHP_EOL) . \PHP_EOL;
299     
300             return $out;
301         }

319         private function formatRuntime(float $time, string $color = ''): string
320         {
321             if (!$this->colors) {
322                 return \sprintf('[%.2f ms]', $time * 1000);
323             }
324     
325             if ($time > 1) {
326                 $color = 'fg-magenta';
327             }
328     
329             return Color::colorize($color, ' ' . (int) \ceil($time * 1000) . ' ' . Color::dim('ms'));

332         private function printNonSuccessfulTestsSummary(int $numberOfExecutedTests): void
333         {
334             if (empty($this->nonSuccessfulTestResults)) {
335                 return;
336             }
337     
338             if ((\count($this->nonSuccessfulTestResults) / $numberOfExecutedTests) >= 0.7) {
339                 return;
340             }
341     
342             $this->write("Summary of non-successful tests:\n\n");
343     
344             $prevResult = $this->getEmptyTestResult();
345     
346             foreach ($this->nonSuccessfulTestResults as $testIndex) {
347                 $result = $this->testResults[$testIndex];
348                 $this->writeTestResult($prevResult, $result);
349                 $prevResult = $result;
