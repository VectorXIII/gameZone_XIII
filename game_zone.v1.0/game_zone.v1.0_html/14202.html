10     namespace PHPUnit\Framework;
11     
12     use PHPUnit\Util\Filter;

14     /**
15      * Base class for all PHPUnit Framework exceptions.
16      *
17      * Ensures that exceptions thrown during a test run do not leave stray
18      * references behind.
19      *
20      * Every Exception contains a stack trace. Each stack frame contains the 'args'
21      * of the called function. The function arguments can contain references to
22      * instantiated objects. The references prevent the objects from being
23      * destructed (until test results are eventually printed), so memory cannot be
24      * freed up.
25      *
26      * With enabled process isolation, test results are serialized in the child
27      * process and unserialized in the parent process. The stack trace of Exceptions
28      * may contain objects that cannot be serialized or unserialized (e.g., PDO
29      * connections). Unserializing user-space objects from the child process into
30      * the parent would break the intended encapsulation of process isolation.
31      *
32      * @see http://fabien.potencier.org/article/9/php-serialization-stack-traces-and-exceptions
33      *
34      * @internal This class is not covered by the backward compatibility promise for PHPUnit
35      */
36     class Exception extends \RuntimeException implements \PHPUnit\Exception
37     {
38         /**
39          * @var array
40          */
41         protected $serializableTrace;
42     
43         public function __construct($message = '', $code = 0, \Throwable $previous = null)
44         {
45             parent::__construct($message, $code, $previous);
46     
47             $this->serializableTrace = $this->getTrace();
48     
49             foreach (\array_keys($this->serializableTrace) as $key) {
50                 unset($this->serializableTrace[$key]['args']);
51             }
52         }
53     
54         public function __toString(): string
55         {
56             $string = TestFailure::exceptionToString($this);
57     
58             if ($trace = Filter::getFilteredStacktrace($this)) {
59                 $string .= "\n" . $trace;
60             }
61     
62             return $string;
63         }
64     
65         public function __sleep(): array
66         {
67             return \array_keys(\get_object_vars($this));
68         }
69     
70         /**
71          * Returns the serializable trace (without 'args').
72          */
73         public function getSerializableTrace(): array
74         {
75             return $this->serializableTrace;
76         }
77     }
