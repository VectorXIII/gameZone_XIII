309          * Create a new CarbonInterval instance.
310          *
311          * @param int|null $years
312          * @param int|null $months
313          * @param int|null $weeks
314          * @param int|null $days
315          * @param int|null $hours
316          * @param int|null $minutes
317          * @param int|null $seconds
318          * @param int|null $microseconds
319          *
320          * @throws Exception when the interval_spec (passed as $years) cannot be parsed as an interval.
321          */
322         public function __construct($years = 1, $months = null, $weeks = null, $days = null, $hours = null, $minutes = null, $seconds = null, $microseconds = null)
323         {
324             if ($years instanceof DateInterval) {
325                 parent::__construct(static::getDateIntervalSpec($years));
326                 $this->f = $years->f;
327                 static::copyNegativeUnits($years, $this);
328     
329                 return;
330             }
331     
332             $spec = $years;
333     
334             if (!is_string($spec) || floatval($years) || preg_match('/^[0-9.]/', $years)) {
335                 $spec = static::PERIOD_PREFIX;
336     
337                 $spec .= $years > 0 ? $years.static::PERIOD_YEARS : '';
338                 $spec .= $months > 0 ? $months.static::PERIOD_MONTHS : '';
339     
340                 $specDays = 0;
341                 $specDays += $weeks > 0 ? $weeks * static::getDaysPerWeek() : 0;
342                 $specDays += $days > 0 ? $days : 0;
343     
344                 $spec .= $specDays > 0 ? $specDays.static::PERIOD_DAYS : '';
345     
346                 if ($hours > 0 || $minutes > 0 || $seconds > 0) {
347                     $spec .= static::PERIOD_TIME_PREFIX;
348                     $spec .= $hours > 0 ? $hours.static::PERIOD_HOURS : '';
349                     $spec .= $minutes > 0 ? $minutes.static::PERIOD_MINUTES : '';
350                     $spec .= $seconds > 0 ? $seconds.static::PERIOD_SECONDS : '';
351                 }
352     
353                 if ($spec === static::PERIOD_PREFIX) {
354                     // Allow the zero interval.
355                     $spec .= '0'.static::PERIOD_YEARS;
356                 }
357             }
358     
359             parent::__construct($spec);
360     
361             if (!is_null($microseconds)) {
362                 $this->f = $microseconds / Carbon::MICROSECONDS_PER_SECOND;
363             }
364         }

473         /**
474          * Parse a string into a new CarbonInterval object according to the specified format.
475          *
476          * @example
477          * ```
478          * echo Carboninterval::createFromFormat('H:i', '1:30');
479          * ```
480          *
481          * @param string $format   Format of the $interval input string
482          * @param string $interval Input string to convert into an interval
483          *
484          * @throws Exception when the $interval cannot be parsed as an interval.
485          *
486          * @return static
487          */
488         public static function createFromFormat(string $format, ?string $interval)
489         {
490             $instance = new static(0);
491             $length = mb_strlen($format);
492     
493             if (preg_match('/s([,.])([uv])$/', $format, $match)) {
494                 $interval = explode($match[1], $interval);
495                 $index = count($interval) - 1;
496                 $interval[$index] = str_pad($interval[$index], $match[2] === 'v' ? 3 : 6, '0');
497                 $interval = implode($match[1], $interval);
498             }
499     
500             for ($index = 0; $index < $length; $index++) {
501                 $expected = mb_substr($format, $index, 1);
502                 $nextCharacter = mb_substr($interval, 0, 1);
503                 $unit = static::$formats[$expected] ?? null;
504     
505                 if ($unit) {
506                     if (!preg_match('/^-?\d+/', $interval, $match)) {
507                         throw new ParseErrorException('number', $nextCharacter);
508                     }
509     
510                     $interval = mb_substr($interval, mb_strlen($match[0]));
511                     $instance->$unit += intval($match[0]);
512     
513                     continue;
514                 }
515     
516                 if ($nextCharacter !== $expected) {
517                     throw new ParseErrorException(
518                         "'$expected'",
519                         $nextCharacter,
520                         'Allowed substitutes for interval formats are '.implode(', ', array_keys(static::$formats))."\n".
521                         'See https://www.php.net/manual/en/function.date.php for their meaning'
522                     );
523                 }
524     
525                 $interval = mb_substr($interval, 1);
526             }
527     
528             if ($interval !== '') {
529                 throw new ParseErrorException(
530                     'end of string',
531                     $interval
532                 );
533             }
534     
535             return $instance;
536         }

1373                 }

1377                     return implode($glue, $list);
1378                 };
1379             }
1380     
1381             $interpolations = [
1382                 ':optional-space' => $optionalSpace,
1383             ];
1384     
1385             return [$syntax, $short, $parts, $options, $join, $aUnit, $altNumbers, $interpolations, $minimumUnit];

1471         /**
1472          * Get the current interval in a human readable format in the current locale.
1473          *
1474          * @example
1475          * ```
1476          * echo CarbonInterval::fromString('4d 3h 40m')->forHumans() . "\n";
1477          * echo CarbonInterval::fromString('4d 3h 40m')->forHumans(['parts' => 2]) . "\n";
1478          * echo CarbonInterval::fromString('4d 3h 40m')->forHumans(['parts' => 3, 'join' => true]) . "\n";
1479          * echo CarbonInterval::fromString('4d 3h 40m')->forHumans(['short' => true]) . "\n";
1480          * echo CarbonInterval::fromString('1d 24h')->forHumans(['join' => ' or ']) . "\n";
1481          * echo CarbonInterval::fromString('1d 24h')->forHumans(['minimumUnit' => 'hour']) . "\n";
1482          * ```
1483          *
1484          * @param int|array $syntax  if array passed, parameters will be extracted from it, the array may contains:
1485          *                           - 'syntax' entry (see below)
1486          *                           - 'short' entry (see below)
1487          *                           - 'parts' entry (see below)
1488          *                           - 'options' entry (see below)
1489          *                           - 'aUnit' entry, prefer "an hour" over "1 hour" if true
1490          *                           - 'join' entry determines how to join multiple parts of the string
1491          *                           `  - if $join is a string, it's used as a joiner glue
1492          *                           `  - if $join is a callable/closure, it get the list of string and should return a string
1493          *                           `  - if $join is an array, the first item will be the default glue, and the second item
1494          *                           `    will be used instead of the glue for the last item
1495          *                           `  - if $join is true, it will be guessed from the locale ('list' translation file entry)
1496          *                           `  - if $join is missing, a space will be used as glue
1497          *                           - 'minimumUnit' entry determines the smallest unit of time to display can be long or
1498          *                           `  short form of the units, e.g. 'hour' or 'h' (default value: s)
1499          *                           if int passed, it add modifiers:
1500          *                           Possible values:
1501          *                           - CarbonInterface::DIFF_ABSOLUTE          no modifiers
1502          *                           - CarbonInterface::DIFF_RELATIVE_TO_NOW   add ago/from now modifier
1503          *                           - CarbonInterface::DIFF_RELATIVE_TO_OTHER add before/after modifier
1504          *                           Default value: CarbonInterface::DIFF_ABSOLUTE
1505          * @param bool      $short   displays short format of time units
1506          * @param int       $parts   maximum number of parts to display (default value: -1: no limits)
1507          * @param int       $options human diff options
1508          *
1509          * @return string
1510          */
1511         public function forHumans($syntax = null, $short = false, $parts = -1, $options = null)
1512         {
1513             [$syntax, $short, $parts, $options, $join, $aUnit, $altNumbers, $interpolations, $minimumUnit] = $this->getForHumansParameters($syntax, $short, $parts, $options);
1514     
1515             $interval = [];
1516     
1517             $syntax = (int) ($syntax === null ? CarbonInterface::DIFF_ABSOLUTE : $syntax);
1518             $absolute = $syntax === CarbonInterface::DIFF_ABSOLUTE;
1519             $relativeToNow = $syntax === CarbonInterface::DIFF_RELATIVE_TO_NOW;
1520             $count = 1;
1521             $unit = $short ? 's' : 'second';
1522             $isFuture = $this->invert === 1;
1523             $transId = $relativeToNow ? ($isFuture ? 'from_now' : 'ago') : ($isFuture ? 'after' : 'before');
1524     
1525             /** @var \Symfony\Component\Translation\Translator $translator */
1526             $translator = $this->getLocalTranslator();
1527     
1528             $handleDeclensions = function ($unit, $count) use ($interpolations, $transId, $translator, $altNumbers, $absolute) {
1529                 if (!$absolute) {
1530                     // Some languages have special pluralization for past and future tense.
1531                     $key = $unit.'_'.$transId;
1532                     $result = $this->translate($key, $interpolations, $count, $translator, $altNumbers);
1533     
1534                     if ($result !== $key) {
1535                         return $result;
1536                     }
1537                 }
1538     
1539                 $result = $this->translate($unit, $interpolations, $count, $translator, $altNumbers);
1540     
1541                 if ($result !== $unit) {
1542                     return $result;
1543                 }
1544     
1545                 return null;
1546             };
1547     
1548             $intervalValues = $this;
1549             $method = static::getRoundingMethodFromOptions($options);
1550     
1551             if ($method) {
1552                 while (
1553                     count($intervalValues->getNonZeroValues()) > $parts &&
1554                     ($count = count($keys = array_keys($intervalValues->getValuesSequence()))) > 1
1555                 ) {
1556                     $intervalValues = $this->copy()->roundUnit($keys[$count - 2], 1, $method);
1557                 }
1558             }
1559     
1560             $diffIntervalArray = [
1561                 ['value' => $intervalValues->years,            'unit' => 'year',        'unitShort' => 'y'],
1562                 ['value' => $intervalValues->months,           'unit' => 'month',       'unitShort' => 'm'],
1563                 ['value' => $intervalValues->weeks,            'unit' => 'week',        'unitShort' => 'w'],
1564                 ['value' => $intervalValues->daysExcludeWeeks, 'unit' => 'day',         'unitShort' => 'd'],
1565                 ['value' => $intervalValues->hours,            'unit' => 'hour',        'unitShort' => 'h'],
1566                 ['value' => $intervalValues->minutes,          'unit' => 'minute',      'unitShort' => 'min'],
1567                 ['value' => $intervalValues->seconds,          'unit' => 'second',      'unitShort' => 's'],
1568                 ['value' => $intervalValues->milliseconds,     'unit' => 'millisecond', 'unitShort' => 'ms'],
1569             ];
1570     
1571             $transChoice = function ($short, $unitData) use ($absolute, $handleDeclensions, $translator, $aUnit, $altNumbers, $interpolations) {
1572                 $count = $unitData['value'];
1573     
1574                 if ($short) {
1575                     $result = $handleDeclensions($unitData['unitShort'], $count);
1576     
1577                     if ($result !== null) {
1578                         return $result;
1579                     }
1580                 } elseif ($aUnit) {
1581                     $result = $handleDeclensions('a_'.$unitData['unit'], $count);
1582     
1583                     if ($result !== null) {
1584                         return $result;
1585                     }
1586                 }
1587     
1588                 if (!$absolute) {
1589                     return $handleDeclensions($unitData['unit'], $count);
1590                 }
1591     
1592                 return $this->translate($unitData['unit'], $interpolations, $count, $translator, $altNumbers);
1593             };
1594     
1595             $fallbackUnit = ['second', 's'];
1596             foreach ($diffIntervalArray as $diffIntervalData) {
1597                 if ($diffIntervalData['value'] > 0) {
1598                     $unit = $short ? $diffIntervalData['unitShort'] : $diffIntervalData['unit'];
1599                     $count = $diffIntervalData['value'];
1600                     $interval[] = $transChoice($short, $diffIntervalData);
1601                 } elseif ($options & CarbonInterface::SEQUENTIAL_PARTS_ONLY && count($interval) > 0) {
1602                     break;
1603                 }
1604     
1605                 // break the loop after we get the required number of parts in array
1606                 if (count($interval) >= $parts) {
1607                     break;
1608                 }
1609     
1610                 // break the loop after we have reached the minimum unit
1611                 if (in_array($minimumUnit, [$diffIntervalData['unit'], $diffIntervalData['unitShort']])) {
1612                     $fallbackUnit = [$diffIntervalData['unit'], $diffIntervalData['unitShort']];
1613     
1614                     break;
1615                 }
1616             }
1617     
1618             if (count($interval) === 0) {
1619                 if ($relativeToNow && $options & CarbonInterface::JUST_NOW) {
1620                     $key = 'diff_now';
1621                     $translation = $this->translate($key, $interpolations, null, $translator);
1622     
1623                     if ($translation !== $key) {
1624                         return $translation;
1625                     }
1626                 }
1627     
1628                 $count = $options & CarbonInterface::NO_ZERO_DIFF ? 1 : 0;
1629                 $unit = $fallbackUnit[$short ? 1 : 0];
1630                 $interval[] = $this->translate($unit, $interpolations, $count, $translator, $altNumbers);
1631             }
1632     
1633             // join the interval parts by a space
1634             $time = $join($interval);
1635     
1636             unset($diffIntervalArray, $interval);
1637     
1638             if ($absolute) {
1639                 return $time;
1640             }
1641     
1642             $isFuture = $this->invert === 1;
1643     
1644             $transId = $relativeToNow ? ($isFuture ? 'from_now' : 'ago') : ($isFuture ? 'after' : 'before');
1645     
1646             if ($parts === 1) {
1647                 if ($relativeToNow && $unit === 'day') {
1648                     if ($count === 1 && $options & CarbonInterface::ONE_DAY_WORDS) {
1649                         $key = $isFuture ? 'diff_tomorrow' : 'diff_yesterday';
1650                         $translation = $this->translate($key, $interpolations, null, $translator);
1651     
1652                         if ($translation !== $key) {
1653                             return $translation;
1654                         }
1655                     }
1656     
1657                     if ($count === 2 && $options & CarbonInterface::TWO_DAY_WORDS) {
1658                         $key = $isFuture ? 'diff_after_tomorrow' : 'diff_before_yesterday';
1659                         $translation = $this->translate($key, $interpolations, null, $translator);
1660     
1661                         if ($translation !== $key) {
1662                             return $translation;
1663                         }
1664                     }
1665                 }
1666     
1667                 $aTime = $aUnit ? $handleDeclensions('a_'.$unit, $count) : null;
1668     
1669                 $time = $aTime ?: $handleDeclensions($unit, $count) ?: $time;
1670             }
1671     
1672             $time = [':time' => $time];
1673     
1674             return $this->translate($transId, array_merge($time, $interpolations, $time), null, $translator);
1675         }

2016         /**
2017          * Convert overflowed values into bigger units.
2018          *
2019          * @return $this
2020          */
2021         public function cascade()
2022         {
2023             foreach (static::getFlipCascadeFactors() as $source => [$target, $factor]) {
2024                 if ($source === 'dayz' && $target === 'weeks') {
2025                     continue;
2026                 }
2027     
2028                 $value = $this->$source;
2029                 $this->$source = $modulo = ($factor + ($value % $factor)) % $factor;
2030                 $this->$target += ($value - $modulo) / $factor;
2031                 if ($this->$source > 0 && $this->$target < 0) {
2032                     $this->$source -= $factor;
2033                     $this->$target++;
2034                 }
2035             }

2047          * @return float
2048          */

2050         {

2052     
2053             if (in_array($unit, ['days', 'weeks'])) {
2054                 $realUnit = 'dayz';
2055             } elseif (!in_array($unit, ['microseconds', 'milliseconds', 'seconds', 'minutes', 'hours', 'dayz', 'months', 'years'])) {
2056                 throw new InvalidArgumentException("Unknown unit '$unit'.");
2057             }
2058     
2059             $result = 0;
2060             $cumulativeFactor = 0;
2061             $unitFound = false;
2062             $factors = static::getFlipCascadeFactors();
2063             $daysPerWeek = static::getDaysPerWeek();
2064     
2065             $values = [
2066                 'years' => $this->years,
2067                 'months' => $this->months,
2068                 'weeks' => (int) floor($this->d / $daysPerWeek),
2069                 'dayz' => (int) ($this->d % $daysPerWeek),
2070                 'hours' => $this->hours,
2071                 'minutes' => $this->minutes,
2072                 'seconds' => $this->seconds,
2073                 'milliseconds' => (int) floor($this->microseconds / Carbon::MICROSECONDS_PER_MILLISECOND),
2074                 'microseconds' => (int) ($this->microseconds % Carbon::MICROSECONDS_PER_MILLISECOND),
2075             ];
2076     
2077             if (isset($factors['dayz']) && $factors['dayz'][0] !== 'weeks') {
2078                 $values['dayz'] += $values['weeks'] * $daysPerWeek;
2079                 $values['weeks'] = 0;
2080             }
2081     
2082             foreach ($factors as $source => [$target, $factor]) {
2083                 if ($source === $realUnit) {
2084                     $unitFound = true;
2085                     $value = $values[$source];
2086                     $result += $value;
2087                     $cumulativeFactor = 1;
2088                 }
2089     
2090                 if ($factor === false) {
2091                     if ($unitFound) {
2092                         break;
2093                     }
2094     
2095                     $result = 0;
2096                     $cumulativeFactor = 0;
2097     
2098                     continue;
2099                 }
2100     
2101                 if ($target === $realUnit) {
2102                     $unitFound = true;
2103                 }
2104     
2105                 if ($cumulativeFactor) {
2106                     $cumulativeFactor *= $factor;
2107                     $result += $values[$target] * $cumulativeFactor;
2108     
2109                     continue;
2110                 }
2111     
2112                 $value = $values[$source];
2113     
2114                 $result = ($result + $value) / $factor;
2115             }
2116     
2117             if (isset($target) && !$cumulativeFactor) {
2118                 $result += $values[$target];
2119             }
2120     
2121             if (!$unitFound) {
2122                 throw new InvalidArgumentException("Unit $unit have no configuration to get total from other units.");
2123             }
2124     
2125             if ($unit === 'weeks') {
2126                 return $result / $daysPerWeek;
2127             }
2128     
2129             return $result;
2130         }
