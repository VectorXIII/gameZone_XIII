18         public function testInputOutputRatioIs3to4Bytes()
19         {
20             /*
21             RFC 2045, 6.8
22     
23              The encoding process represents 24-bit groups of input bits as output
24              strings of 4 encoded characters.  Proceeding from left to right, a
25              24-bit input group is formed by concatenating 3 8bit input groups.
26              These 24 bits are then treated as 4 concatenated 6-bit groups, each
27              of which is translated into a single digit in the base64 alphabet.
28              */
29     
30             $this->assertEquals(
31                 'MTIz', $this->encoder->encodeString('123'),
32                 '%s: 3 bytes of input should yield 4 bytes of output'
33                 );
34             $this->assertEquals(
35                 'MTIzNDU2', $this->encoder->encodeString('123456'),
36                 '%s: 6 bytes in input should yield 8 bytes of output'
37                 );
38             $this->assertEquals(
39                 'MTIzNDU2Nzg5', $this->encoder->encodeString('123456789'),
40                 '%s: 9 bytes in input should yield 12 bytes of output'
41                 );
42         }

44         public function testPadLength()
45         {
46             /*
47             RFC 2045, 6.8
48     
49            Special processing is performed if fewer than 24 bits are available
50            at the end of the data being encoded.  A full encoding quantum is
51            always completed at the end of a body.  When fewer than 24 input bits
52            are available in an input group, zero bits are added (on the right)
53            to form an integral number of 6-bit groups.  Padding at the end of
54            the data is performed using the "=" character.  Since all base64
55            input is an integral number of octets, only the following cases can
56            arise: (1) the final quantum of encoding input is an integral
57            multiple of 24 bits; here, the final unit of encoded output will be
58            an integral multiple of 4 characters with no "=" padding, (2) the
59            final quantum of encoding input is exactly 8 bits; here, the final
60            unit of encoded output will be two characters followed by two "="
61            padding characters, or (3) the final quantum of encoding input is
62            exactly 16 bits; here, the final unit of encoded output will be three
63            characters followed by one "=" padding character.
64            */
65     
66             for ($i = 0; $i < 30; ++$i) {
67                 $input = pack('C', random_int(0, 255));
68                 $this->assertRegExp(
69                     '~^[a-zA-Z0-9/\+]{2}==$~', $this->encoder->encodeString($input),
70                     '%s: A single byte should have 2 bytes of padding'
71                     );
72             }
73     
74             for ($i = 0; $i < 30; ++$i) {
75                 $input = pack('C*', random_int(0, 255), random_int(0, 255));
76                 $this->assertRegExp(
77                     '~^[a-zA-Z0-9/\+]{3}=$~', $this->encoder->encodeString($input),
78                     '%s: Two bytes should have 1 byte of padding'
79                     );
80             }
81     
82             for ($i = 0; $i < 30; ++$i) {
83                 $input = pack('C*', random_int(0, 255), random_int(0, 255), random_int(0, 255));
84                 $this->assertRegExp(
85                     '~^[a-zA-Z0-9/\+]{4}$~', $this->encoder->encodeString($input),
86                     '%s: Three bytes should have no padding'
87                     );
88             }
89         }
