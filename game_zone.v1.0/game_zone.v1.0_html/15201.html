30     /**
31      * Remove the `.css` extension from a string if it has one.
32      *
33      * @internal
34      *
35      * @param string $name
36      *
37      * @return string
38      */
39     function _getNoCssExtension($name)
40     {
41         if (substr($name, -4, 4) === ".css") {
42             $name = preg_replace("/\.css$/", "", $name);
43         }
44     
45         return $name;
46     }
