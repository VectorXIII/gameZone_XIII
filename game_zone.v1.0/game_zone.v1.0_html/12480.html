212     class Factory
213     {
214         protected $className = Carbon::class;
215     
216         protected $settings = [];
217     
218         public function __construct(array $settings = [], string $className = null)
219         {
220             if ($className) {
221                 $this->className = $className;
222             }
223             $this->settings = $settings;
224         }
225     
226         public function getClassName()
227         {
228             return $this->className;
229         }
230     
231         public function setClassName(string $className)
232         {
233             $this->className = $className;
234     
235             return $this;
236         }
237     
238         public function className(string $className = null)
239         {
240             return $className === null ? $this->getClassName() : $this->setClassName($className);
241         }
242     
243         public function getSettings()
244         {
245             return $this->settings;
246         }
247     
248         public function setSettings(array $settings)
249         {
250             $this->settings = $settings;
251     
252             return $this;
253         }
254     
255         public function settings(array $settings = null)
256         {
257             return $settings === null ? $this->getSettings() : $this->setSettings($settings);
258         }
259     
260         public function mergeSettings(array $settings)
261         {
262             $this->settings = array_merge($this->settings, $settings);
263     
264             return $this;
265         }
266     
267         public function __call($name, $arguments)
268         {
269             $result = $this->className::$name(...$arguments);
270     
271             return $result instanceof CarbonInterface ? $result->settings($this->settings) : $result;
272         }
