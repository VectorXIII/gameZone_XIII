339         protected function dumpKey(Cursor $cursor)
340         {
341             if (null !== $key = $cursor->hashKey) {
342                 if ($cursor->hashKeyIsBinary) {
343                     $key = $this->utf8Encode($key);
344                 }
345                 $attr = ['binary' => $cursor->hashKeyIsBinary];
346                 $bin = $cursor->hashKeyIsBinary ? 'b' : '';
347                 $style = 'key';
348                 switch ($cursor->hashType) {
349                     default:
350                     case Cursor::HASH_INDEXED:
351                         if (self::DUMP_LIGHT_ARRAY & $this->flags) {
352                             break;
353                         }
354                         $style = 'index';
355                         // no break
356                     case Cursor::HASH_ASSOC:
357                         if (\is_int($key)) {
358                             $this->line .= $this->style($style, $key).' => ';
359                         } else {
360                             $this->line .= $bin.'"'.$this->style($style, $key).'" => ';
361                         }
362                         break;
363     
364                     case Cursor::HASH_RESOURCE:
365                         $key = "\0~\0".$key;
366                         // no break
367                     case Cursor::HASH_OBJECT:
368                         if (!isset($key[0]) || "\0" !== $key[0]) {
369                             $this->line .= '+'.$bin.$this->style('public', $key).': ';
370                         } elseif (0 < strpos($key, "\0", 1)) {
371                             $key = explode("\0", substr($key, 1), 2);
372     
373                             switch ($key[0][0]) {
374                                 case '+': // User inserted keys
375                                     $attr['dynamic'] = true;
376                                     $this->line .= '+'.$bin.'"'.$this->style('public', $key[1], $attr).'": ';
377                                     break 2;
378                                 case '~':
379                                     $style = 'meta';
380                                     if (isset($key[0][1])) {
381                                         parse_str(substr($key[0], 1), $attr);
382                                         $attr += ['binary' => $cursor->hashKeyIsBinary];
383                                     }
384                                     break;
385                                 case '*':
386                                     $style = 'protected';
387                                     $bin = '#'.$bin;
388                                     break;
389                                 default:
390                                     $attr['class'] = $key[0];
391                                     $style = 'private';
392                                     $bin = '-'.$bin;
393                                     break;
394                             }
395     
396                             if (isset($attr['collapse'])) {
397                                 if ($attr['collapse']) {
398                                     $this->collapseNextHash = true;
399                                 } else {
400                                     $this->expandNextHash = true;
401                                 }
402                             }
403     
404                             $this->line .= $bin.$this->style($style, $key[1], $attr).(isset($attr['separator']) ? $attr['separator'] : ': ');
405                         } else {
406                             // This case should not happen
407                             $this->line .= '-'.$bin.'"'.$this->style('private', $key, ['class' => '']).'": ';
408                         }
409                         break;
410                 }
411     
412                 if ($cursor->hardRefTo) {
413                     $this->line .= $this->style('ref', '&'.($cursor->hardRefCount ? $cursor->hardRefTo : ''), ['count' => $cursor->hardRefCount]).' ';
414                 }
415             }

418         /**
419          * Decorates a value with some style.
420          *
421          * @param string $style The type of style being applied
422          * @param string $value The value being styled
423          * @param array  $attr  Optional context information
424          *
425          * @return string The value with style decoration
426          */
427         protected function style($style, $value, $attr = [])
428         {
429             if (null === $this->colors) {
430                 $this->colors = $this->supportsColors();
431             }
432     
433             if (null === $this->handlesHrefGracefully) {
434                 $this->handlesHrefGracefully = 'JetBrains-JediTerm' !== getenv('TERMINAL_EMULATOR') && !getenv('KONSOLE_VERSION');
435             }
436     
437             if (isset($attr['ellipsis'], $attr['ellipsis-type'])) {
438                 $prefix = substr($value, 0, -$attr['ellipsis']);
439                 if ('cli' === \PHP_SAPI && 'path' === $attr['ellipsis-type'] && isset($_SERVER[$pwd = '\\' === \DIRECTORY_SEPARATOR ? 'CD' : 'PWD']) && 0 === strpos($prefix, $_SERVER[$pwd])) {
440                     $prefix = '.'.substr($prefix, \strlen($_SERVER[$pwd]));
441                 }
442                 if (!empty($attr['ellipsis-tail'])) {
443                     $prefix .= substr($value, -$attr['ellipsis'], $attr['ellipsis-tail']);
444                     $value = substr($value, -$attr['ellipsis'] + $attr['ellipsis-tail']);
445                 } else {
446                     $value = substr($value, -$attr['ellipsis']);
447                 }
448     
449                 $value = $this->style('default', $prefix).$this->style($style, $value);
450     
451                 goto href;
452             }
453     
454             $map = static::$controlCharsMap;
455             $startCchr = $this->colors ? "\033[m\033[{$this->styles['default']}m" : '';
456             $endCchr = $this->colors ? "\033[m\033[{$this->styles[$style]}m" : '';
457             $value = preg_replace_callback(static::$controlCharsRx, function ($c) use ($map, $startCchr, $endCchr) {
458                 $s = $startCchr;
459                 $c = $c[$i = 0];
460                 do {
461                     $s .= isset($map[$c[$i]]) ? $map[$c[$i]] : sprintf('\x%02X', \ord($c[$i]));
462                 } while (isset($c[++$i]));
463     
464                 return $s.$endCchr;
465             }, $value, -1, $cchrCount);
466     
467             if ($this->colors) {
468                 if ($cchrCount && "\033" === $value[0]) {
469                     $value = substr($value, \strlen($startCchr));
470                 } else {
471                     $value = "\033[{$this->styles[$style]}m".$value;
472                 }
473                 if ($cchrCount && $endCchr === substr($value, -\strlen($endCchr))) {
474                     $value = substr($value, 0, -\strlen($endCchr));
475                 } else {
476                     $value .= "\033[{$this->styles['default']}m";
477                 }
478             }
479     
480             href:
481             if ($this->colors && $this->handlesHrefGracefully) {
482                 if (isset($attr['file']) && $href = $this->getSourceLink($attr['file'], isset($attr['line']) ? $attr['line'] : 0)) {
483                     if ('note' === $style) {
484                         $value .= "\033]8;;{$href}\033\\^\033]8;;\033\\";
485                     } else {
486                         $attr['href'] = $href;
487                     }
488                 }
489                 if (isset($attr['href'])) {
490                     $value = "\033]8;;{$attr['href']}\033\\{$value}\033]8;;\033\\";
491                 }
492             } elseif ($attr['if_links'] ?? false) {
493                 return '';
494             }
495     
496             return $value;
497         }

499         /**
500          * @return bool Tells if the current output stream supports ANSI colors or not
501          */
502         protected function supportsColors()
503         {
504             if ($this->outputStream !== static::$defaultOutput) {
505                 return $this->hasColorSupport($this->outputStream);
506             }
507             if (null !== static::$defaultColors) {
508                 return static::$defaultColors;
509             }
510             if (isset($_SERVER['argv'][1])) {
511                 $colors = $_SERVER['argv'];
512                 $i = \count($colors);
513                 while (--$i > 0) {
514                     if (isset($colors[$i][5])) {
515                         switch ($colors[$i]) {
516                             case '--ansi':
517                             case '--color':
518                             case '--color=yes':
519                             case '--color=force':
520                             case '--color=always':
521                                 return static::$defaultColors = true;
522     
523                             case '--no-ansi':
524                             case '--color=no':
525                             case '--color=none':
526                             case '--color=never':
527                                 return static::$defaultColors = false;
528                         }
529                     }
530                 }
531             }
532     
533             $h = stream_get_meta_data($this->outputStream) + ['wrapper_type' => null];
534             $h = 'Output' === $h['stream_type'] && 'PHP' === $h['wrapper_type'] ? fopen('php://stdout', 'wb') : $this->outputStream;
535     
536             return static::$defaultColors = $this->hasColorSupport($h);
537         }

567         /**
568          * Returns true if the stream supports colorization.
569          *
570          * Reference: Composer\XdebugHandler\Process::supportsColor
571          * https://github.com/composer/xdebug-handler
572          *
573          * @param mixed $stream A CLI output stream
574          */
575         private function hasColorSupport($stream): bool
576         {
577             if (!\is_resource($stream) || 'stream' !== get_resource_type($stream)) {
578                 return false;
579             }
580     
581             // Follow https://no-color.org/
582             if (isset($_SERVER['NO_COLOR']) || false !== getenv('NO_COLOR')) {
583                 return false;
584             }
585     
586             if ('Hyper' === getenv('TERM_PROGRAM')) {
587                 return true;
588             }
589     
590             if (\DIRECTORY_SEPARATOR === '\\') {
591                 return (\function_exists('sapi_windows_vt100_support')
592                     && @sapi_windows_vt100_support($stream))
593                     || false !== getenv('ANSICON')
594                     || 'ON' === getenv('ConEmuANSI')
595                     || 'xterm' === getenv('TERM');
596             }
597     
598             return stream_isatty($stream);
599         }
