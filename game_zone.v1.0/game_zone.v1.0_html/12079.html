15     namespace League\CommonMark\Inline\Parser;
16     
17     use League\CommonMark\Delimiter\Delimiter;
18     use League\CommonMark\Inline\Element\Text;
19     use League\CommonMark\InlineParserContext;
20     
21     final class BangParser implements InlineParserInterface
22     {
23         public function getCharacters(): array
24         {
25             return ['!'];
26         }
27     
28         public function parse(InlineParserContext $inlineContext): bool
29         {
30             $cursor = $inlineContext->getCursor();
31             if ($cursor->peek() === '[') {
32                 $cursor->advanceBy(2);
33                 $node = new Text('![', ['delim' => true]);
34                 $inlineContext->getContainer()->appendChild($node);
35     
36                 // Add entry to stack for this opener
37                 $delimiter = new Delimiter('!', 1, $node, true, false, $cursor->getPosition());
38                 $inlineContext->getDelimiterStack()->push($delimiter);
39     
40                 return true;
41             }
42     
43             return false;
44         }
45     }
