21     /**
22      * Interface for an Environment which can be configured with config settings, parsers, processors, and renderers
23      */
24     interface ConfigurableEnvironmentInterface extends EnvironmentInterface
25     {
26         /**
27          * @param array<string, mixed> $config
28          *
29          * @return void
30          */
31         public function mergeConfig(array $config = []);
32     
33         /**
34          * @param array<string, mixed> $config
35          *
36          * @return void
37          */
38         public function setConfig(array $config = []);
39     
40         /**
41          * Registers the given extension with the Environment
42          *
43          * @param ExtensionInterface $extension
44          *
45          * @return ConfigurableEnvironmentInterface
46          */
47         public function addExtension(ExtensionInterface $extension): ConfigurableEnvironmentInterface;
48     
49         /**
50          * Registers the given block parser with the Environment
51          *
52          * @param BlockParserInterface $parser   Block parser instance
53          * @param int                  $priority Priority (a higher number will be executed earlier)
54          *
55          * @return self
56          */
57         public function addBlockParser(BlockParserInterface $parser, int $priority = 0): ConfigurableEnvironmentInterface;
58     
59         /**
60          * Registers the given inline parser with the Environment
61          *
62          * @param InlineParserInterface $parser   Inline parser instance
63          * @param int                   $priority Priority (a higher number will be executed earlier)
64          *
65          * @return self
66          */
67         public function addInlineParser(InlineParserInterface $parser, int $priority = 0): ConfigurableEnvironmentInterface;
68     
69         /**
70          * Registers the given delimiter processor with the Environment
71          *
72          * @param DelimiterProcessorInterface $processor Delimiter processors instance
73          *
74          * @return ConfigurableEnvironmentInterface
75          */
76         public function addDelimiterProcessor(DelimiterProcessorInterface $processor): ConfigurableEnvironmentInterface;
77     
78         /**
79          * @param string                 $blockClass    The fully-qualified block element class name the renderer below should handle
80          * @param BlockRendererInterface $blockRenderer The renderer responsible for rendering the type of element given above
81          * @param int                    $priority      Priority (a higher number will be executed earlier)
82          *
83          * @return self
84          */
85         public function addBlockRenderer($blockClass, BlockRendererInterface $blockRenderer, int $priority = 0): ConfigurableEnvironmentInterface;
86     
87         /**
88          * Registers the given inline renderer with the Environment
89          *
90          * @param string                  $inlineClass The fully-qualified inline element class name the renderer below should handle
91          * @param InlineRendererInterface $renderer    The renderer responsible for rendering the type of element given above
92          * @param int                     $priority    Priority (a higher number will be executed earlier)
93          *
94          * @return self
95          */
96         public function addInlineRenderer(string $inlineClass, InlineRendererInterface $renderer, int $priority = 0): ConfigurableEnvironmentInterface;
97     
98         /**
99          * Registers the given event listener
100          *
101          * @param string   $eventClass Fully-qualified class name of the event this listener should respond to
102          * @param callable $listener   Listener to be executed
103          * @param int      $priority   Priority (a higher number will be executed earlier)
104          *
105          * @return self
106          */
107         public function addEventListener(string $eventClass, callable $listener, int $priority = 0): ConfigurableEnvironmentInterface;
108     }
