474         /**
475          * Parse given ISO 8601 string into an array of arguments.
476          *
477          * @SuppressWarnings(PHPMD.ElseExpression)
478          *
479          * @param string $iso
480          *
481          * @return array
482          */
483         protected static function parseIso8601($iso)
484         {
485             $result = [];
486     
487             $interval = null;
488             $start = null;
489             $end = null;
490     
491             foreach (explode('/', $iso) as $key => $part) {
492                 if ($key === 0 && preg_match('/^R([0-9]*)$/', $part, $match)) {
493                     $parsed = strlen($match[1]) ? (int) $match[1] : null;
494                 } elseif ($interval === null && $parsed = CarbonInterval::make($part)) {
495                     $interval = $part;
496                 } elseif ($start === null && $parsed = Carbon::make($part)) {
497                     $start = $part;
498                 } elseif ($end === null && $parsed = Carbon::make(static::addMissingParts($start, $part))) {
499                     $end = $part;
500                 } else {
501                     throw new InvalidArgumentException("Invalid ISO 8601 specification: $iso.");
502                 }
503     
504                 $result[] = $parsed;
505             }
506     
507             return $result;
508         }

612         /**
613          * CarbonPeriod constructor.
614          *
615          * @SuppressWarnings(PHPMD.ElseExpression)
616          *
617          * @throws InvalidArgumentException
618          */
619         public function __construct(...$arguments)
620         {
621             // Parse and assign arguments one by one. First argument may be an ISO 8601 spec,
622             // which will be first parsed into parts and then processed the same way.
623     
624             $agumentsCount = count($arguments);
625     
626             if ($agumentsCount && static::isIso8601($iso = $arguments[0])) {
627                 array_splice($arguments, 0, 1, static::parseIso8601($iso));
628             }
629     
630             if ($agumentsCount === 1) {
631                 if ($arguments[0] instanceof DatePeriod) {
632                     $arguments = [
633                         $arguments[0]->start,
634                         $arguments[0]->end ?: ($arguments[0]->recurrences - 1),
635                         $arguments[0]->interval,
636                         $arguments[0]->include_start_date ? 0 : static::EXCLUDE_START_DATE,
637                     ];
638                 } elseif ($arguments[0] instanceof self) {
639                     $arguments = [
640                         $arguments[0]->getStartDate(),
641                         $arguments[0]->getEndDate() ?: $arguments[0]->getRecurrences(),
642                         $arguments[0]->getDateInterval(),
643                         $arguments[0]->getOptions(),
644                     ];
645                 }
646             }
647     
648             foreach ($arguments as $argument) {
649                 if ($this->dateInterval === null &&
650                     (
651                         is_string($argument) && preg_match('/^(\d.*|P[T0-9].*|(?:\h*\d+(?:\.\d+)?\h*[a-z]+)+)$/i', $argument) ||
652                         $argument instanceof DateInterval
653                     ) &&
654                     $parsed = @CarbonInterval::make($argument)
655                 ) {
656                     $this->setDateInterval($parsed);
657                 } elseif ($this->startDate === null && $parsed = Carbon::make($argument)) {
658                     $this->setStartDate($parsed);
659                 } elseif ($this->endDate === null && $parsed = Carbon::make($argument)) {
660                     $this->setEndDate($parsed);
661                 } elseif ($this->recurrences === null && $this->endDate === null && is_numeric($argument)) {
662                     $this->setRecurrences($argument);
663                 } elseif ($this->options === null && (is_int($argument) || $argument === null)) {
664                     $this->setOptions($argument);
665                 } else {
666                     throw new InvalidArgumentException('Invalid constructor parameters.');
667                 }
668             }
669     
670             if ($this->startDate === null) {
671                 $this->setStartDate(Carbon::now());
672             }
673     
674             if ($this->dateInterval === null) {
675                 $this->setDateInterval(CarbonInterval::day());
676     
677                 $this->isDefaultInterval = true;
678             }
679     
680             if ($this->options === null) {
681                 $this->setOptions(0);
682             }
683         }
