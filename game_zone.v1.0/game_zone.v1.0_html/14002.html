46         {
47             if (\substr($target, -1, 1) !== \DIRECTORY_SEPARATOR) {
48                 $target .= \DIRECTORY_SEPARATOR;
49             }
50     
51             $this->target = $target;
52             $this->initTargetDirectory($target);
53     
54             $report = $coverage->getReport();
55     
56             $this->project = new Project(
57                 $coverage->getReport()->getName()

78         private function initTargetDirectory(string $directory): void
79         {
80             if (\file_exists($directory)) {
81                 if (!\is_dir($directory)) {
82                     throw new RuntimeException(
83                         "'$directory' exists but is not a directory."
84                     );
85                 }
86     
87                 if (!\is_writable($directory)) {
88                     throw new RuntimeException(
89                         "'$directory' exists but is not writable."
90                     );
91                 }
92             } elseif (!$this->createDirectory($directory)) {
93                 throw new RuntimeException(
94                     "'$directory' could not be created."
95                 );

99         private function processDirectory(DirectoryNode $directory, Node $context): void
100         {
101             $directoryName = $directory->getName();
102     
103             if ($this->project->getProjectSourceDirectory() === $directoryName) {
104                 $directoryName = '/';
105             }
106     
107             $directoryObject = $context->addDirectory($directoryName);
108     
109             $this->setTotals($directory, $directoryObject->getTotals());
110     
111             foreach ($directory->getDirectories() as $node) {
112                 $this->processDirectory($node, $directoryObject);
113             }
114     
115             foreach ($directory->getFiles() as $node) {
116                 $this->processFile($node, $directoryObject);
117             }
118         }

120         /**
121          * @throws RuntimeException
122          */
123         private function processFile(FileNode $file, Directory $context): void
124         {
125             $fileObject = $context->addFile(
126                 $file->getName(),
127                 $file->getId() . '.xml'
128             );
129     
130             $this->setTotals($file, $fileObject->getTotals());
131     
132             $path = \substr(
133                 $file->getPath(),
134                 \strlen($this->project->getProjectSourceDirectory())
135             );
136     
137             $fileReport = new Report($path);
138     
139             $this->setTotals($file, $fileReport->getTotals());
140     
141             foreach ($file->getClassesAndTraits() as $unit) {
142                 $this->processUnit($unit, $fileReport);
143             }
144     
145             foreach ($file->getFunctions() as $function) {
146                 $this->processFunction($function, $fileReport);
147             }
148     
149             foreach ($file->getCoverageData() as $line => $tests) {
150                 if (!\is_array($tests) || \count($tests) === 0) {
151                     continue;
152                 }
153     
154                 $coverage = $fileReport->getLineCoverage((string) $line);
155     
156                 foreach ($tests as $test) {
157                     $coverage->addTest($test);
158                 }
159     
160                 $coverage->finalize();
161             }
162     
163             $fileReport->getSource()->setSourceCode(
164                 \file_get_contents($file->getPath())
165             );
166     
167             $this->saveDocument($fileReport->asDom(), $file->getId());
168         }

170         private function processUnit(array $unit, Report $report): void
171         {
172             if (isset($unit['className'])) {
173                 $unitObject = $report->getClassObject($unit['className']);
174             } else {
175                 $unitObject = $report->getTraitObject($unit['traitName']);
176             }
177     
178             $unitObject->setLines(
179                 $unit['startLine'],
180                 $unit['executableLines'],
181                 $unit['executedLines']
182             );
183     
184             $unitObject->setCrap((float) $unit['crap']);
185     
186             $unitObject->setPackage(
187                 $unit['package']['fullPackage'],
188                 $unit['package']['package'],
189                 $unit['package']['subpackage'],
190                 $unit['package']['category']
191             );
192     
193             $unitObject->setNamespace($unit['package']['namespace']);
194     
195             foreach ($unit['methods'] as $method) {
196                 $methodObject = $unitObject->addMethod($method['methodName']);
197                 $methodObject->setSignature($method['signature']);
198                 $methodObject->setLines((string) $method['startLine'], (string) $method['endLine']);
199                 $methodObject->setCrap($method['crap']);
200                 $methodObject->setTotals(
201                     (string) $method['executableLines'],
202                     (string) $method['executedLines'],
203                     (string) $method['coverage']
204                 );
205             }
206         }

218         private function processTests(array $tests): void
219         {
220             $testsObject = $this->project->getTests();
221     
222             foreach ($tests as $test => $result) {
223                 if ($test === 'UNCOVERED_FILES_FROM_WHITELIST') {
224                     continue;
225                 }
226     
227                 $testsObject->addTest($test, $result);
228             }
229         }
