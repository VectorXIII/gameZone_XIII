14     /**
15      * Constraint that asserts that the array it is evaluated for has a given key.
16      *
17      * Uses array_key_exists() to check if the key is found in the input array, if
18      * not found the evaluation fails.
19      *
20      * The array key is passed in the constructor.
21      */
22     final class ArrayHasKey extends Constraint
23     {
24         /**
25          * @var int|string
26          */
27         private $key;
28     
29         /**
30          * @param int|string $key
31          */
32         public function __construct($key)
33         {
34             $this->key = $key;
35         }
36     
37         /**
38          * Returns a string representation of the constraint.
39          *
40          * @throws \SebastianBergmann\RecursionContext\InvalidArgumentException
41          */
42         public function toString(): string
43         {
44             return 'has the key ' . $this->exporter()->export($this->key);
45         }
46     
47         /**
48          * Evaluates the constraint for parameter $other. Returns true if the
49          * constraint is met, false otherwise.
50          *
51          * @param mixed $other value or object to evaluate
52          */
53         protected function matches($other): bool
54         {
55             if (\is_array($other)) {
56                 return \array_key_exists($this->key, $other);
57             }
58     
59             if ($other instanceof ArrayAccess) {
60                 return $other->offsetExists($this->key);
61             }
62     
63             return false;
64         }

74          * @throws \SebastianBergmann\RecursionContext\InvalidArgumentException

76         protected function failureDescription($other): string
