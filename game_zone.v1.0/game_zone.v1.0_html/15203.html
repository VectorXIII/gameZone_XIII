35     /**
36      * Get a list of available stylesheets.
37      *
38      * By default, a list of filenames without the `.css` extension will be returned.
39      * This can be configured with the `$filePaths` argument.
40      *
41      * @api
42      *
43      * @since 9.15.8.1
44      *
45      * @param bool $filePaths Return absolute paths to stylesheets instead
46      *
47      * @return string[]
48      */
49     function getAvailableStyleSheets($filePaths = false)
50     {
51         $results = array();
52     
53         $folder = getStyleSheetFolder();
54         $dh = @dir($folder);
55     
56         if ($dh) {
57             while (($entry = $dh->read()) !== false) {
58                 if (substr($entry, -4, 4) !== ".css") {
59                     continue;
60                 }
61     
62                 if ($filePaths) {
63                     $results[] = implode(DIRECTORY_SEPARATOR, array($folder, $entry));
64                 } else {
65                     $results[] = basename($entry, ".css");
66                 }
67             }
68     
69             $dh->close();
70         }
71     
72         return $results;
73     }

131     /**
132      * Get the absolute path to a given stylesheet distributed in this package.
133      *
134      * @api
135      *
136      * @since 9.15.8.1
137      *
138      * @param string $name The stylesheet name (with or without the extension)
139      *
140      * @throws \DomainException when the no stylesheet with this name exists
141      *
142      * @return string The absolute path to the stylesheet with the given name
143      */
144     function getStyleSheetPath($name)
145     {
146         $name = _getNoCssExtension($name);
147         $path = implode(DIRECTORY_SEPARATOR, array(getStyleSheetFolder(), $name)) . ".css";
148     
149         if (!file_exists($path)) {
150             throw new \DomainException("There is no stylesheet with by the name of '$name'.");
151         }
152     
153         return $path;
154     }

156     /**
157      * Convert the HTML generated by Highlighter and split it up into an array of lines.
158      *
159      * @api
160      *
161      * @since 9.15.6.1
162      *
163      * @param string $html An HTML string generated by `Highlighter::highlight()`
164      *
165      * @throws \RuntimeException         when the DOM extension is not available
166      * @throws \UnexpectedValueException when the given HTML could not be parsed
167      *
168      * @return string[]|false An array of lines of code as strings. False if an error occurred in splitting up by lines
169      */
170     function splitCodeIntoArray($html)
171     {
172         if (!extension_loaded("dom")) {
173             throw new \RuntimeException("The DOM extension is not loaded but is required.");
174         }
175     
176         $dom = new \DOMDocument();
177     
178         if (!$dom->loadHTML($html)) {
179             throw new \UnexpectedValueException("The given HTML could not be parsed correctly.");
180         }
181     
182         $spans = $dom->getElementsByTagName("span");
183     
184         /** @var \DOMElement $span */
185         foreach ($spans as $span) {
186             $classes = $span->getAttribute("class");
187             $renderedSpan = $dom->saveHTML($span);
188     
189             if (preg_match('/\R/', $renderedSpan)) {
190                 $finished = preg_replace(
191                     '/\R/',
192                     sprintf('</span>%s<span class="%s">', PHP_EOL, $classes),
193                     $renderedSpan
194                 );
195                 $html = str_replace($renderedSpan, $finished, $html);
196             }
197         }
198     
199         return preg_split('/\R/', $html);
200     }
