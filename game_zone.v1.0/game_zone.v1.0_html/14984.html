15     namespace Ramsey\Collection;

17     /**
18      * A collection represents a group of objects, known as its elements.
19      *
20      * Some collections allow duplicate elements and others do not. Some are ordered
21      * and others unordered.
22      */
23     interface CollectionInterface extends ArrayInterface
24     {
25         /**
26          * Ascending sort type.
27          */
28         public const SORT_ASC = 'asc';
29     
30         /**
31          * Descending sort type.
32          */
33         public const SORT_DESC = 'desc';
34     
35         /**
36          * Ensures that this collection contains the specified element (optional
37          * operation).
38          *
39          * Returns `true` if this collection changed as a result of the call.
40          * (Returns `false` if this collection does not permit duplicates and
41          * already contains the specified element.)
42          *
43          * Collections that support this operation may place limitations on what
44          * elements may be added to this collection. In particular, some
45          * collections will refuse to add `null` elements, and others will impose
46          * restrictions on the type of elements that may be added. Collection
47          * classes should clearly specify in their documentation any restrictions
48          * on what elements may be added.
49          *
50          * If a collection refuses to add a particular element for any reason other
51          * than that it already contains the element, it must throw an exception
52          * (rather than returning `false`). This preserves the invariant that a
53          * collection always contains the specified element after this call returns.
54          *
55          * @param mixed $element The element to add to the collection.
56          *
57          * @return bool `true` if this collection changed as a result of the call.
58          */
59         public function add($element): bool;
60     
61         /**
62          * Returns `true` if this collection contains the specified element.
63          *
64          * @param mixed $element The element to check whether the collection contains.
65          * @param bool $strict Whether to perform a strict type check on the value.
66          */
67         public function contains($element, bool $strict = true): bool;
68     
69         /**
70          * Returns the type associated with this collection.
71          */
72         public function getType(): string;
73     
74         /**
75          * Removes a single instance of the specified element from this collection,
76          * if it is present.
77          *
78          * @param mixed $element The element to remove from the collection.
79          *
80          * @return bool `true` if an element was removed as a result of this call.
81          */
82         public function remove($element): bool;
83     
84         /**
85          * Returns the values from the given property or method.
86          *
87          * @param string $propertyOrMethod The property or method name to filter by.
88          *
89          * @return mixed[]
90          */
91         public function column(string $propertyOrMethod): array;
92     
93         /**
94          * Returns the first item of the collection.
95          *
96          * @return mixed
97          */
98         public function first();
99     
100         /**
101          * Returns the last item of the collection.
102          *
103          * @return mixed
104          */
105         public function last();
106     
107         /**
108          * Sort the collection by a property or method with the given sort order.
109          *
110          * This will always leave the original collection untouched and will return
111          * a new one.
112          *
113          * @param string $propertyOrMethod The property or method to sort by.
114          * @param string $order The sort order for the resulting collection (one of
115          *     this interface's `SORT_*` constants).
116          *
117          * @return CollectionInterface<mixed, mixed>
118          */
119         public function sort(string $propertyOrMethod, string $order = self::SORT_ASC): self;
120     
121         /**
122          * Filter out items of the collection which don't match the criteria of
123          * given callback.
124          *
125          * This will always leave the original collection untouched and will return
126          * a new one.
127          *
128          * See the {@link http://php.net/manual/en/function.array-filter.php PHP array_filter() documentation}
129          * for examples of how the `$callback` parameter works.
130          *
131          * @param callable $callback A callable to use for filtering elements.
132          *
133          * @return CollectionInterface<mixed, mixed>
134          */
135         public function filter(callable $callback): self;
136     
137         /**
138          * Create a new collection where items match the criteria of given callback.
139          *
140          * This will always leave the original collection untouched and will return
141          * a new one.
142          *
143          * @param string $propertyOrMethod The property or method to evaluate.
144          * @param mixed  $value The value to match.
145          *
146          * @return CollectionInterface<mixed, mixed>
147          */
148         public function where(string $propertyOrMethod, $value): self;
149     
150         /**
151          * Apply a given callback method on each item of the collection.
152          *
153          * This will always leave the original collection untouched and will return
154          * a new one.
155          *
156          * See the {@link http://php.net/manual/en/function.array-map.php PHP array_map() documentation}
157          * for examples of how the `$callback` parameter works.
158          *
159          * @param callable $callback A callable to apply to each item of the
160          *     collection.
161          *
162          * @return CollectionInterface<mixed, mixed>
163          */
164         public function map(callable $callback): self;
165     
166         /**
167          * Create a new collection with divergent items between current and given
168          * collection.
169          *
170          * @param CollectionInterface<mixed, mixed> $other The collection to check for divergent
171          *     items.
172          *
173          * @return CollectionInterface<mixed, mixed>
174          */
175         public function diff(CollectionInterface $other): self;
176     
177         /**
178          * Create a new collection with intersecting item between current and given
179          * collection.
180          *
181          * @param CollectionInterface<mixed, mixed> $other The collection to check for
182          *     intersecting items.
183          *
184          * @return CollectionInterface<mixed, mixed>
185          */
186         public function intersect(CollectionInterface $other): self;
187     
188         /**
189          * Merge current items and items of given collections into a new one.
190          *
191          * @param CollectionInterface<mixed, mixed> ...$collections The collections to merge.
192          *
193          * @return CollectionInterface<mixed, mixed>
194          */
195         public function merge(CollectionInterface ...$collections): self;
196     }
