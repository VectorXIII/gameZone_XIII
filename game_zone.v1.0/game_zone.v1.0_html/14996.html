21     /**
22      * This class provides a basic implementation of `TypedMapInterface`, to
23      * minimize the effort required to implement this interface.
24      */
25     abstract class AbstractTypedMap extends AbstractMap implements TypedMapInterface
26     {
27         use TypeTrait;
28         use ValueToStringTrait;
29     
30         /**
31          * Sets the given value to the given offset in the map.
32          *
33          * @param mixed $offset The offset to set.
34          * @param mixed $value The value to set at the given offset.
35          *
36          * @throws InvalidArgumentException if the offset or value do not match the
37          *     expected types.
38          */
39         public function offsetSet($offset, $value): void
40         {
41             if ($this->checkType($this->getKeyType(), $offset) === false) {
42                 throw new InvalidArgumentException(
43                     'Key must be of type ' . $this->getKeyType() . '; key is '
44                     . $this->toolValueToString($offset)
45                 );
46             }
47     
48             if ($this->checkType($this->getValueType(), $value) === false) {
49                 throw new InvalidArgumentException(
50                     'Value must be of type ' . $this->getValueType() . '; value is '
51                     . $this->toolValueToString($value)
52                 );
53             }
54     
55             parent::offsetSet($offset, $value);
56         }
57     }
