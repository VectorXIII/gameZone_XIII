58         /**
59          * Compile a foreign key command.
60          *
61          * @param  \Illuminate\Database\Schema\Blueprint  $blueprint
62          * @param  \Illuminate\Support\Fluent  $command
63          * @return string
64          */
65         public function compileForeign(Blueprint $blueprint, Fluent $command)
66         {
67             // We need to prepare several of the elements of the foreign key definition
68             // before we can create the SQL, such as wrapping the tables and convert
69             // an array of columns to comma-delimited strings for the SQL queries.
70             $sql = sprintf('alter table %s add constraint %s ',
71                 $this->wrapTable($blueprint),
72                 $this->wrap($command->index)
73             );
74     
75             // Once we have the initial portion of the SQL statement we will add on the
76             // key name, table name, and referenced columns. These will complete the
77             // main portion of the SQL statement and this SQL will almost be done.
78             $sql .= sprintf('foreign key (%s) references %s (%s)',
79                 $this->columnize($command->columns),
80                 $this->wrapTable($command->on),
81                 $this->columnize((array) $command->references)
82             );
83     
84             // Once we have the basic foreign key creation statement constructed we can
85             // build out the syntax for what should happen on an update or delete of
86             // the affected columns, which will get something like "cascade", etc.
87             if (! is_null($command->onDelete)) {
88                 $sql .= " on delete {$command->onDelete}";
89             }
90     
91             if (! is_null($command->onUpdate)) {
92                 $sql .= " on update {$command->onUpdate}";
93             }
94     
95             return $sql;
96         }
