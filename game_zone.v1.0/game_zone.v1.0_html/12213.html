27         public function apply($code, MockConfiguration $config)
28         {
29             if ($config->requiresCallTypeHintRemoval()) {

31                     'public function __call($method, array $args)',
32                     'public function __call($method, $args)',
33                     $code
34                 );
35             }
36

39                     'public static function __callStatic($method, array $args)',
40                     'public static function __callStatic($method, $args)',
41                     $code
42                 );
43             }
44

46         }
