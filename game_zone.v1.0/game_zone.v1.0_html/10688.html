65         /**
66          * @example 'Fietsenmaker Zijlemans'
67          *
68          * @return string
69          */
70         public function company()
71         {
72             $determinator = static::numberBetween(0, 2);
73             switch ($determinator) {
74                 case 0:
75                     $companyName = static::randomElement(static::$product) . ' ' . static::randomElement(static::$type);
76                     break;
77                 case 1:
78                     $companyName = static::randomElement(static::$product) . strtolower(static::randomElement(static::$type));
79                     break;
80                 case 2:
81                     $companyName = static::randomElement(static::$store) . ' ' . $this->generator->lastName;
82                     break;
83             }
84     
85             if (0 !== static::numberBetween(0, 1)) {
86                 return $companyName . ' ' . static::randomElement(static::$companySuffix);
87             }
88     
89             return $companyName;
90         }
