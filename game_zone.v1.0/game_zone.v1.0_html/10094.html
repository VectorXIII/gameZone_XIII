55          * Factory method to create a new CronExpression.
56          *
57          * @param string $expression The CRON expression to create.  There are
58          *                           several special predefined values which can be used to substitute the
59          *                           CRON expression:
60          *
61          *      `@yearly`, `@annually` - Run once a year, midnight, Jan. 1 - 0 0 1 1 *
62          *      `@monthly` - Run once a month, midnight, first of month - 0 0 1 * *
63          *      `@weekly` - Run once a week, midnight on Sun - 0 0 * * 0
64          *      `@daily` - Run once a day, midnight - 0 0 * * *
65          *      `@hourly` - Run once an hour, first minute - 0 * * * *
66          * @param FieldFactory|null $fieldFactory Field factory to use
67          *
68          * @return CronExpression
69          */
70         public static function factory($expression, FieldFactory $fieldFactory = null)
71         {
72             $mappings = array(
73                 '@yearly' => '0 0 1 1 *',
74                 '@annually' => '0 0 1 1 *',
75                 '@monthly' => '0 0 1 * *',
76                 '@weekly' => '0 0 * * 0',
77                 '@daily' => '0 0 * * *',
78                 '@hourly' => '0 * * * *'
79             );
80     
81             if (isset($mappings[$expression])) {
82                 $expression = $mappings[$expression];
83             }
84     
85             return new static($expression, $fieldFactory ?: new FieldFactory());
86         }

311         /**
312          * Get the next or previous run date of the expression relative to a date
313          *
314          * @param string|\DateTimeInterface $currentTime      Relative calculation date
315          * @param int                       $nth              Number of matches to skip before returning
316          * @param bool                      $invert           Set to TRUE to go backwards in time
317          * @param bool                      $allowCurrentDate Set to TRUE to return the
318          *                                                    current date if it matches the cron expression
319          * @param string|null               $timeZone         TimeZone to use instead of the system default
320          *
321          * @return \DateTime
322          * @throws \RuntimeException on too many iterations
323          */
324         protected function getRunDate($currentTime = null, $nth = 0, $invert = false, $allowCurrentDate = false, $timeZone = null)
325         {
326             $timeZone = $this->determineTimeZone($currentTime, $timeZone);
327     
328             if ($currentTime instanceof DateTime) {
329                 $currentDate = clone $currentTime;
330             } elseif ($currentTime instanceof DateTimeImmutable) {
331                 $currentDate = DateTime::createFromFormat('U', $currentTime->format('U'));
332             } else {
333                 $currentDate = new DateTime($currentTime ?: 'now');
334             }
335     
336             $currentDate->setTimeZone(new DateTimeZone($timeZone));
337             $currentDate->setTime($currentDate->format('H'), $currentDate->format('i'), 0);
338             $nextRun = clone $currentDate;
339             $nth = (int) $nth;
340     
341             // We don't have to satisfy * or null fields
342             $parts = array();
343             $fields = array();
344             foreach (self::$order as $position) {
345                 $part = $this->getExpression($position);
346                 if (null === $part || '*' === $part) {
347                     continue;
348                 }
349                 $parts[$position] = $part;
350                 $fields[$position] = $this->fieldFactory->getField($position);
351             }
352     
353             // Set a hard limit to bail on an impossible date
354             for ($i = 0; $i < $this->maxIterationCount; $i++) {
355     
356                 foreach ($parts as $position => $part) {
357                     $satisfied = false;
358                     // Get the field object used to validate this part
359                     $field = $fields[$position];
360                     // Check if this is singular or a list
361                     if (strpos($part, ',') === false) {
362                         $satisfied = $field->isSatisfiedBy($nextRun, $part);
363                     } else {
364                         foreach (array_map('trim', explode(',', $part)) as $listPart) {
365                             if ($field->isSatisfiedBy($nextRun, $listPart)) {
366                                 $satisfied = true;
367                                 break;
368                             }
369                         }
370                     }
371     
372                     // If the field is not satisfied, then start over
373                     if (!$satisfied) {
374                         $field->increment($nextRun, $invert, $part);
375                         continue 2;
376                     }
377                 }
378     
379                 // Skip this match if needed
380                 if ((!$allowCurrentDate && $nextRun == $currentDate) || --$nth > -1) {
381                     $this->fieldFactory->getField(0)->increment($nextRun, $invert, isset($parts[0]) ? $parts[0] : null);
382                     continue;
383                 }
384     
385                 return $nextRun;
386             }
387     
388             // @codeCoverageIgnoreStart
389             throw new RuntimeException('Impossible CRON expression');
390             // @codeCoverageIgnoreEnd
391         }
