175         /**
176          * Class constructor
177          *
178          * @param mixed $paper The size of paper to use in this PDF ({@link CPDF::$PAPER_SIZES})
179          * @param string $orientation The orientation of the document (either 'landscape' or 'portrait')
180          * @param Dompdf $dompdf The Dompdf instance
181          */
182         public function __construct($paper = "letter", $orientation = "portrait", Dompdf $dompdf)
183         {
184             if (is_array($paper)) {
185                 $size = $paper;
186             } else if (isset(self::$PAPER_SIZES[mb_strtolower($paper)])) {
187                 $size = self::$PAPER_SIZES[mb_strtolower($paper)];
188             } else {
189                 $size = self::$PAPER_SIZES["letter"];
190             }
191     
192             if (mb_strtolower($orientation) === "landscape") {
193                 [$size[2], $size[3]] = [$size[3], $size[2]];
194             }
195     
196             $this->_dompdf = $dompdf;
197     
198             $this->_pdf = new \Dompdf\Cpdf(
199                 $size,
200                 true,
201                 $dompdf->getOptions()->getFontCache(),
202                 $dompdf->getOptions()->getTempDir()
203             );
204     
205             $this->_pdf->addInfo("Producer", sprintf("%s + CPDF", $dompdf->version));
206             $time = substr_replace(date('YmdHisO'), '\'', -2, 0) . '\'';
207             $this->_pdf->addInfo("CreationDate", "D:$time");
208             $this->_pdf->addInfo("ModDate", "D:$time");
209     
210             $this->_width = $size[2] - $size[0];
211             $this->_height = $size[3] - $size[1];
212     
213             $this->_page_number = $this->_page_count = 1;
214             $this->_page_text = [];

229         /**
230          * Class destructor
231          *
232          * Deletes all temporary image files
233          */
234         public function __destruct()
235         {
236             foreach ($this->_image_cache as $img) {
237                 // The file might be already deleted by 3rd party tmp cleaner,
238                 // the file might not have been created at all
239                 // (if image outputting commands failed)
240                 // or because the destructor was called twice accidentally.
241                 if (!file_exists($img)) {
242                     continue;
243                 }
244     
245                 if ($this->_dompdf->getOptions()->getDebugPng()) {
246                     print '[__destruct unlink ' . $img . ']';
247                 }
248                 if (!$this->_dompdf->getOptions()->getDebugKeepTemp()) {
249                     unlink($img);
250                 }
251             }

603         /**
604          * Convert a GIF or BMP image to a PNG image
605          *
606          * @param string $image_url
607          * @param integer $type
608          *
609          * @throws Exception
610          * @return string The url of the newly converted image
611          */
612         protected function _convert_gif_bmp_to_png($image_url, $type)
613         {
614             $func_name = "imagecreatefrom$type";
615     
616             if (!function_exists($func_name)) {
617                 if (!method_exists(Helpers::class, $func_name)) {
618                     throw new Exception("Function $func_name() not found.  Cannot convert $type image: $image_url.  Please install the image PHP extension.");
619                 }
620                 $func_name = "\\Dompdf\\Helpers::" . $func_name;
621             }
622     
623             set_error_handler([Helpers::class, 'record_warnings']);
624     
625             try {
626                 $im = call_user_func($func_name, $image_url);
627     
628                 if ($im) {
629                     imageinterlace($im, false);
630     
631                     $tmp_dir = $this->_dompdf->getOptions()->getTempDir();
632                     $tmp_name = @tempnam($tmp_dir, "{$type}dompdf_img_");
633                     @unlink($tmp_name);
634                     $filename = "$tmp_name.png";
635                     $this->_image_cache[] = $filename;
636     
637                     imagepng($im, $filename);
638                     imagedestroy($im);
639                 } else {
640                     $filename = Cache::$broken_image;
641                 }
642             } finally {
643                 restore_error_handler();
644             }
645     
646             return $filename;
647         }

833          * @param float $x
834          * @param float $y
835          * @param int $w
836          * @param int $h
837          * @param string $resolution
838          */
839         public function image($img, $x, $y, $w, $h, $resolution = "normal")
840         {
841             [$width, $height, $type] = Helpers::dompdf_getimagesize($img, $this->get_dompdf()->getHttpContext());
842     
843             $debug_png = $this->_dompdf->getOptions()->getDebugPng();
844     
845             if ($debug_png) {
846                 print "[image:$img|$width|$height|$type]";
847             }
848     
849             switch ($type) {
850                 case "jpeg":
851                     if ($debug_png) {
852                         print '!!!jpg!!!';
853                     }
854                     $this->_pdf->addJpegFromFile($img, $x, $this->y($y) - $h, $w, $h);
855                     break;
856     
857                 case "gif":
858                 /** @noinspection PhpMissingBreakStatementInspection */
859                 case "bmp":
860                     if ($debug_png) print '!!!bmp or gif!!!';
861                     // @todo use cache for BMP and GIF
862                     $img = $this->_convert_gif_bmp_to_png($img, $type);
863     
864                 case "png":
865                     if ($debug_png) print '!!!png!!!';
866     
867                     $this->_pdf->addPngFromFile($img, $x, $this->y($y) - $h, $w, $h);
868                     break;
869     
870                 case "svg":
871                     if ($debug_png) print '!!!SVG!!!';
872     
873                     $this->_pdf->addSvgFromFile($img, $x, $this->y($y) - $h, $w, $h);
874                     break;
875     
876                 default:
877                     if ($debug_png) print '!!!unknown!!!';
878             }
879         }

1086         /**
1087          * Add text to each page after rendering is complete
1088          */
1089         protected function _add_page_text()
1090         {
1091             if (!count($this->_page_text)) {
1092                 return;
1093             }
1094     
1095             $page_number = 1;
1096             $eval = null;
1097     
1098             foreach ($this->_pages as $pid) {
1099                 $this->reopen_object($pid);
1100     
1101                 foreach ($this->_page_text as $pt) {
1102                     extract($pt);
1103     
1104                     switch ($_t) {
1105                         case "text":
1106                             $text = str_replace(["{PAGE_NUM}", "{PAGE_COUNT}"],
1107                                 [$page_number, $this->_page_count], $text);
1108                             $this->text($x, $y, $text, $font, $size, $color, $word_space, $char_space, $angle);
1109                             break;
1110     
1111                         case "script":
1112                             if (!$eval) {
1113                                 $eval = new PhpEvaluator($this);
1114                             }
1115                             $eval->evaluate($code, ['PAGE_NUM' => $page_number, 'PAGE_COUNT' => $this->_page_count]);
1116                             break;
1117     
1118                         case 'line':
1119                             $this->line( $x1, $y1, $x2, $y2, $color, $width, $style );
1120                             break;
1121                     }
1122                 }
1123     
1124                 $this->close_object();
1125                 $page_number++;
1126             }
1127         }

1129         /**
1130          * Streams the PDF to the client.
1131          *
1132          * @param string $filename The filename to present to the client.
1133          * @param array $options Associative array: 'compress' => 1 or 0 (default 1); 'Attachment' => 1 or 0 (default 1).
1134          */
1135         public function stream($filename = "document.pdf", $options = [])
1136         {
1137             if (headers_sent()) {
1138                 die("Unable to stream pdf: headers already sent");
1139             }
1140     
1141             if (!isset($options["compress"])) $options["compress"] = true;
1142             if (!isset($options["Attachment"])) $options["Attachment"] = true;
1143     
1144             $this->_add_page_text();
1145     
1146             $debug = !$options['compress'];
1147             $tmp = ltrim($this->_pdf->output($debug));
1148     
1149             header("Cache-Control: private");
1150             header("Content-Type: application/pdf");
1151             header("Content-Length: " . mb_strlen($tmp, "8bit"));
1152     
1153             $filename = str_replace(["\n", "'"], "", basename($filename, ".pdf")) . ".pdf";
1154             $attachment = $options["Attachment"] ? "attachment" : "inline";
1155             header(Helpers::buildContentDispositionHeader($attachment, $filename));
1156     
1157             echo $tmp;
1158             flush();
1159         }
