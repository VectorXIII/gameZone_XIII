153         /**
154          * Construct a Configuration object from Symfony Console input.
155          *
156          * This is great for adding psysh-compatible command line options to framework- or app-specific
157          * wrappers.
158          *
159          * $input should already be bound to an appropriate InputDefinition (see self::getInputOptions
160          * if you want to build your own) before calling this method. It's not required, but things work
161          * a lot better if we do.
162          *
163          * @see self::getInputOptions
164          *
165          * @throws \InvalidArgumentException
166          *
167          * @param InputInterface $input
168          *
169          * @return self
170          */
171         public static function fromInput(InputInterface $input)
172         {
173             $config = new self(['configFile' => self::getConfigFileFromInput($input)]);
174     
175             // Handle --color and --no-color (and --ansi and --no-ansi aliases)
176             if (self::getOptionFromInput($input, ['color', 'ansi'])) {
177                 $config->setColorMode(self::COLOR_MODE_FORCED);
178             } elseif (self::getOptionFromInput($input, ['no-color', 'no-ansi'])) {
179                 $config->setColorMode(self::COLOR_MODE_DISABLED);
180             }
181     
182             // Handle verbosity options
183             if ($verbosity = self::getVerbosityFromInput($input)) {
184                 $config->setVerbosity($verbosity);
185             }
186     
187             // Handle interactive mode
188             if (self::getOptionFromInput($input, ['interactive', 'interaction'], ['-a', '-i'])) {
189                 $config->setInteractiveMode(self::INTERACTIVE_MODE_FORCED);
190             } elseif (self::getOptionFromInput($input, ['no-interactive', 'no-interaction'], ['-n'])) {
191                 $config->setInteractiveMode(self::INTERACTIVE_MODE_DISABLED);
192             }
193     
194             // Handle --raw-output
195             // @todo support raw output with interactive input?
196             if (!$config->getInputInteractive()) {
197                 if (self::getOptionFromInput($input, ['raw-output'], ['-r'])) {
198                     $config->setRawOutput(true);
199                 }
200             }
201     
202             return $config;
203         }

220         /**
221          * Get a boolean option from the given input.
222          *
223          * This helper allows fallback for unbound and unvalidated input. It's not perfect--for example,
224          * it can't deal with several short options squished together--but it's better than falling over
225          * any time someone gives us unbound input.
226          *
227          * @return bool true if the option (or an alias) is present
228          */
229         private static function getOptionFromInput(InputInterface $input, array $names, array $otherParams = [])
230         {
231             // Best case, input is properly bound and validated.
232             foreach ($names as $name) {
233                 if ($input->hasOption($name) && $input->getOption($name)) {
234                     return true;
235                 }
236             }
237     
238             foreach ($names as $name) {
239                 $otherParams[] = '--' . $name;
240             }
241     
242             foreach ($otherParams as $name) {
243                 if ($input->hasParameterOption($name, true)) {
244                     return true;
245                 }
246             }
247     
248             return false;
249         }

251         /**
252          * Get the desired verbosity from the given input.
253          *
254          * This is a bit more complext than the other options parsers. It handles `--quiet` and
255          * `--verbose`, along with their short aliases, and fancy things like `-vvv`.
256          *
257          * @return string|null configuration constant, or null if no verbosity option is specified
258          */
259         private static function getVerbosityFromInput(InputInterface $input)
260         {
261             // --quiet wins!
262             if (self::getOptionFromInput($input, ['quiet'], ['-q'])) {
263                 return self::VERBOSITY_QUIET;
264             }
265     
266             // Best case, input is properly bound and validated.
267             //
268             // Note that if the `--verbose` option is incorrectly defined as `VALUE_NONE` rather than
269             // `VALUE_OPTIONAL` (as it is in Symfony Console by default) it doesn't actually work with
270             // multiple verbosity levels as it claims.
271             //
272             // We can detect this by checking whether the the value === true, and fall back to unbound
273             // parsing for this option.
274             if ($input->hasOption('verbose') && $input->getOption('verbose') !== true) {
275                 switch ($input->getOption('verbose')) {
276                     case '-1':
277                         return self::VERBOSITY_QUIET;
278                     case '0': // explicitly normal, overrides config file default
279                         return self::VERBOSITY_NORMAL;
280                     case '1':
281                     case null: // `--verbose` and `-v`
282                         return self::VERBOSITY_VERBOSE;
283                     case '2':
284                     case 'v': // `-vv`
285                         return self::VERBOSITY_VERY_VERBOSE;
286                     case '3':
287                     case 'vv': // `-vvv`
288                         return self::VERBOSITY_DEBUG;
289                     default: // implicitly normal, config file default wins
290                         return;
291                 }
292             }
293     
294             // quiet and normal have to come before verbose, because it eats everything else.
295             if ($input->hasParameterOption('--verbose=-1', true) || $input->getParameterOption('--verbose', false, true) === '-1') {
296                 return self::VERBOSITY_QUIET;
297             }
298     
299             if ($input->hasParameterOption('--verbose=0', true) || $input->getParameterOption('--verbose', false, true) === '0') {
300                 return self::VERBOSITY_NORMAL;
301             }
302     
303             // `-vvv`, `-vv` and `-v` have to come in descending length order, because `hasParameterOption` matches prefixes.
304             if ($input->hasParameterOption('-vvv', true) || $input->hasParameterOption('--verbose=3', true) || $input->getParameterOption('--verbose', false, true) === '3') {
305                 return self::VERBOSITY_DEBUG;
306             }
307     
308             if ($input->hasParameterOption('-vv', true) || $input->hasParameterOption('--verbose=2', true) || $input->getParameterOption('--verbose', false, true) === '2') {
309                 return self::VERBOSITY_VERY_VERBOSE;
310             }
311     
312             if ($input->hasParameterOption('-v', true) || $input->hasParameterOption('--verbose=1', true) || $input->hasParameterOption('--verbose', true)) {
313                 return self::VERBOSITY_VERBOSE;
314             }
315         }

421         /**
422          * Load configuration values from an array of options.
423          *
424          * @param array $options
425          */
426         public function loadConfig(array $options)
427         {
428             foreach (self::$AVAILABLE_OPTIONS as $option) {
429                 if (isset($options[$option])) {
430                     $method = 'set' . \ucfirst($option);
431                     $this->$method($options[$option]);
432                 }
433             }
434     
435             // legacy `tabCompletion` option
436             if (isset($options['tabCompletion'])) {
437                 $msg = '`tabCompletion` is deprecated; use `useTabCompletion` instead.';
438                 @\trigger_error($msg, E_USER_DEPRECATED);
439     
440                 $this->setUseTabCompletion($options['tabCompletion']);
441             }
442     
443             foreach (['commands', 'matchers', 'casters'] as $option) {
444                 if (isset($options[$option])) {
445                     $method = 'add' . \ucfirst($option);
446                     $this->$method($options[$option]);
447                 }
448             }
449     
450             // legacy `tabCompletionMatchers` option
451             if (isset($options['tabCompletionMatchers'])) {
452                 $msg = '`tabCompletionMatchers` is deprecated; use `matchers` instead.';
453                 @\trigger_error($msg, E_USER_DEPRECATED);
454     
455                 $this->addMatchers($options['tabCompletionMatchers']);
456             }
457         }

459         /**
460          * Load a configuration file (default: `$HOME/.config/psysh/config.php`).
461          *
462          * This configuration instance will be available to the config file as $config.
463          * The config file may directly manipulate the configuration, or may return
464          * an array of options which will be merged with the current configuration.
465          *
466          * @throws \InvalidArgumentException if the config file does not exist or returns a non-array result
467          *
468          * @param string $file
469          */
470         public function loadConfigFile($file)
471         {
472             if (!\is_file($file)) {
473                 throw new \InvalidArgumentException(\sprintf('Invalid configuration file specified, %s does not exist', $file));
474             }
475     
476             $__psysh_config_file__ = $file;
477             $load = function ($config) use ($__psysh_config_file__) {
478                 $result = require $__psysh_config_file__;
479                 if ($result !== 1) {
480                     return $result;
481                 }
482             };
483             $result = $load($this);
484     
485             if (!empty($result)) {
486                 if (\is_array($result)) {
487                     $this->loadConfig($result);
488                 } else {
489                     throw new \InvalidArgumentException('Psy Shell configuration must return an array of options');
490                 }
491             }
492         }

615                     $msg = \sprintf('Multiple history files found: %s. Using %s', \implode(', ', $files), $files[0]);
616                     \trigger_error($msg, E_USER_NOTICE);

619                 $this->setHistoryFile($files[0]);
620             } else {
621                 // fallback: create our own history file
622                 $dir = $this->configDir ?: ConfigPaths::getCurrentConfigDir();
623                 $this->setHistoryFile($dir . '/psysh_history');
624             }
625

1743         /**
1744          * Guess whether an input or output stream is piped.
1745          *
1746          * @param resource|int $stream
1747          *
1748          * @return bool
1749          */
1750         private static function looksLikeAPipe($stream)
1751         {
1752             if (\function_exists('posix_isatty')) {
1753                 return !\posix_isatty($stream);
1754             }
1755     
1756             $stat = \fstat($stream);
1757             $mode = $stat['mode'] & 0170000;
1758     
1759             return $mode === 0010000 || $mode === 0040000 || $mode === 0100000 || $mode === 0120000;
1760         }
