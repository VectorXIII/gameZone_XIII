156         /**
157          * Normalizes a type: Converts plain-text type names into proper AST representation.
158          *
159          * In particular, builtin types become Identifiers, custom types become Names and nullables
160          * are wrapped in NullableType nodes.
161          *
162          * @param string|Name|Identifier|NullableType|UnionType $type The type to normalize
163          *
164          * @return Name|Identifier|NullableType|UnionType The normalized type
165          */
166         public static function normalizeType($type) {
167             if (!is_string($type)) {
168                 if (
169                     !$type instanceof Name && !$type instanceof Identifier &&
170                     !$type instanceof NullableType && !$type instanceof UnionType
171                 ) {
172                     throw new \LogicException(
173                         'Type must be a string, or an instance of Name, Identifier, NullableType or UnionType'
174                     );
175                 }
176                 return $type;
177             }
178     
179             $nullable = false;
180             if (strlen($type) > 0 && $type[0] === '?') {
181                 $nullable = true;
182                 $type = substr($type, 1);
183             }
184     
185             $builtinTypes = [
186                 'array', 'callable', 'string', 'int', 'float', 'bool', 'iterable', 'void', 'object'
187             ];
188     
189             $lowerType = strtolower($type);
190             if (in_array($lowerType, $builtinTypes)) {
191                 $type = new Identifier($lowerType);
192             } else {
193                 $type = self::normalizeName($type);
194             }
195     
196             if ($nullable && (string) $type === 'void') {
197                 throw new \LogicException('void type cannot be nullable');
198             }
199     
200             return $nullable ? new NullableType($type) : $type;
201         }

203         /**
204          * Normalizes a value: Converts nulls, booleans, integers,
205          * floats, strings and arrays into their respective nodes
206          *
207          * @param Node\Expr|bool|null|int|float|string|array $value The value to normalize
208          *
209          * @return Expr The normalized value
210          */
211         public static function normalizeValue($value) : Expr {
212             if ($value instanceof Node\Expr) {
213                 return $value;
214             } elseif (is_null($value)) {
215                 return new Expr\ConstFetch(
216                     new Name('null')
217                 );
218             } elseif (is_bool($value)) {
219                 return new Expr\ConstFetch(
220                     new Name($value ? 'true' : 'false')
221                 );
222             } elseif (is_int($value)) {
223                 return new Scalar\LNumber($value);
224             } elseif (is_float($value)) {
225                 return new Scalar\DNumber($value);
226             } elseif (is_string($value)) {
227                 return new Scalar\String_($value);
228             } elseif (is_array($value)) {
229                 $items = [];
230                 $lastKey = -1;
231                 foreach ($value as $itemKey => $itemValue) {
232                     // for consecutive, numeric keys don't generate keys
233                     if (null !== $lastKey && ++$lastKey === $itemKey) {
234                         $items[] = new Expr\ArrayItem(
235                             self::normalizeValue($itemValue)
236                         );
237                     } else {
238                         $lastKey = null;
239                         $items[] = new Expr\ArrayItem(
240                             self::normalizeValue($itemValue),
241                             self::normalizeValue($itemKey)
242                         );
243                     }
244                 }
245     
246                 return new Expr\Array_($items);
247             } else {
248                 throw new \LogicException('Invalid value');
249             }
250         }
