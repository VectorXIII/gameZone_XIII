265         private static function filterExceptionArray(string $xClass, array $a, string $xPrefix, int $filter): array
266         {
267             if (isset($a[$xPrefix.'trace'])) {
268                 $trace = $a[$xPrefix.'trace'];
269                 unset($a[$xPrefix.'trace']); // Ensures the trace is always last
270             } else {
271                 $trace = [];
272             }
273     
274             if (!($filter & Caster::EXCLUDE_VERBOSE) && $trace) {
275                 if (isset($a[Caster::PREFIX_PROTECTED.'file'], $a[Caster::PREFIX_PROTECTED.'line'])) {
276                     self::traceUnshift($trace, $xClass, $a[Caster::PREFIX_PROTECTED.'file'], $a[Caster::PREFIX_PROTECTED.'line']);
277                 }
278                 $a[Caster::PREFIX_VIRTUAL.'trace'] = new TraceStub($trace, self::$traceArgs);
279             }
280             if (empty($a[$xPrefix.'previous'])) {
281                 unset($a[$xPrefix.'previous']);
282             }
283             unset($a[$xPrefix.'string'], $a[Caster::PREFIX_DYNAMIC.'xdebug_message'], $a[Caster::PREFIX_DYNAMIC.'__destructorException']);
284     
285             if (isset($a[Caster::PREFIX_PROTECTED.'message']) && false !== strpos($a[Caster::PREFIX_PROTECTED.'message'], "class@anonymous\0")) {
286                 $a[Caster::PREFIX_PROTECTED.'message'] = preg_replace_callback('/class@anonymous\x00.*?\.php(?:0x?|:[0-9]++\$)[0-9a-fA-F]++/', function ($m) {
287                     return class_exists($m[0], false) ? get_parent_class($m[0]).'@anonymous' : $m[0];
288                 }, $a[Caster::PREFIX_PROTECTED.'message']);
289             }
290     
291             if (isset($a[Caster::PREFIX_PROTECTED.'file'], $a[Caster::PREFIX_PROTECTED.'line'])) {
292                 $a[Caster::PREFIX_PROTECTED.'file'] = new LinkStub($a[Caster::PREFIX_PROTECTED.'file'], $a[Caster::PREFIX_PROTECTED.'line']);
293             }
294     
295             return $a;
296         }

298         private static function traceUnshift(array &$trace, ?string $class, string $file, int $line): void
299         {
300             if (isset($trace[0]['file'], $trace[0]['line']) && $trace[0]['file'] === $file && $trace[0]['line'] === $line) {
301                 return;
302             }
303             array_unshift($trace, [
304                 'function' => $class ? 'new '.$class : null,
305                 'file' => $file,
306                 'line' => $line,
307             ]);
308         }

310         private static function extractSource(string $srcLines, int $line, int $srcContext, string $lang, ?string $file, array $frame): EnumStub
311         {
312             $srcLines = explode("\n", $srcLines);
313             $src = [];
314     
315             for ($i = $line - 1 - $srcContext; $i <= $line - 1 + $srcContext; ++$i) {
316                 $src[] = (isset($srcLines[$i]) ? $srcLines[$i] : '')."\n";
317             }
318     
319             if ($frame['function'] ?? false) {
320                 $stub = new CutStub(new \stdClass());
321                 $stub->class = (isset($frame['class']) ? $frame['class'].$frame['type'] : '').$frame['function'];
322                 $stub->type = Stub::TYPE_OBJECT;
323                 $stub->attr['cut_hash'] = true;
324                 $stub->attr['file'] = $frame['file'];
325                 $stub->attr['line'] = $frame['line'];
326     
327                 try {
328                     $caller = isset($frame['class']) ? new \ReflectionMethod($frame['class'], $frame['function']) : new \ReflectionFunction($frame['function']);
329                     $stub->class .= ReflectionCaster::getSignature(ReflectionCaster::castFunctionAbstract($caller, [], $stub, true, Caster::EXCLUDE_VERBOSE));
330     
331                     if ($f = $caller->getFileName()) {
332                         $stub->attr['file'] = $f;
333                         $stub->attr['line'] = $caller->getStartLine();
334                     }
335                 } catch (\ReflectionException $e) {
336                     // ignore fake class/function
337                 }
338     
339                 $srcLines = ["\0~separator=\0" => $stub];
340             } else {
341                 $stub = null;
342                 $srcLines = [];
343             }
344     
345             $ltrim = 0;
346             do {
347                 $pad = null;
348                 for ($i = $srcContext << 1; $i >= 0; --$i) {
349                     if (isset($src[$i][$ltrim]) && "\r" !== ($c = $src[$i][$ltrim]) && "\n" !== $c) {
350                         if (null === $pad) {
351                             $pad = $c;
352                         }
353                         if ((' ' !== $c && "\t" !== $c) || $pad !== $c) {
354                             break;
355                         }
356                     }
357                 }
358                 ++$ltrim;
359             } while (0 > $i && null !== $pad);
360     
361             --$ltrim;
362     
363             foreach ($src as $i => $c) {
364                 if ($ltrim) {
365                     $c = isset($c[$ltrim]) && "\r" !== $c[$ltrim] ? substr($c, $ltrim) : ltrim($c, " \t");
366                 }
367                 $c = substr($c, 0, -1);
368                 if ($i !== $srcContext) {
369                     $c = new ConstStub('default', $c);
370                 } else {
371                     $c = new ConstStub($c, $stub ? 'in '.$stub->class : '');
372                     if (null !== $file) {
373                         $c->attr['file'] = $file;
374                         $c->attr['line'] = $line;
375                     }
376                 }
377                 $c->attr['lang'] = $lang;
378                 $srcLines[sprintf("\0~separator=â€º &%d\0", $i + $line - $srcContext)] = $c;
379             }
380     
381             return new EnumStub($srcLines);
382         }
