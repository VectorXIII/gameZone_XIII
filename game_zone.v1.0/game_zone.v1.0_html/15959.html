12     namespace Symfony\Component\EventDispatcher;
13     
14     use Symfony\Contracts\EventDispatcher\EventDispatcherInterface as ContractsEventDispatcherInterface;
15     
16     /**
17      * The EventDispatcherInterface is the central point of Symfony's event listener system.
18      * Listeners are registered on the manager and events are dispatched through the
19      * manager.
20      *
21      * @author Bernhard Schussek <bschussek@gmail.com>
22      */
23     interface EventDispatcherInterface extends ContractsEventDispatcherInterface
24     {
25         /**
26          * Adds an event listener that listens on the specified events.
27          *
28          * @param callable $listener The listener
29          * @param int      $priority The higher this value, the earlier an event
30          *                           listener will be triggered in the chain (defaults to 0)
31          */
32         public function addListener(string $eventName, $listener, int $priority = 0);
33     
34         /**
35          * Adds an event subscriber.
36          *
37          * The subscriber is asked for all the events it is
38          * interested in and added as a listener for these events.
39          */
40         public function addSubscriber(EventSubscriberInterface $subscriber);
41     
42         /**
43          * Removes an event listener from the specified events.
44          *
45          * @param callable $listener The listener to remove
46          */
47         public function removeListener(string $eventName, $listener);
48     
49         public function removeSubscriber(EventSubscriberInterface $subscriber);
50     
51         /**
52          * Gets the listeners of a specific event or all listeners sorted by descending priority.
53          *
54          * @return array The event listeners for the specified event, or all event listeners by event name
55          */
56         public function getListeners(string $eventName = null);
57     
58         /**
59          * Gets the listener priority for a specific event.
60          *
61          * Returns null if the event or the listener does not exist.
62          *
63          * @param callable $listener The listener
64          *
65          * @return int|null The event listener priority
66          */
67         public function getListenerPriority(string $eventName, $listener);
68     
69         /**
70          * Checks whether an event has any registered listeners.
71          *
72          * @return bool true if the specified event has any listeners, false otherwise
73          */
74         public function hasListeners(string $eventName = null);
75     }
