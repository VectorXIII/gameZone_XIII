675          * replaced with their closest ASCII counterparts, and the rest are removed
676          * by default. The language or locale of the source string can be supplied
677          * for language-specific transliteration in any of the following formats:
678          * en, en_GB, or en-GB. For example, passing "de" results in "Ã¤Ã¶Ã¼" mapping

682          * @param string    $language                  [optional] <p>Language of the source string.
683          *                                             (default is 'en') | ASCII::*_LANGUAGE_CODE</p>

685          *                                             unsupported characters.</p>

693          *
694          * @psalm-pure
695          *
696          * @return string
697          *                <p>A string that contains only ASCII characters.</p>
698          */
699         public static function to_ascii(
700             string $str,
701             string $language = self::ENGLISH_LANGUAGE_CODE,
702             bool $remove_unsupported_chars = true,
703             bool $replace_extra_symbols = false,
704             bool $use_transliterate = false,
705             bool $replace_single_chars_only = null

708                 return '';

711             $language = self::get_language($language);

716              * @psalm-suppress ImpureStaticVariable
717              *
718              * @var array<string,array<string,string>>

723             if (!isset($REPLACE_HELPER_CACHE[$cacheKey])) {
724                 $langAll = self::charsArrayWithSingleLanguageValues($replace_extra_symbols, false);
725     
726                 $langSpecific = self::charsArrayWithOneLanguage($language, $replace_extra_symbols, false);
727     
728                 $REPLACE_HELPER_CACHE[$cacheKey] = \array_merge([], $langAll, $langSpecific);
729             }
730     
731             if (
732                 $replace_extra_symbols
733                 &&
734                 $EXTRA_SYMBOLS_CACHE === null
735             ) {
736                 $EXTRA_SYMBOLS_CACHE = [];
737                 foreach (self::$ASCII_EXTRAS ?? [] as $extrasLanguageTmp => $extrasDataTmp) {
738                     foreach ($extrasDataTmp as $extrasDataKeyTmp => $extrasDataValueTmp) {
739                         $EXTRA_SYMBOLS_CACHE[$extrasDataKeyTmp] = $extrasDataKeyTmp;
740                     }
741                 }
742                 $EXTRA_SYMBOLS_CACHE = \implode('',$EXTRA_SYMBOLS_CACHE);
743             }

745             $charDone = [];
746             if (\preg_match_all('/' . self::$REGEX_ASCII . ($replace_extra_symbols ? '|[' . $EXTRA_SYMBOLS_CACHE . ']' : '') . '/u', $str, $matches)) {
747     
748                 if (!$replace_single_chars_only) {
749                     foreach ($matches[0] as $keyTmp => $char) {
750                         if (isset($matches[0][$keyTmp + 2])) {
751                             $threeChars = $matches[0][$keyTmp + 0] . $matches[0][$keyTmp + 1] . $matches[0][$keyTmp + 2];
752                         } else {
753                             $threeChars = null;
754                         }
755                         if (
756                             $threeChars
757                             &&
758                             !isset($charDone[$threeChars])
759                             &&
760                             isset($REPLACE_HELPER_CACHE[$cacheKey][$threeChars])
761                             &&
762                             strpos($str, $threeChars) !== false
763                         ) {
764                             // DEBUG
765                             //\var_dump($str, $threeChars, $REPLACE_HELPER_CACHE[$cacheKey][$threeChars]);
766     
767                             $charDone[$threeChars] = true;
768                             $str = \str_replace($threeChars, $REPLACE_HELPER_CACHE[$cacheKey][$threeChars], $str);
769     
770                             // DEBUG
771                             //\var_dump($str, "\n");
772                         }
773                     }
774     
775                     foreach ($matches[0] as $keyTmp => $char) {
776                         if (isset($matches[0][$keyTmp + 1])) {
777                             $twoChars = $matches[0][$keyTmp + 0] . $matches[0][$keyTmp + 1];
778                         } else {
779                             $twoChars = null;
780                         }
781                         if (
782                             $twoChars
783                             &&
784                             !isset($charDone[$twoChars])
785                             &&
786                             isset($REPLACE_HELPER_CACHE[$cacheKey][$twoChars])
787                             &&
788                             strpos($str, $twoChars) !== false
789                         ) {
790                             // DEBUG
791                             //\var_dump($str, $twoChars, $REPLACE_HELPER_CACHE[$cacheKey][$twoChars]);
792     
793                             $charDone[$twoChars] = true;
794                             $str = \str_replace($twoChars, $REPLACE_HELPER_CACHE[$cacheKey][$twoChars], $str);
795     
796                             // DEBUG
797                             //\var_dump($str, "\n");
798                         }
799                     }
800                 }
801     
802                 foreach ($matches[0] as $keyTmp => $char) {
803                     if (
804                         !isset($charDone[$char])
805                         &&
806                         isset($REPLACE_HELPER_CACHE[$cacheKey][$char])
807                         &&
808                         strpos($str, $char) !== false
809                     ) {
810                         // DEBUG
811                         //\var_dump($str, $char, $REPLACE_HELPER_CACHE[$cacheKey][$char]);
812     
813                         $charDone[$char] = true;
814                         $str = \str_replace($char, $REPLACE_HELPER_CACHE[$cacheKey][$char], $str);
815     
816                         // DEBUG
817                         //\var_dump($str, "\n");
818                     }
819                 }
820     
821             }
822     
823             /** @psalm-suppress PossiblyNullOperand - we use the prepare* methods here, so we don't get NULL here */
824             if (!isset(self::$ASCII_MAPS[$language])) {
825                 $use_transliterate = true;
826             }
827     
828             if ($use_transliterate) {
829                 /** @noinspection ArgumentEqualsDefaultValueInspection */
830                 $str = self::to_transliterate($str, null, false);
831             }
832     
833             if ($remove_unsupported_chars) {
834                 $str = (string) \str_replace(["\n\r", "\n", "\r", "\t"], ' ', $str);
835                 $str = (string) \preg_replace('/' . self::$REGEX_ASCII . '/', '', $str);
836             }
837     
838             return $str;
839         }

882         /**
883          * Converts the string into an URL slug. This includes replacing non-ASCII
884          * characters with their closest ASCII equivalents, removing remaining
885          * non-ASCII and non-alphanumeric characters, and replacing whitespace with
886          * $separator. The separator defaults to a single dash, and the string
887          * is also converted to lowercase. The language of the source string can
888          * also be supplied for language-specific transliteration.
889          *
890          * @param string                $str
891          * @param string                $separator             [optional] <p>The string used to replace whitespace.</p>
892          * @param string                $language              [optional] <p>Language of the source string.
893          *                                                     (default is 'en') | ASCII::*_LANGUAGE_CODE</p>
894          * @param array<string, string> $replacements          [optional] <p>A map of replaceable strings.</p>
895          * @param bool                  $replace_extra_symbols [optional]  <p>Add some more replacements e.g. "Â£" with "
896          *                                                     pound ".</p>
897          * @param bool                  $use_str_to_lower      [optional] <p>Use "string to lower" for the input.</p>
898          * @param bool                  $use_transliterate     [optional]  <p>Use ASCII::to_transliterate() for unknown
899          *                                                     chars.</p>
900          * @psalm-pure
901          *
902          * @return string
903          *                <p>A string that has been converted to an URL slug.</p>
904          */
905         public static function to_slugify(
906             string $str,
907             string $separator = '-',
908             string $language = self::ENGLISH_LANGUAGE_CODE,
909             array $replacements = [],
910             bool $replace_extra_symbols = false,
911             bool $use_str_to_lower = true,
912             bool $use_transliterate = false
913         ): string {
914             if ($str === '') {
915                 return '';
916             }
917     
918             foreach ($replacements as $from => $to) {
919                 $str = \str_replace($from, $to, $str);
920             }
921     
922             $str = self::to_ascii(
923                 $str,
924                 $language,
925                 false,
926                 $replace_extra_symbols,
927                 $use_transliterate
928             );
929     
930             $str = \str_replace('@', $separator, $str);
931     
932             $str = (string) \preg_replace(
933                 '/[^a-zA-Z\\d\\s\\-_' . \preg_quote($separator, '/') . ']/',
934                 '',
935                 $str
936             );
937     
938             if ($use_str_to_lower) {
939                 $str = \strtolower($str);
940             }
941     
942             $str = (string) \preg_replace('/^[\'\\s]+|[\'\\s]+$/', '', $str);
943             $str = (string) \preg_replace('/\\B([A-Z])/', '-\1', $str);
944             $str = (string) \preg_replace('/[\\-_\\s]+/', $separator, $str);
945     
946             $l = \strlen($separator);
947             if ($l && \strpos($str, $separator) === 0) {
948                 $str = (string) \substr($str, $l);
949             }
950     
951             if (\substr($str, -$l) === $separator) {
952                 $str = (string) \substr($str, 0, \strlen($str) - $l);
953             }
954     
955             return $str;
956         }

959          * Returns an ASCII version of the string. A set of non-ASCII characters are
960          * replaced with their closest ASCII counterparts, and the rest are removed
961          * unless instructed otherwise.
962          *
963          * @param string      $str     <p>The input string.</p>
964          * @param string|null $unknown [optional] <p>Character use if character unknown. (default is '?')
965          *                             But you can also use NULL to keep the unknown chars.</p>
966          * @param bool        $strict  [optional] <p>Use "transliterator_transliterate()" from PHP-Intl

972          *
973          * @noinspection ParameterDefaultValueIsNotNullInspection
974          */
975         public static function to_transliterate(
976             string $str,
977             $unknown = '?',
978             bool $strict = false
979         ): string {
980             /**
981              * @psalm-suppress ImpureStaticVariable
982              *
983              * @var array<int,string>|null
984              */
985             static $UTF8_TO_TRANSLIT = null;
986     
987             /**
988              * @psalm-suppress ImpureStaticVariable
989              *
990              * null|\Transliterator
991              */
992             static $TRANSLITERATOR = null;
993     
994             /**
995              * @psalm-suppress ImpureStaticVariable
996              *
997              * @var bool|null
998              */
999             static $SUPPORT_INTL = null;
1000     
1001             if ($str === '') {
1002                 return '';
1003             }
1004     
1005             if ($SUPPORT_INTL === null) {
1006                 $SUPPORT_INTL = \extension_loaded('intl');
1007             }
1008     
1009             // check if we only have ASCII, first (better performance)
1010             $str_tmp = $str;
1011             if (self::is_ascii($str)) {
1012                 return $str;
1013             }
1014     
1015             $str = self::clean($str);
1016     
1017             // check again, if we only have ASCII, now ...
1018             if (
1019                 $str_tmp !== $str
1020                 &&
1021                 self::is_ascii($str)
1022             ) {
1023                 return $str;
1024             }
1025     
1026             if (
1027                 $strict
1028                 &&
1029                 $SUPPORT_INTL === true
1030             ) {
1031                 if (!isset($TRANSLITERATOR)) {
1032                     // INFO: see "*-Latin" rules via "transliterator_list_ids()"
1033                     /**
1034                      * @noinspection PhpComposerExtensionStubsInspection
1035                      *
1036                      * @var \Transliterator
1037                      */
1038                     $TRANSLITERATOR = \transliterator_create('NFKC; [:Nonspacing Mark:] Remove; NFKC; Any-Latin; Latin-ASCII;');
1039                 }
1040     
1041                 // INFO: https://unicode.org/cldr/utility/character.jsp
1042                 /** @noinspection PhpComposerExtensionStubsInspection */

1044     
1045                 if ($str_tmp !== false) {
1046     
1047                     // check again, if we only have ASCII, now ...
1048                     if (
1049                         $str_tmp !== $str
1050                         &&
1051                         self::is_ascii($str_tmp)
1052                     ) {

1058                 }
1059             }
1060     
1061             if (self::$ORD === null) {
1062                 self::$ORD = self::getData('ascii_ord');
1063             }
1064     
1065             \preg_match_all('/.|[^\x00]$/us', $str, $array_tmp);
1066             $chars = $array_tmp[0];
1067             $ord = null;
1068             $str_tmp = '';
1069             foreach ($chars as &$c) {
1070                 $ordC0 = self::$ORD[$c[0]];

1073                     $str_tmp .= $c;

1075                     continue;
1076                 }
1077     
1078                 $ordC1 = self::$ORD[$c[1]];
1079     
1080                 // ASCII - next please
1081                 if ($ordC0 >= 192 && $ordC0 <= 223) {
1082                     $ord = ($ordC0 - 192) * 64 + ($ordC1 - 128);
1083                 }
1084     
1085                 if ($ordC0 >= 224) {
1086                     $ordC2 = self::$ORD[$c[2]];
1087     
1088                     if ($ordC0 <= 239) {
1089                         $ord = ($ordC0 - 224) * 4096 + ($ordC1 - 128) * 64 + ($ordC2 - 128);
1090                     }
1091     
1092                     if ($ordC0 >= 240) {
1093                         $ordC3 = self::$ORD[$c[3]];
1094     
1095                         if ($ordC0 <= 247) {
1096                             $ord = ($ordC0 - 240) * 262144 + ($ordC1 - 128) * 4096 + ($ordC2 - 128) * 64 + ($ordC3 - 128);
1097                         }
1098     
1099                         // We only process valid UTF-8 chars (<= 4 byte), so we don't need this code here ...
1100                         /*
1101                         if ($ordC0 >= 248) {
1102                             $ordC4 = self::$ORD[$c[4]];
1103     
1104                             if ($ordC0 <= 251) {
1105                                 $ord = ($ordC0 - 248) * 16777216 + ($ordC1 - 128) * 262144 + ($ordC2 - 128) * 4096 + ($ordC3 - 128) * 64 + ($ordC4 - 128);
1106                             }
1107     
1108                             if ($ordC0 >= 252) {
1109                                 $ordC5 = self::$ORD[$c[5]];
1110     
1111                                 if ($ordC0 <= 253) {
1112                                     $ord = ($ordC0 - 252) * 1073741824 + ($ordC1 - 128) * 16777216 + ($ordC2 - 128) * 262144 + ($ordC3 - 128) * 4096 + ($ordC4 - 128) * 64 + ($ordC5 - 128);
1113                                 }
1114                             }
1115                         }
1116                          */
1117                     }
1118                 }
1119     
1120                 if (
1121                     $ordC0 === 254
1122                     ||
1123                     $ordC0 === 255
1124                     ||
1125                     $ord === null
1126                 ) {
1127                     $str_tmp .= $unknown ?? $c;
1128     
1129                     continue;
1130                 }
1131     
1132                 $bank = $ord >> 8;
1133                 if (!isset($UTF8_TO_TRANSLIT[$bank])) {
1134                     $UTF8_TO_TRANSLIT[$bank] = self::getDataIfExists(\sprintf('x%02x', $bank));
1135                 }
1136     
1137                 $new_char = $ord & 255;
1138     
1139                 if (isset($UTF8_TO_TRANSLIT[$bank][$new_char])) {
1140

1148                     echo "ascii: " . $UTF8_TO_TRANSLIT[$bank][$new_char] . "\n";
1149                     echo "bank:" . $bank . "\n\n";
1150                      */
1151     
1152                     $new_char = $UTF8_TO_TRANSLIT[$bank][$new_char];
1153     
1154                     if ($unknown === null && $new_char === '') {
1155                         $c = $unknown ?? $c;
1156                     } elseif (
1157                         $new_char === '[?]'
1158                         ||
1159                         $new_char === '[?] '
1160                     ) {
1161                         $c = $unknown ?? $c;
1162                     } else {
1163                         $c = $new_char;
1164                     }
1165                 } else {
1166     
1167                     // keep for debugging missing chars
1168                     /*
1169                     echo "file: " . sprintf('x%02x', $bank) . "\n";
1170                     echo "char: " . $c . "\n";
1171                     echo "ord: " . $ord . "\n";
1172                     echo "new_char: " . $new_char . "\n";
1173                     echo "new_char: " . mb_chr($new_char) . "\n";
1174                     echo "bank:" . $bank . "\n\n";
1175                      */
1176     
1177                     $c = $unknown ?? $c;
1178                 }
1179     
1180                 $str_tmp .= $c;
1181             }
1182     
1183             return $str_tmp;
1184         }
