607         /**
608          * Register a service provider with the application.
609          *
610          * @param  \Illuminate\Support\ServiceProvider|string  $provider
611          * @param  bool  $force
612          * @return \Illuminate\Support\ServiceProvider
613          */
614         public function register($provider, $force = false)
615         {
616             if (($registered = $this->getProvider($provider)) && ! $force) {
617                 return $registered;
618             }
619     
620             // If the given "provider" is a string, we will resolve it, passing in the
621             // application instance automatically for the developer. This is simply
622             // a more convenient way of specifying your service provider classes.
623             if (is_string($provider)) {
624                 $provider = $this->resolveProvider($provider);
625             }
626     
627             $provider->register();
628     
629             // If there are bindings / singletons set as properties on the provider we
630             // will spin through them and register them with the application, which
631             // serves as a convenience layer while registering a lot of bindings.
632             if (property_exists($provider, 'bindings')) {
633                 foreach ($provider->bindings as $key => $value) {
634                     $this->bind($key, $value);
635                 }
636             }
637     
638             if (property_exists($provider, 'singletons')) {
639                 foreach ($provider->singletons as $key => $value) {
640                     $this->singleton($key, $value);
641                 }
642             }
643     
644             $this->markAsRegistered($provider);
645     
646             // If the application has already booted, we will call this boot method on
647             // the provider class so it has an opportunity to do its boot logic and
648             // will be ready for any usage by this developer's application logic.
649             if ($this->isBooted()) {
650                 $this->bootProvider($provider);
651             }
652     
653             return $provider;
654         }

723         /**
724          * Load the provider for a deferred service.
725          *
726          * @param  string  $service
727          * @return void
728          */
729         public function loadDeferredProvider($service)
730         {
731             if (! $this->isDeferredService($service)) {
732                 return;
733             }
734     
735             $provider = $this->deferredServices[$service];
736     
737             // If the service provider has not already been loaded and registered we can
738             // register it with the application and remove the service from this list
739             // of deferred services, since it will already be loaded on subsequent.
740             if (! isset($this->loadedProviders[$provider])) {
741                 $this->registerDeferredProvider($provider, $service);
742             }
743         }

745         /**
746          * Register a deferred provider and service.
747          *
748          * @param  string  $provider
749          * @param  string|null  $service
750          * @return void
751          */
752         public function registerDeferredProvider($provider, $service = null)
753         {
754             // Once the provider that provides the deferred service has been registered we
755             // will remove it from our local list of the deferred services with related
756             // providers so that this container does not try to resolve it out again.
757             if ($service) {
758                 unset($this->deferredServices[$service]);
759             }
760     
761             $this->register($instance = new $provider($this));
762     
763             if (! $this->isBooted()) {
764                 $this->booting(function () use ($instance) {
765                     $this->bootProvider($instance);
766                 });
767             }
768         }

1280                 return $this->namespace;

1283             $composer = json_decode(file_get_contents($this->basePath('composer.json')), true);
1284

1288                         return $this->namespace = $namespace;
