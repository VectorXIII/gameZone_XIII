24     /**
25      * Dompdf - PHP5 HTML to PDF renderer
26      *
27      * Dompdf loads HTML and does its best to render it as a PDF.  It gets its
28      * name from the new DomDocument PHP5 extension.  Source HTML is first
29      * parsed by a DomDocument object.  Dompdf takes the resulting DOM tree and
30      * attaches a {@link Frame} object to each node.  {@link Frame} objects store
31      * positioning and layout information and each has a reference to a {@link
32      * Style} object.
33      *
34      * Style information is loaded and parsed (see {@link Stylesheet}) and is
35      * applied to the frames in the tree by using XPath.  CSS selectors are
36      * converted into XPath queries, and the computed {@link Style} objects are
37      * applied to the {@link Frame}s.
38      *
39      * {@link Frame}s are then decorated (in the design pattern sense of the
40      * word) based on their CSS display property ({@link
41      * http://www.w3.org/TR/CSS21/visuren.html#propdef-display}).
42      * Frame_Decorators augment the basic {@link Frame} class by adding
43      * additional properties and methods specific to the particular type of
44      * {@link Frame}.  For example, in the CSS layout model, block frames
45      * (display: block;) contain line boxes that are usually filled with text or
46      * other inline frames.  The Block therefore adds a $lines
47      * property as well as methods to add {@link Frame}s to lines and to add
48      * additional lines.  {@link Frame}s also are attached to specific
49      * AbstractPositioner and {@link AbstractFrameReflower} objects that contain the
50      * positioining and layout algorithm for a specific type of frame,
51      * respectively.  This is an application of the Strategy pattern.
52      *
53      * Layout, or reflow, proceeds recursively (post-order) starting at the root
54      * of the document.  Space constraints (containing block width & height) are
55      * pushed down, and resolved positions and sizes bubble up.  Thus, every
56      * {@link Frame} in the document tree is traversed once (except for tables
57      * which use a two-pass layout algorithm).  If you are interested in the
58      * details, see the reflow() method of the Reflower classes.
59      *
60      * Rendering is relatively straightforward once layout is complete. {@link
61      * Frame}s are rendered using an adapted {@link Cpdf} class, originally
62      * written by Wayne Munro, http://www.ros.co.nz/pdf/.  (Some performance
63      * related changes have been made to the original {@link Cpdf} class, and
64      * the {@link Dompdf\Adapter\CPDF} class provides a simple, stateless interface to
65      * PDF generation.)  PDFLib support has now also been added, via the {@link
66      * Dompdf\Adapter\PDFLib}.
67      *
68      *
69      * @package dompdf

418         /**
419          * Loads an HTML string
420          * Parse errors are stored in the global array _dompdf_warnings.
421          *
422          * @param string $str HTML text to load
423          * @param string $encoding Encoding of $str
424          */
425         public function loadHtml($str, $encoding = null)
426         {
427             $this->saveLocale();
428     
429             // Determine character encoding when $encoding parameter not used
430             if ($encoding === null) {
431                 mb_detect_order('auto');
432                 if (($encoding = mb_detect_encoding($str, null, true)) === false) {
433     
434                     //"auto" is expanded to "ASCII,JIS,UTF-8,EUC-JP,SJIS"
435                     $encoding = "auto";
436                 }
437             }
438     
439             if (in_array(strtoupper($encoding), array('UTF-8','UTF8')) === false) {
440                 $str = mb_convert_encoding($str, 'UTF-8', $encoding);
441     
442                 //Update encoding after converting
443                 $encoding = 'UTF-8';
444             }
445     
446             $metatags = [
447                 '@<meta\s+http-equiv="Content-Type"\s+content="(?:[\w/]+)(?:;\s*?charset=([^\s"]+))?@i',
448                 '@<meta\s+content="(?:[\w/]+)(?:;\s*?charset=([^\s"]+))"?\s+http-equiv="Content-Type"@i',
449                 '@<meta [^>]*charset\s*=\s*["\']?\s*([^"\' ]+)@i',
450             ];
451             foreach ($metatags as $metatag) {
452                 if (preg_match($metatag, $str, $matches)) {
453                     if (isset($matches[1]) && in_array($matches[1], mb_list_encodings())) {
454                         $document_encoding = $matches[1];
455                         break;
456                     }
457                 }
458             }
459             if (isset($document_encoding) && in_array(strtoupper($document_encoding), ['UTF-8','UTF8']) === false) {
460                 $str = preg_replace('/charset=([^\s"]+)/i', 'charset=UTF-8', $str);
461             } elseif (isset($document_encoding) === false && strpos($str, '<head>') !== false) {
462                 $str = str_replace('<head>', '<head><meta http-equiv="Content-Type" content="text/html;charset=UTF-8">', $str);
463             } elseif (isset($document_encoding) === false) {
464                 $str = '<meta http-equiv="Content-Type" content="text/html;charset=UTF-8">' . $str;
465             }
466     
467             // remove BOM mark from UTF-8, it's treated as document text by DOMDocument
468             // FIXME: roll this into the encoding detection using UTF-8/16/32 BOM (http://us2.php.net/manual/en/function.mb-detect-encoding.php#91051)?
469             if (substr($str, 0, 3) == chr(0xEF) . chr(0xBB) . chr(0xBF)) {
470                 $str = substr($str, 3);
471             }
472     
473             // Store parsing warnings as messages
474             set_error_handler([Helpers::class, 'record_warnings']);
475     
476             try {
477                 // @todo Take the quirksmode into account
478                 // http://hsivonen.iki.fi/doctype/
479                 // https://developer.mozilla.org/en/mozilla's_quirks_mode
480                 $quirksmode = false;
481     
482                 if ($this->options->isHtml5ParserEnabled() && class_exists(HTML5_Tokenizer::class)) {
483                     $tokenizer = new HTML5_Tokenizer($str);
484                     $tokenizer->parse();
485                     $doc = $tokenizer->save();
486     
487                     // Remove #text children nodes in nodes that shouldn't have
488                     $tag_names = ["html", "head", "table", "tbody", "thead", "tfoot", "tr"];
489                     foreach ($tag_names as $tag_name) {
490                         $nodes = $doc->getElementsByTagName($tag_name);
491     
492                         foreach ($nodes as $node) {
493                             self::removeTextNodes($node);
494                         }
495                     }
496     
497                     $quirksmode = ($tokenizer->getTree()->getQuirksMode() > HTML5_TreeBuilder::NO_QUIRKS);
498                 } else {
499                     // loadHTML assumes ISO-8859-1 unless otherwise specified on the HTML document header.
500                     // http://devzone.zend.com/1538/php-dom-xml-extension-encoding-processing/ (see #4)
501                     // http://stackoverflow.com/a/11310258/264628
502                     $doc = new DOMDocument("1.0", $encoding);
503                     $doc->preserveWhiteSpace = true;
504                     $doc->loadHTML($str);
505                     $doc->encoding = $encoding;
506     
507                     // Remove #text children nodes in nodes that shouldn't have
508                     $tag_names = ["html", "head", "table", "tbody", "thead", "tfoot", "tr"];
509                     foreach ($tag_names as $tag_name) {
510                         $nodes = $doc->getElementsByTagName($tag_name);
511     
512                         foreach ($nodes as $node) {
513                             self::removeTextNodes($node);
514                         }
515                     }
516     
517                     // If some text is before the doctype, we are in quirksmode
518                     if (preg_match("/^(.+)<!doctype/i", ltrim($str), $matches)) {
519                         $quirksmode = true;
520                     } // If no doctype is provided, we are in quirksmode
521                     elseif (!preg_match("/^<!doctype/i", ltrim($str), $matches)) {
522                         $quirksmode = true;
523                     } else {
524                         // HTML5 <!DOCTYPE html>
525                         if (!$doc->doctype->publicId && !$doc->doctype->systemId) {
526                             $quirksmode = false;
527                         }
528     
529                         // not XHTML
530                         if (!preg_match("/xhtml/i", $doc->doctype->publicId)) {
531                             $quirksmode = true;
532                         }
533                     }
534                 }
535     
536                 $this->dom = $doc;
537                 $this->quirksmode = $quirksmode;
538     
539                 $this->tree = new FrameTree($this->dom);
540             } finally {
541                 restore_error_handler();
542                 $this->restoreLocale();
543             }
544         }

573         /**
574          * Builds the {@link FrameTree}, loads any CSS and applies the styles to
575          * the {@link FrameTree}
576          */
577         private function processHtml()
578         {
579             $this->tree->build_tree();
580     
581             $this->css->load_css_file(Stylesheet::getDefaultStylesheet(), Stylesheet::ORIG_UA);
582     
583             $acceptedmedia = Stylesheet::$ACCEPTED_GENERIC_MEDIA_TYPES;
584             $acceptedmedia[] = $this->options->getDefaultMediaType();
585     
586             // <base href="" />
587             $base_nodes = $this->dom->getElementsByTagName("base");
588             if ($base_nodes->length && ($href = $base_nodes->item(0)->getAttribute("href"))) {
589                 [$this->protocol, $this->baseHost, $this->basePath] = Helpers::explode_url($href);
590             }
591     
592             // Set the base path of the Stylesheet to that of the file being processed
593             $this->css->set_protocol($this->protocol);
594             $this->css->set_host($this->baseHost);
595             $this->css->set_base_path($this->basePath);
596     
597             // Get all the stylesheets so that they are processed in document order
598             $xpath = new DOMXPath($this->dom);
599             $stylesheets = $xpath->query("//*[name() = 'link' or name() = 'style']");
600     
601             /** @var \DOMElement $tag */
602             foreach ($stylesheets as $tag) {
603                 switch (strtolower($tag->nodeName)) {
604                     // load <link rel="STYLESHEET" ... /> tags
605                     case "link":
606                         if (mb_strtolower(stripos($tag->getAttribute("rel"), "stylesheet") !== false) || // may be "appendix stylesheet"
607                             mb_strtolower($tag->getAttribute("type")) === "text/css"
608                         ) {
609                             //Check if the css file is for an accepted media type
610                             //media not given then always valid
611                             $formedialist = preg_split("/[\s\n,]/", $tag->getAttribute("media"), -1, PREG_SPLIT_NO_EMPTY);
612                             if (count($formedialist) > 0) {
613                                 $accept = false;
614                                 foreach ($formedialist as $type) {
615                                     if (in_array(mb_strtolower(trim($type)), $acceptedmedia)) {
616                                         $accept = true;
617                                         break;
618                                     }
619                                 }
620     
621                                 if (!$accept) {
622                                     //found at least one mediatype, but none of the accepted ones
623                                     //Skip this css file.
624                                     break;
625                                 }
626                             }
627     
628                             $url = $tag->getAttribute("href");
629                             $url = Helpers::build_url($this->protocol, $this->baseHost, $this->basePath, $url);
630     
631                             $this->css->load_css_file($url, Stylesheet::ORIG_AUTHOR);
632                         }
633                         break;
634     
635                     // load <style> tags
636                     case "style":
637                         // Accept all <style> tags by default (note this is contrary to W3C
638                         // HTML 4.0 spec:
639                         // http://www.w3.org/TR/REC-html40/present/styles.html#adef-media
640                         // which states that the default media type is 'screen'
641                         if ($tag->hasAttributes() &&
642                             ($media = $tag->getAttribute("media")) &&
643                             !in_array($media, $acceptedmedia)
644                         ) {
645                             break;
646                         }
647     
648                         $css = "";
649                         if ($tag->hasChildNodes()) {
650                             $child = $tag->firstChild;
651                             while ($child) {
652                                 $css .= $child->nodeValue; // Handle <style><!-- blah --></style>
653                                 $child = $child->nextSibling;
654                             }
655                         } else {
656                             $css = $tag->nodeValue;
657                         }
658     
659                         $this->css->load_css($css, Stylesheet::ORIG_AUTHOR);
660                         break;
661                 }
662             }
663         }

717         {
718             $this->saveLocale();
719             $options = $this->options;
720     
721             $logOutputFile = $options->getLogOutputFile();
722             if ($logOutputFile) {
723                 if (!file_exists($logOutputFile) && is_writable(dirname($logOutputFile))) {
724                     touch($logOutputFile);
725                 }
726     
727                 $this->startTime = microtime(true);
728                 if (is_writable($logOutputFile)) {
729                     ob_start();
730                 }
731             }
732     
733             $this->processHtml();
734     
735             $this->css->apply_styles($this->tree);
736     
737             // @page style rules : size, margins
738             $pageStyles = $this->css->get_page_styles();
739             $basePageStyle = $pageStyles["base"];
740             unset($pageStyles["base"]);
741     
742             foreach ($pageStyles as $pageStyle) {
743                 $pageStyle->inherit($basePageStyle);
744             }
745     
746             $defaultOptionPaperSize = $this->getPaperSize($options->getDefaultPaperSize());
747             // If there is a CSS defined paper size compare to the paper size used to create the canvas to determine a
748             // recreation need
749             if (is_array($basePageStyle->size)) {
750                 $basePageStyleSize = $basePageStyle->size;
751                 $this->setPaper([0, 0, $basePageStyleSize[0], $basePageStyleSize[1]]);
752             }
753     
754             $paperSize = $this->getPaperSize();
755             if (
756                 $defaultOptionPaperSize[2] !== $paperSize[2] ||
757                 $defaultOptionPaperSize[3] !== $paperSize[3] ||
758                 $options->getDefaultPaperOrientation() !== $this->paperOrientation
759             ) {
760                 $this->setCanvas(CanvasFactory::get_instance($this, $this->paperSize, $this->paperOrientation));
761                 $this->fontMetrics->setCanvas($this->getCanvas());
762             }
763     
764             $canvas = $this->getCanvas();
765     
766             if ($options->isFontSubsettingEnabled() && $canvas instanceof CPDF) {
767                 foreach ($this->tree->get_frames() as $frame) {
768                     $style = $frame->get_style();
769                     $node = $frame->get_node();
770     
771                     // Handle text nodes
772                     if ($node->nodeName === "#text") {
773                         $chars = mb_strtoupper($node->nodeValue) . mb_strtolower($node->nodeValue);
774                         $canvas->register_string_subset($style->font_family, $chars);
775                         continue;
776                     }
777     
778                     // Handle generated content (list items)
779                     if ($style->display === "list-item") {
780                         $chars = ListBullet::get_counter_chars($style->list_style_type);
781                         $canvas->register_string_subset($style->font_family, $chars);
782                         $canvas->register_string_subset($style->font_family, '.');
783                         continue;
784                     }
785     
786                     // Handle other generated content (pseudo elements)
787                     // FIXME: This only captures the text of the stylesheet declaration,
788                     //        not the actual generated content, and forces all possible counter
789                     //        values. See notes in issue #750.
790                     if ($frame->get_node()->nodeName == "dompdf_generated") {
791                         // all possible counter values, just in case
792                         $chars = ListBullet::get_counter_chars('decimal');
793                         $canvas->register_string_subset($style->font_family, $chars);
794                         $chars = ListBullet::get_counter_chars('upper-alpha');
795                         $canvas->register_string_subset($style->font_family, $chars);
796                         $chars = ListBullet::get_counter_chars('lower-alpha');
797                         $canvas->register_string_subset($style->font_family, $chars);
798                         $chars = ListBullet::get_counter_chars('lower-greek');
799                         $canvas->register_string_subset($style->font_family, $chars);
800     
801                         // the hex-decoded text of the content property, duplicated from AbstrctFrameReflower::_parse_string
802                         $decoded_string = preg_replace_callback("/\\\\([0-9a-fA-F]{0,6})/",
803                             function ($matches) { return \Dompdf\Helpers::unichr(hexdec($matches[1])); },
804                             $style->content);
805                         $chars = mb_strtoupper($style->content) . mb_strtolower($style->content) . mb_strtoupper($decoded_string) . mb_strtolower($decoded_string);
806                         $canvas->register_string_subset($style->font_family, $chars);
807                         continue;
808                     }
809                 }
810             }
811     
812             $root = null;
813     
814             foreach ($this->tree->get_frames() as $frame) {
815                 // Set up the root frame
816                 if (is_null($root)) {
817                     $root = Factory::decorate_root($this->tree->get_root(), $this);
818                     continue;
819                 }
820     
821                 // Create the appropriate decorators, reflowers & positioners.
822                 Factory::decorate_frame($frame, $this, $root);
823             }
824     
825             // Add meta information
826             $title = $this->dom->getElementsByTagName("title");
827             if ($title->length) {
828                 $canvas->add_info("Title", trim($title->item(0)->nodeValue));
829             }
830     
831             $metas = $this->dom->getElementsByTagName("meta");
832             $labels = [
833                 "author" => "Author",
834                 "keywords" => "Keywords",
835                 "description" => "Subject",
836             ];
837             /** @var \DOMElement $meta */
838             foreach ($metas as $meta) {
839                 $name = mb_strtolower($meta->getAttribute("name"));
840                 $value = trim($meta->getAttribute("content"));
841     
842                 if (isset($labels[$name])) {
843                     $canvas->add_info($labels[$name], $value);
844                     continue;
845                 }
846     
847                 if ($name === "dompdf.view" && $this->parseDefaultView($value)) {
848                     $canvas->set_default_view($this->defaultView, $this->defaultViewOptions);
849                 }
850             }
851     
852             $root->set_containing_block(0, 0, $canvas->get_width(), $canvas->get_height());
853             $root->set_renderer(new Renderer($this));
854     
855             // This is where the magic happens:
856             $root->reflow();
857     
858             // Clean up cached images
859             Cache::clear();
860     
861             global $_dompdf_warnings, $_dompdf_show_warnings;
862             if ($_dompdf_show_warnings && isset($_dompdf_warnings)) {
863                 echo '<b>Dompdf Warnings</b><br><pre>';
864                 foreach ($_dompdf_warnings as $msg) {
865                     echo $msg . "\n";
866                 }
867     
868                 if ($canvas instanceof CPDF) {
869                     echo $canvas->get_cpdf()->messages;
870                 }
871                 echo '</pre>';
872                 flush();
873             }
874     
875             if ($logOutputFile && is_writable($logOutputFile)) {
876                 $this->write_log();
877                 ob_end_clean();
878             }
879     
880             $this->restoreLocale();
881         }

894         /**
895          * Writes the output buffer in the log file
896          *
897          * @return void
898          */
899         private function write_log()
900         {
901             $log_output_file = $this->getOptions()->getLogOutputFile();
902             if (!$log_output_file || !is_writable($log_output_file)) {
903                 return;
904             }
905     
906             $frames = Frame::$ID_COUNTER;
907             $memory = memory_get_peak_usage(true) / 1024;
908             $time = (microtime(true) - $this->startTime) * 1000;
909     
910             $out = sprintf(
911                 "<span style='color: #000' title='Frames'>%6d</span>" .
912                 "<span style='color: #009' title='Memory'>%10.2f KB</span>" .
913                 "<span style='color: #900' title='Time'>%10.2f ms</span>" .
914                 "<span  title='Quirksmode'>  " .
915                 ($this->quirksmode ? "<span style='color: #d00'> ON</span>" : "<span style='color: #0d0'>OFF</span>") .
916                 "</span><br />", $frames, $memory, $time);
917     
918             $out .= ob_get_contents();
919             ob_clean();
920     
921             file_put_contents($log_output_file, $out);
922         }

1441          * Sets callbacks for events like rendering of pages and elements.
1442          * The callbacks array contains arrays with 'event' set to 'begin_page',
1443          * 'end_page', 'begin_frame', or 'end_frame' and 'f' set to a function or
1444          * object plus method to be called.

1446          * The function 'f' must take an array as argument, which contains info
1447          * about the event.
1448          *
1449          * @param array $callbacks the set of callbacks to set
1450          */
1451         public function setCallbacks($callbacks)
1452         {
1453             if (is_array($callbacks)) {
1454                 $this->callbacks = [];
1455                 foreach ($callbacks as $c) {
1456                     if (is_array($c) && isset($c['event']) && isset($c['f'])) {
1457                         $event = $c['event'];
1458                         $f = $c['f'];
1459                         if (is_callable($f) && is_string($event)) {
1460                             $this->callbacks[$event][] = $f;
1461                         }
1462                     }
1463                 }
1464             }
1465         }

1504         /**
1505          * PHP5 overloaded getter
1506          * Along with {@link Dompdf::__set()} __get() provides access to all
1507          * properties directly.  Typically __get() is not called directly outside
1508          * of this class.
1509          *
1510          * @param string $prop
1511          *
1512          * @throws Exception
1513          * @return mixed
1514          */
1515         function __get($prop)
1516         {
1517             switch ($prop)
1518             {
1519                 case 'version' :
