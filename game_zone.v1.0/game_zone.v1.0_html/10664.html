601          *
602          * @param null|string $state 'state' or null
603          *
604          * @return @string
605          */
606         public static function postcode($state = null)
607         {
608             $format = array(
609                 'perlis' => array( // (01000 - 02800)
610                     '0' . mt_rand(1000, 2800)
611                 ),
612                 'kedah' => array( // (05000 - 09810)
613                     '0' . mt_rand(5000, 9810)
614                 ),
615                 'penang' => array( // (10000 - 14400)
616                     mt_rand(10000, 14400)
617                 ),
618                 'kelantan' => array( // (15000 - 18500)
619                     mt_rand(15000, 18500)
620                 ),
621                 'terengganu' => array( // (20000 - 24300)
622                     mt_rand(20000, 24300)
623                 ),
624                 'pahang' => array( // (25000 - 28800 | 39000 - 39200 | 49000, 69000)
625                     mt_rand(25000, 28800),
626                     mt_rand(39000, 39200),
627                     mt_rand(49000, 69000)
628                 ),
629                 'perak' => array( // (30000 - 36810)
630                     mt_rand(30000, 36810)
631                 ),
632                 'selangor' => array( // (40000 - 48300 | 63000 - 68100)
633                     mt_rand(40000, 48300),
634                     mt_rand(63000, 68100)
635                 ),
636                 'kl' => array( // (50000 - 60000)
637                     mt_rand(50000, 60000),
638                 ),
639                 'putrajaya' => array( // (62000 - 62988)

642                 'nsembilan' => array( // (70000 - 73509)

666         /**
667          * Return the complete town address with matching postcode and state
668          *
669          * @example 55100 Bukit Bintang, Kuala Lumpur
670          *
671          * @return @string
672          */
673         public function townState()
674         {
675             $state = static::randomElement(array_keys(static::$states));
676             $postcode = static::postcode($state);
677             $town = static::randomElement(static::$towns[$state]);
678             $state = static::randomElement(static::$states[$state]);
679     
680             return $postcode . ' ' . $town . ', ' . $state;
681         }
