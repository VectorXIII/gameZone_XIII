84     
85             $cache = $this->table()->where('key', '=', $prefixed)->first();
86     
87             // If we have a cache record we will check the expiration time against current
88             // time on the system and see if the record has expired. If it has, we will
89             // remove the records from the database table so it isn't returned again.
90             if (is_null($cache)) {

92             }

167          * @return int|bool

169         protected function incrementOrDecrement($key, $value, Closure $callback)
170         {
171             return $this->connection->transaction(function () use ($key, $value, $callback) {
172                 $prefixed = $this->prefix.$key;
173     
174                 $cache = $this->table()->where('key', $prefixed)
175                             ->lockForUpdate()->first();
176     
177                 // If there is no value in the cache, we will return false here. Otherwise the
178                 // value will be decrypted and we will proceed with this function to either
179                 // increment or decrement this value based on the given action callbacks.
180                 if (is_null($cache)) {
181                     return false;
182                 }
183     
184                 $cache = is_array($cache) ? (object) $cache : $cache;
185     
186                 $current = $this->unserialize($cache->value);
187     
188                 // Here we'll call this callback function that was given to the function which
189                 // is used to either increment or decrement the function. We use a callback
190                 // so we do not have to recreate all this logic in each of the functions.
191                 $new = $callback((int) $current, $value);
192     
193                 if (! is_numeric($current)) {
194                     return false;
195                 }
196     
197                 // Here we will update the values in the table. We will also encrypt the value
198                 // since database cache values are encrypted by default with secure storage
199                 // that can't be easily read. We will return the new value after storing.
200                 $this->table()->where('key', $prefixed)->update([
201                     'value' => $this->serialize($new),
202                 ]);
203     
204                 return $new;
205             });
206         }
