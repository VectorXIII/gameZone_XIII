320             foreach ($this->xpath->query('php/ini') as $ini) {

332                 $name  = (string) $const->getAttribute('name');
333                 $value = (string) $const->getAttribute('value');
334     
335                 $result['const'][$name]['value'] = $this->getBoolean($value, $value);
336             }
337     
338             foreach (['var', 'env', 'post', 'get', 'cookie', 'server', 'files', 'request'] as $array) {
339                 foreach ($this->xpath->query('php/' . $array) as $var) {
340                     \assert($var instanceof DOMElement);
341     
342                     $name     = (string) $var->getAttribute('name');
343                     $value    = (string) $var->getAttribute('value');
344                     $verbatim = false;
345     
346                     if ($var->hasAttribute('verbatim')) {
347                         $verbatim                          = $this->getBoolean($var->getAttribute('verbatim'), false);
348                         $result[$array][$name]['verbatim'] = $verbatim;
349                     }
350     
351                     if ($var->hasAttribute('force')) {
352                         $force                          = $this->getBoolean($var->getAttribute('force'), false);
353                         $result[$array][$name]['force'] = $force;
354                     }
355     
356                     if (!$verbatim) {
357                         $value = $this->getBoolean($value, $value);
358                     }
359     
360                     $result[$array][$name]['value'] = $value;
361                 }
362             }

367         /**
368          * Handles the PHP configuration.
369          */
370         public function handlePHPConfiguration(): void
371         {
372             $configuration = $this->getPHPConfiguration();
373     
374             if (!empty($configuration['include_path'])) {
375                 \ini_set(
376                     'include_path',
377                     \implode(\PATH_SEPARATOR, $configuration['include_path']) .
378                     \PATH_SEPARATOR .
379                     \ini_get('include_path')
380                 );
381             }
382     
383             foreach ($configuration['ini'] as $name => $data) {
384                 $value = $data['value'];
385     
386                 if (\defined($value)) {
387                     $value = (string) \constant($value);
388                 }
389     
390                 \ini_set($name, $value);
391             }
392     
393             foreach ($configuration['const'] as $name => $data) {
394                 $value = $data['value'];
395     
396                 if (!\defined($name)) {
397                     \define($name, $value);
398                 }
399             }
400     
401             foreach (['var', 'post', 'get', 'cookie', 'server', 'files', 'request'] as $array) {
402                 /*
403                  * @see https://github.com/sebastianbergmann/phpunit/issues/277
404                  */
405                 switch ($array) {
406                     case 'var':
407                         $target = &$GLOBALS;
408     
409                         break;
410     
411                     case 'server':
412                         $target = &$_SERVER;
413     
414                         break;
415     
416                     default:
417                         $target = &$GLOBALS['_' . \strtoupper($array)];
418     
419                         break;
420                 }
421     
422                 foreach ($configuration[$array] as $name => $data) {
423                     $target[$name] = $data['value'];
424                 }
425             }
426     
427             foreach ($configuration['env'] as $name => $data) {
428                 $value = $data['value'];
429                 $force = $data['force'] ?? false;
430     
431                 if ($force || \getenv($name) === false) {
432                     \putenv("{$name}={$value}");
433                 }
434     
435                 $value = \getenv($name);
436     
437                 if (!isset($_ENV[$name])) {
438                     $_ENV[$name] = $value;
439                 }
440     
441                 if ($force) {
442                     $_ENV[$name] = $value;
443                 }
444             }
445         }

918         /**
919          * Collects and returns the configuration arguments from the PHPUnit
920          * XML configuration
921          */
922         private function getConfigurationArguments(\DOMNodeList $nodes): array
923         {
924             $arguments = [];
925     
926             if ($nodes->length === 0) {
927                 return $arguments;
928             }
929     
930             foreach ($nodes as $node) {
931                 if (!$node instanceof DOMElement) {
932                     continue;
933                 }
934     
935                 if ($node->tagName !== 'arguments') {
936                     continue;
937                 }
938     
939                 foreach ($node->childNodes as $argument) {
940                     if (!$argument instanceof DOMElement) {
941                         continue;
942                     }
943     
944                     if ($argument->tagName === 'file' || $argument->tagName === 'directory') {
945                         $arguments[] = $this->toAbsolutePath((string) $argument->textContent);
946                     } else {
947                         $arguments[] = Xml::xmlToVariable($argument);
948                     }
949                 }
950             }
951     
952             return $arguments;
953         }

955         /**
956          * @throws \PHPUnit\Framework\Exception
957          */
958         private function getTestSuite(DOMElement $testSuiteNode, string $testSuiteFilter = ''): TestSuite
959         {
960             if ($testSuiteNode->hasAttribute('name')) {
961                 $suite = new TestSuite(
962                     (string) $testSuiteNode->getAttribute('name')
963                 );
964             } else {
965                 $suite = new TestSuite;
966             }
967     
968             $exclude = [];
969     
970             foreach ($testSuiteNode->getElementsByTagName('exclude') as $excludeNode) {
971                 $excludeFile = (string) $excludeNode->textContent;
972     
973                 if ($excludeFile) {
974                     $exclude[] = $this->toAbsolutePath($excludeFile);
975                 }
976             }
977     
978             $fileIteratorFacade = new FileIteratorFacade;
979             $testSuiteFilter    = $testSuiteFilter ? \explode(',', $testSuiteFilter) : [];
980     
981             foreach ($testSuiteNode->getElementsByTagName('directory') as $directoryNode) {
982                 \assert($directoryNode instanceof DOMElement);
983     
984                 if (!empty($testSuiteFilter) && !\in_array($directoryNode->parentNode->getAttribute('name'), $testSuiteFilter)) {
985                     continue;
986                 }
987     
988                 $directory = (string) $directoryNode->textContent;
989     
990                 if (empty($directory)) {
991                     continue;
992                 }
993     
994                 if (!$this->satisfiesPhpVersion($directoryNode)) {
995                     continue;
996                 }
997     
998                 $files = $fileIteratorFacade->getFilesAsArray(
999                     $this->toAbsolutePath($directory),
1000                     $directoryNode->hasAttribute('suffix') ? (string) $directoryNode->getAttribute('suffix') : 'Test.php',
1001                     $directoryNode->hasAttribute('prefix') ? (string) $directoryNode->getAttribute('prefix') : '',
1002                     $exclude
1003                 );
1004     
1005                 $suite->addTestFiles($files);
1006             }
1007     
1008             foreach ($testSuiteNode->getElementsByTagName('file') as $fileNode) {
1009                 \assert($fileNode instanceof DOMElement);
1010     
1011                 if (!empty($testSuiteFilter) && !\in_array($fileNode->parentNode->getAttribute('name'), $testSuiteFilter)) {
1012                     continue;
1013                 }
1014     
1015                 $file = (string) $fileNode->textContent;
1016     
1017                 if (empty($file)) {
1018                     continue;
1019                 }
1020     
1021                 $file = $fileIteratorFacade->getFilesAsArray(
1022                     $this->toAbsolutePath($file)
1023                 );
1024     
1025                 if (!isset($file[0])) {
1026                     continue;
1027                 }
1028     
1029                 $file = $file[0];
1030     
1031                 if (!$this->satisfiesPhpVersion($fileNode)) {
1032                     continue;
1033                 }
1034     
1035                 $suite->addTestFile($file);
1036             }
1037     
1038             return $suite;
1039         }

1041         private function satisfiesPhpVersion(DOMElement $node): bool
1042         {
1043             $phpVersion         = \PHP_VERSION;
1044             $phpVersionOperator = '>=';
1045     
1046             if ($node->hasAttribute('phpVersion')) {
1047                 $phpVersion = (string) $node->getAttribute('phpVersion');
1048             }
1049     
1050             if ($node->hasAttribute('phpVersionOperator')) {
1051                 $phpVersionOperator = (string) $node->getAttribute('phpVersionOperator');
1052             }
1053     
1054             return \version_compare(\PHP_VERSION, $phpVersion, (new VersionComparisonOperator($phpVersionOperator))->asString());
1055         }

1130         private function toAbsolutePath(string $path, bool $useIncludePath = false): string
1131         {
1132             $path = \trim($path);
1133     
1134             if (\strpos($path, '/') === 0) {
1135                 return $path;
1136             }
1137     
1138             // Matches the following on Windows:
1139             //  - \\NetworkComputer\Path
1140             //  - \\.\D:
1141             //  - \\.\c:
1142             //  - C:\Windows
1143             //  - C:\windows
1144             //  - C:/windows
1145             //  - c:/windows
1146             if (\defined('PHP_WINDOWS_VERSION_BUILD') &&
1147                 ($path[0] === '\\' || (\strlen($path) >= 3 && \preg_match('#^[A-Z]\:[/\\\]#i', \substr($path, 0, 3))))) {
1148                 return $path;
1149             }
1150     
1151             if (\strpos($path, '://') !== false) {
1152                 return $path;
1153             }
1154     
1155             $file = \dirname($this->filename) . \DIRECTORY_SEPARATOR . $path;
1156     
1157             if ($useIncludePath && !\file_exists($file)) {
1158                 $includePathFile = \stream_resolve_include_path($path);
1159     
1160                 if ($includePathFile) {
1161                     $file = $includePathFile;
1162                 }
1163             }
1164     
1165             return $file;
1166         }
