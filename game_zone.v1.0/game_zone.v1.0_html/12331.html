37         public function __construct(string $index, string $type)

42             $this->index = $index;

44         }

46         /**
47          * {@inheritdoc}
48          */
49         public function format(array $record)
50         {
51             $record = parent::format($record);
52     
53             return $this->getDocument($record);
54         }
55     
56         public function getIndex(): string
57         {
58             return $this->index;
59         }
60     
61         public function getType(): string
62         {
63             return $this->type;
64         }
65     
66         /**
67          * Convert a log message into an Elastica Document
68          * @param  array    $record
69          * @return Document
70          */
71         protected function getDocument(array $record): Document
72         {
73             $document = new Document();
74             $document->setData($record);

76             $document->setIndex($this->index);

78             return $document;
