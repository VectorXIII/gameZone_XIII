265         private function initializeExtensions(): void
266         {
267             // Ask all extensions to register their components
268             while (!empty($this->uninitializedExtensions)) {
269                 foreach ($this->uninitializedExtensions as $i => $extension) {
270                     $extension->register($this);
271                     unset($this->uninitializedExtensions[$i]);
272                 }
273             }
274     
275             $this->extensionsInitialized = true;
276     
277             // Lastly, let's build a regex which matches non-inline characters
278             // This will enable a huge performance boost with inline parsing
279             $this->buildInlineParserCharacterRegex();
280         }

327         public function addEventListener(string $eventClass, callable $listener, int $priority = 0): ConfigurableEnvironmentInterface
328         {
329             $this->assertUninitialized('Failed to add event listener.');
330     
331             if (!isset($this->listeners[$eventClass])) {
332                 $this->listeners[$eventClass] = new PrioritizedList();
333             }
334     
335             $this->listeners[$eventClass]->add($listener, $priority);
336     
337             if (\is_object($listener)) {
338                 $this->injectEnvironmentAndConfigurationIfNeeded($listener);
339             } elseif (\is_array($listener) && \is_object($listener[0])) {
340                 $this->injectEnvironmentAndConfigurationIfNeeded($listener[0]);
341             }
342     
343             return $this;
344         }

346         public function dispatch(AbstractEvent $event): void
347         {
348             if (!$this->extensionsInitialized) {
349                 $this->initializeExtensions();
350             }
351     
352             $type = \get_class($event);
353     
354             foreach ($this->listeners[$type] ?? [] as $listener) {
355                 if ($event->isPropagationStopped()) {
356                     return;
357                 }
358     
359                 $listener($event);
360             }
361         }

363         private function buildInlineParserCharacterRegex(): void
364         {
365             $chars = \array_unique(\array_merge(
366                 \array_keys($this->inlineParsersByCharacter),
367                 $this->delimiterProcessors->getDelimiterCharacters()
368             ));
369     
370             if (empty($chars)) {
371                 // If no special inline characters exist then parse the whole line
372                 $this->inlineParserCharacterRegex = '/^.+$/u';
373             } else {
374                 // Match any character which inline parsers are not interested in
375                 $this->inlineParserCharacterRegex = '/^[^' . \preg_quote(\implode('', $chars), '/') . ']+/u';
376             }
377         }

391         /**
392          * @param array<string, PrioritizedList> $list
393          * @param string                         $class
394          * @param string                         $type
395          *
396          * @return iterable
397          *
398          * @phpstan-template T
399          *
400          * @phpstan-param array<string, PrioritizedList<T>> $list
401          * @phpstan-param string                            $class
402          * @phpstan-param class-string<T>                   $type
403          *
404          * @phpstan-return iterable<T>
405          */
406         private function getRenderersByClass(array &$list, string $class, string $type): iterable
407         {
408             // If renderers are defined for this specific class, return them immediately
409             if (isset($list[$class])) {
410                 return $list[$class];
411             }
412     
413             while ($parent = \get_parent_class($parent ?? $class)) {
414                 if (!isset($list[$parent])) {
415                     continue;
416                 }
417     
418                 // "Cache" this result to avoid future loops
419                 return $list[$class] = $list[$parent];
420             }
421     
422             return [];
423         }
