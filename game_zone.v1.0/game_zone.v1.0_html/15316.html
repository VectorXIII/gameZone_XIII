32     // Throw an exception when E_ERRORs occur to better debug the problem
33     function throwExceptionOnError($severity, $message, $file, $line)
34     {
35         $errMessage = sprintf('Language error ("%s") thrown from %s:%s', $message, $file, $line);
36     
37         throw new \Exception($errMessage, $severity);
38     }
39     
40     // If `DISABLE_WARN_EXCEPTIONS` exists, then E_ERRORs will **not** be casted into exceptions
41     if (!getenv('DISABLE_WARN_EXCEPTIONS')) {
