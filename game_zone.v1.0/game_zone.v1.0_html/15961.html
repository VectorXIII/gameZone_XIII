12     namespace Symfony\Component\EventDispatcher;
13     
14     use Symfony\Contracts\EventDispatcher\Event;
15     
16     /**
17      * Event encapsulation class.
18      *
19      * Encapsulates events thus decoupling the observer from the subject they encapsulate.
20      *
21      * @author Drak <drak@zikula.org>
22      */
23     class GenericEvent extends Event implements \ArrayAccess, \IteratorAggregate
24     {
25         protected $subject;
26         protected $arguments;
27     
28         /**
29          * Encapsulate an event with $subject and $args.
30          *
31          * @param mixed $subject   The subject of the event, usually an object or a callable
32          * @param array $arguments Arguments to store in the event
33          */
34         public function __construct($subject = null, array $arguments = [])
35         {
36             $this->subject = $subject;
37             $this->arguments = $arguments;
38         }
39     
40         /**
41          * Getter for subject property.
42          *
43          * @return mixed The observer subject
44          */
45         public function getSubject()
46         {
47             return $this->subject;
48         }
49     
50         /**
51          * Get argument by key.
52          *
53          * @return mixed Contents of array key
54          *
55          * @throws \InvalidArgumentException if key is not found
56          */
57         public function getArgument(string $key)
58         {
59             if ($this->hasArgument($key)) {
60                 return $this->arguments[$key];
61             }
62     
63             throw new \InvalidArgumentException(sprintf('Argument "%s" not found.', $key));
64         }
65     
66         /**
67          * Add argument to event.
68          *
69          * @param mixed $value Value
70          *
71          * @return $this
72          */
73         public function setArgument(string $key, $value)
74         {
75             $this->arguments[$key] = $value;
76     
77             return $this;
78         }
79     
80         /**
81          * Getter for all arguments.
82          *
83          * @return array
84          */
85         public function getArguments()
86         {
87             return $this->arguments;
88         }
89     
90         /**
91          * Set args property.
92          *
93          * @return $this
94          */
95         public function setArguments(array $args = [])
96         {
97             $this->arguments = $args;
98     
99             return $this;
100         }
101     
102         /**
103          * Has argument.
104          *
105          * @return bool
106          */
107         public function hasArgument(string $key)
108         {
109             return \array_key_exists($key, $this->arguments);
110         }
111     
112         /**
113          * ArrayAccess for argument getter.
114          *
115          * @param string $key Array key
116          *
117          * @return mixed
118          *
119          * @throws \InvalidArgumentException if key does not exist in $this->args
120          */
121         public function offsetGet($key)
122         {
123             return $this->getArgument($key);
124         }
125     
126         /**
127          * ArrayAccess for argument setter.
128          *
129          * @param string $key   Array key to set
130          * @param mixed  $value Value
131          */
132         public function offsetSet($key, $value)
133         {
134             $this->setArgument($key, $value);
135         }
136     
137         /**
138          * ArrayAccess for unset argument.
139          *
140          * @param string $key Array key
141          */
142         public function offsetUnset($key)
143         {
144             if ($this->hasArgument($key)) {
145                 unset($this->arguments[$key]);
146             }
147         }
148     
149         /**
150          * ArrayAccess has argument.
151          *
152          * @param string $key Array key
153          *
154          * @return bool
155          */
156         public function offsetExists($key)
157         {
158             return $this->hasArgument($key);
159         }
160     
161         /**
162          * IteratorAggregate for iterating over the object like an array.
163          *
164          * @return \ArrayIterator
165          */
166         public function getIterator()
167         {
168             return new \ArrayIterator($this->arguments);
169         }
170     }
