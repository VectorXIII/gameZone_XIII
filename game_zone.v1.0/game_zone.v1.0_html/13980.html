136         /**

139          * @param float $coveragePercent

141         private function getCrapLoad($crapValue, $cyclomaticComplexity, $coveragePercent): float
142         {
143             $crapLoad = 0;
144     
145             if ($crapValue >= $this->threshold) {
146                 $crapLoad += $cyclomaticComplexity * (1.0 - $coveragePercent / 100);
147                 $crapLoad += $cyclomaticComplexity / $this->threshold;
148             }
149     
150             return $crapLoad;
151         }
