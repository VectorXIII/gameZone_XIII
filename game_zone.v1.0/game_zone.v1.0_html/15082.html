21     /**
22      * FixedTimeProvider uses an known time to provide the time
23      *
24      * This provider allows the use of a previously-generated, or known, time
25      * when generating time-based UUIDs.
26      */
27     class FixedTimeProvider implements TimeProviderInterface
28     {
29         /**
30          * @var Time
31          */
32         private $fixedTime;
33     
34         public function __construct(Time $time)
35         {
36             $this->fixedTime = $time;
37         }
38     
39         /**
40          * Sets the `usec` component of the time
41          *
42          * @param int|string|IntegerObject $value The `usec` value to set
43          */
44         public function setUsec($value): void
45         {
46             $this->fixedTime = new Time($this->fixedTime->getSeconds(), $value);
47         }
48     
49         /**
50          * Sets the `sec` component of the time
51          *
52          * @param int|string|IntegerObject $value The `sec` value to set
53          */
54         public function setSec($value): void
55         {
56             $this->fixedTime = new Time($value, $this->fixedTime->getMicroseconds());
57         }
58     
59         public function getTime(): Time
60         {
61             return $this->fixedTime;
62         }
63     }
