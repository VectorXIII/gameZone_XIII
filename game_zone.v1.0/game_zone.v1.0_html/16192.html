50         public function find(?string $ip, ?string $url, ?int $limit, ?string $method, int $start = null, int $end = null, string $statusCode = null): array
51         {

131          *
132          * @throws \RuntimeException
133          */
134         public function write(Profile $profile): bool
135         {
136             $file = $this->getFilename($profile->getToken());
137     
138             $profileIndexed = is_file($file);
139             if (!$profileIndexed) {
140                 // Create directory
141                 $dir = \dirname($file);
142                 if (!is_dir($dir) && false === @mkdir($dir, 0777, true) && !is_dir($dir)) {
143                     throw new \RuntimeException(sprintf('Unable to create the storage directory (%s).', $dir));
144                 }
145             }
146     
147             $profileToken = $profile->getToken();
148             // when there are errors in sub-requests, the parent and/or children tokens
149             // may equal the profile token, resulting in infinite loops
150             $parentToken = $profile->getParentToken() !== $profileToken ? $profile->getParentToken() : null;
151             $childrenToken = array_filter(array_map(function (Profile $p) use ($profileToken) {
152                 return $profileToken !== $p->getToken() ? $p->getToken() : null;
153             }, $profile->getChildren()));
154     
155             // Store profile
156             $data = [
157                 'token' => $profileToken,
158                 'parent' => $parentToken,
159                 'children' => $childrenToken,
160                 'data' => $profile->getCollectors(),
161                 'ip' => $profile->getIp(),
162                 'method' => $profile->getMethod(),
163                 'url' => $profile->getUrl(),
164                 'time' => $profile->getTime(),
165                 'status_code' => $profile->getStatusCode(),
166             ];
167     
168             $context = stream_context_create();
169     
170             if (\function_exists('gzcompress')) {
171                 $file = 'compress.zlib://'.$file;
172                 stream_context_set_option($context, 'zlib', 'level', 3);
173             }
174     
175             if (false === file_put_contents($file, serialize($data), 0, $context)) {
176                 return false;
177             }
178     
179             if (!$profileIndexed) {
180                 // Add to index
181                 if (false === $file = fopen($this->getIndexFilename(), 'a')) {
182                     return false;
183                 }
184     
185                 fputcsv($file, [
186                     $profile->getToken(),
187                     $profile->getIp(),
188                     $profile->getMethod(),
189                     $profile->getUrl(),
190                     $profile->getTime(),
191                     $profile->getParentToken(),
192                     $profile->getStatusCode(),
193                 ]);
194                 fclose($file);
195             }
196     
197             return true;
198         }

224         /**
225          * Reads a line in the file, backward.
226          *
227          * This function automatically skips the empty lines and do not include the line return in result value.
228          *
229          * @param resource $file The file resource, with the pointer placed at the end of the line to read
230          *
231          * @return mixed A string representing the line or null if beginning of file is reached
232          */
233         protected function readLineFromFile($file)
234         {
235             $line = '';
236             $position = ftell($file);
237     
238             if (0 === $position) {
239                 return null;
240             }
241     
242             while (true) {
243                 $chunkSize = min($position, 1024);
244                 $position -= $chunkSize;
245                 fseek($file, $position);
246     
247                 if (0 === $chunkSize) {
248                     // bof reached
249                     break;
250                 }
251     
252                 $buffer = fread($file, $chunkSize);
253     
254                 if (false === ($upTo = strrpos($buffer, "\n"))) {
255                     $line = $buffer.$line;
256                     continue;
257                 }
258     
259                 $position += $upTo;
260                 $line = substr($buffer, $upTo + 1).$line;
261                 fseek($file, max(0, $position), SEEK_SET);
262     
263                 if ('' !== $line) {
264                     break;
265                 }
266             }
267     
268             return '' === $line ? null : $line;
269         }

272         {
273             $profile = new Profile($token);

275             $profile->setMethod($data['method']);

277             $profile->setTime($data['time']);
278             $profile->setStatusCode($data['status_code']);
279             $profile->setCollectors($data['data']);
280     
281             if (!$parent && $data['parent']) {
282                 $parent = $this->read($data['parent']);
283             }
284     
285             if ($parent) {
286                 $profile->setParent($parent);
287             }
288     
289             foreach ($data['children'] as $token) {
290                 if (!$token || !file_exists($file = $this->getFilename($token))) {
291                     continue;
292                 }
293     
294                 if (\function_exists('gzcompress')) {
295                     $file = 'compress.zlib://'.$file;
296                 }
297

299             }
300     
301             return $profile;
