196         private static function extractFlags(int $flags): ConstStub
197         {
198             $flagsArray = [];
199     
200             foreach (self::$flags as $value => $name) {
201                 if ($flags & $value) {
202                     $flagsArray[] = $name;

204             }

206             if (!$flagsArray) {
