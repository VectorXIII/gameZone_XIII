65         /**
66          * @param resource|string $input
67          */
68         public function __construct($input, ?string $charset = 'utf-8')
69         {
70             $charset = strtolower(trim($charset)) ?: 'utf-8';
71             if ('utf-8' === $charset || 'utf8' === $charset) {
72                 $this->fixedWidth = 0;
73                 $this->map = ['p' => [], 'i' => []];
74             } else {
75                 switch ($charset) {
76                     // 16 bits
77                     case 'ucs2':
78                     case 'ucs-2':
79                     case 'utf16':
80                     case 'utf-16':
81                         $this->fixedWidth = 2;
82                         break;
83     
84                     // 32 bits
85                     case 'ucs4':
86                     case 'ucs-4':
87                     case 'utf32':
88                     case 'utf-32':
89                         $this->fixedWidth = 4;
90                     break;
91     
92                     // 7-8 bit charsets: (us-)?ascii, (iso|iec)-?8859-?[0-9]+, windows-?125[0-9], cp-?[0-9]+, ansi, macintosh,
93                     //                   koi-?7, koi-?8-?.+, mik, (cork|t1), v?iscii
94                     // and fallback
95                     default:
96                         $this->fixedWidth = 1;
97                 }
98             }
99             if (\is_resource($input)) {
100                 $blocks = 512;
101                 if (stream_get_meta_data($input)['seekable'] ?? false) {
102                     rewind($input);
103                 }
104                 while (false !== $read = fread($input, $blocks)) {
105                     $this->write($read);
106                 }
107             } else {
108                 $this->write($input);
109             }
110         }

112         public function read(int $length): ?string
113         {
114             if ($this->currentPos >= $this->charCount) {
115                 return null;
116             }
117             $length = ($this->currentPos + $length > $this->charCount) ? $this->charCount - $this->currentPos : $length;
118             if ($this->fixedWidth > 0) {
119                 $len = $length * $this->fixedWidth;
120                 $ret = substr($this->data, $this->currentPos * $this->fixedWidth, $len);
121                 $this->currentPos += $length;
122             } else {
123                 $end = $this->currentPos + $length;
124                 $end = $end > $this->charCount ? $this->charCount : $end;
125                 $ret = '';
126                 $start = 0;
127                 if ($this->currentPos > 0) {
128                     $start = $this->map['p'][$this->currentPos - 1];
129                 }
130                 $to = $start;
131                 for (; $this->currentPos < $end; ++$this->currentPos) {
132                     if (isset($this->map['i'][$this->currentPos])) {
133                         $ret .= substr($this->data, $start, $to - $start).'?';
134                         $start = $this->map['p'][$this->currentPos];
135                     } else {
136                         $to = $this->map['p'][$this->currentPos];
137                     }
138                 }
139                 $ret .= substr($this->data, $start, $to - $start);
140             }
141     
142             return $ret;
143         }

162         public function write(string $chars): void
163         {
164             $ignored = '';
165             $this->data .= $chars;
166             if ($this->fixedWidth > 0) {
167                 $strlen = \strlen($chars);
168                 $ignoredL = $strlen % $this->fixedWidth;
169                 $ignored = $ignoredL ? substr($chars, -$ignoredL) : '';
170                 $this->charCount += ($strlen - $ignoredL) / $this->fixedWidth;
171             } else {
172                 $this->charCount += $this->getUtf8CharPositions($chars, $this->dataSize, $ignored);
173             }
174             $this->dataSize = \strlen($this->data) - \strlen($ignored);
175         }

177         private function getUtf8CharPositions(string $string, int $startOffset, string &$ignoredChars): int
178         {
179             $strlen = \strlen($string);
180             $charPos = \count($this->map['p']);
181             $foundChars = 0;
182             $invalid = false;
183             for ($i = 0; $i < $strlen; ++$i) {
184                 $char = $string[$i];
185                 $size = self::UTF8_LENGTH_MAP[$char];
186                 if (0 == $size) {
187                     /* char is invalid, we must wait for a resync */
188                     $invalid = true;
189                     continue;
190                 }
191     
192                 if ($invalid) {
193                     /* We mark the chars as invalid and start a new char */
194                     $this->map['p'][$charPos + $foundChars] = $startOffset + $i;
195                     $this->map['i'][$charPos + $foundChars] = true;
196                     ++$foundChars;
197                     $invalid = false;
198                 }
199                 if (($i + $size) > $strlen) {
200                     $ignoredChars = substr($string, $i);
201                     break;
202                 }
203                 for ($j = 1; $j < $size; ++$j) {
204                     $char = $string[$i + $j];
205                     if ($char > "\x7F" && $char < "\xC0") {
206                         // Valid - continue parsing
207                     } else {
208                         /* char is invalid, we must wait for a resync */
209                         $invalid = true;
210                         continue 2;
211                     }
212                 }
213                 /* Ok we got a complete char here */
214                 $this->map['p'][$charPos + $foundChars] = $startOffset + $i + $size;
215                 $i += $j - 1;
216                 ++$foundChars;
217             }
218     
219             return $foundChars;
220         }
