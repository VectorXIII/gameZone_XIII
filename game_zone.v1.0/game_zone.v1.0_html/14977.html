37         /**
38          * Invoke a Psy Shell from the current context.
39          *
40          * For example:
41          *
42          *     foreach ($items as $item) {
43          *         \Psy\debug(get_defined_vars());
44          *     }
45          *
46          * If you would like your shell interaction to affect the state of the
47          * current context, you can extract() the values returned from this call:
48          *
49          *     foreach ($items as $item) {
50          *         extract(\Psy\debug(get_defined_vars()));
51          *         var_dump($item); // will be whatever you set $item to in Psy Shell
52          *     }
53          *
54          * Optionally, supply an object as the `$bindTo` parameter. This determines
55          * the value `$this` will have in the shell, and sets up class scope so that
56          * private and protected members are accessible:
57          *
58          *     class Foo {
59          *         function bar() {
60          *             \Psy\debug(get_defined_vars(), $this);
61          *         }
62          *     }
63          *
64          * For the static equivalent, pass a class name as the `$bindTo` parameter.
65          * This makes `self` work in the shell, and sets up static scope so that
66          * private and protected static members are accessible:
67          *
68          *     class Foo {
69          *         static function bar() {
70          *             \Psy\debug(get_defined_vars(), get_called_class());
71          *         }
72          *     }
73          *
74          * @param array         $vars   Scope variables from the calling context (default: [])
75          * @param object|string $bindTo Bound object ($this) or class (self) value for the shell
76          *
77          * @return array Scope variables from the debugger session
78          */
79         function debug(array $vars = [], $bindTo = null)
80         {
81             echo PHP_EOL;
82     
83             $sh = new Shell();
84             $sh->setScopeVariables($vars);
85     
86             // Show a couple of lines of call context for the debug session.
87             //
88             // @todo come up with a better way of doing this which doesn't involve injecting input :-P
89             if ($sh->has('whereami')) {
90                 $sh->addInput('whereami -n2', true);
91             }
92     
93             if (\is_string($bindTo)) {
94                 $sh->setBoundClass($bindTo);
95             } elseif ($bindTo !== null) {
96                 $sh->setBoundObject($bindTo);
97             }
98     
99             $sh->run();
100     
101             return $sh->getScopeVariables(false);
102         }
103     }

293                         exit(1);
294                     }
295     
296                     if (defined('HHVM_VERSION_ID') && \HHVM_VERSION_ID > 39999) {
297                         fwrite(STDERR, 'HHVM 4 or higher is not supported. You can set the environment variable PSYSH_IGNORE_ENV=1 to override this restriction and proceed anyway.' . PHP_EOL);
298                         exit(1);
299                     }
300     
301                     if (\PHP_VERSION_ID < 50509) {
302                         fwrite(STDERR, 'PHP 5.5.9 or higher is required. You can set the environment variable PSYSH_IGNORE_ENV=1 to override this restriction and proceed anyway.' . PHP_EOL);
303                         exit(1);
304                     }
305     
306                     if (\PHP_VERSION_ID > 89999) {
307                         fwrite(STDERR, 'PHP 9 or higher is not supported. You can set the environment variable PSYSH_IGNORE_ENV=1 to override this restriction and proceed anyway.' . PHP_EOL);
308                         exit(1);
309                     }
310     
311                     if (!function_exists('json_encode')) {
312                         fwrite(STDERR, 'The JSON extension is required. Please install it. You can set the environment variable PSYSH_IGNORE_ENV=1 to override this restriction and proceed anyway.' . PHP_EOL);
313                         exit(1);
314                     }
315     
316                     if (!function_exists('token_get_all')) {
317                         fwrite(STDERR, 'The Tokenizer extension is required. Please install it. You can set the environment variable PSYSH_IGNORE_ENV=1 to override this restriction and proceed anyway.' . PHP_EOL);
318                         exit(1);
319                     }
320                 }
321     
322                 $usageException = null;
323     
324                 $input = new ArgvInput();
325                 try {
326                     $input->bind(new InputDefinition(\array_merge(Configuration::getInputOptions(), [
327                         new InputOption('help',    'h', InputOption::VALUE_NONE),
328                         new InputOption('version', 'V', InputOption::VALUE_NONE),
329     
330                         new InputArgument('include', InputArgument::IS_ARRAY),
331                     ])));
332                 } catch (\RuntimeException $e) {
333                     $usageException = $e;
334                 }
335     
336                 try {
337                     $config = Configuration::fromInput($input);
338                 } catch (\InvalidArgumentException $e) {
339                     $usageException = $e;
340                 }
341     
342                 // Handle --help
343                 if ($usageException !== null || $input->getOption('help')) {
344                     if ($usageException !== null) {
345                         echo $usageException->getMessage() . PHP_EOL . PHP_EOL;
346                     }
347     
348                     $version = Shell::getVersionHeader(false);
349                     $argv    = isset($_SERVER['argv']) ? $_SERVER['argv'] : [];
350                     $name    = $argv ? \basename(\reset($argv)) : 'psysh';
351     
352                     echo <<<EOL

354

356       $name [--version] [--help] [files...]
357     
358     Options:
359       -h, --help            Display this help message.
360       -c, --config FILE     Use an alternate PsySH config file location.
361           --cwd PATH        Use an alternate working directory.
362       -V, --version         Display the PsySH version.
363           --color           Force colors in output.
364           --no-color        Disable colors in output.
365       -i, --interactive     Force PsySH to run in interactive mode.
366       -n, --no-interactive  Run PsySH without interactive input. Requires input from stdin.
367       -r, --raw-output      Print var_export-style return values (for non-interactive input)
368       -q, --quiet           Shhhhhh.
369       -v|vv|vvv, --verbose  Increase the verbosity of messages.
370     
371     EOL;
372                     exit($usageException === null ? 0 : 1);
373                 }
374     
375                 // Handle --version
376                 if ($input->getOption('version')) {
377                     echo Shell::getVersionHeader($config->useUnicode()) . PHP_EOL;
378                     exit(0);
379                 }
380     
381                 $shell = new Shell($config);
382     
383                 // Pass additional arguments to Shell as 'includes'
384                 $shell->setIncludes($input->getArgument('include'));
385     
386                 try {
387                     // And go!
388                     $shell->run();
389                 } catch (\Exception $e) {
390                     fwrite(STDERR, $e->getMessage() . PHP_EOL);
391     
392                     // @todo this triggers the "exited unexpectedly" logic in the
393                     // ForkingLoop, so we can't exit(1) after starting the shell...
394                     // fix this :)
395     
396                     // exit(1);
397                 }
398             };
