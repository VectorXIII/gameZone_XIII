15     namespace Ramsey\Collection\Map;

17     use Ramsey\Collection\AbstractArray;

20     use function array_key_exists;

24     /**
25      * This class provides a basic implementation of `MapInterface`, to minimize the
26      * effort required to implement this interface.
27      */
28     abstract class AbstractMap extends AbstractArray implements MapInterface
29     {
30         /**
31          * Sets the given value to the given offset in the map.
32          *
33          * @param mixed $offset The offset to set.
34          * @param mixed $value The value to set at the given offset.
35          *
36          * @throws InvalidArgumentException if the offset provided is `null`.
37          */
38         public function offsetSet($offset, $value): void
39         {
40             if ($offset === null) {
41                 throw new InvalidArgumentException(
42                     'Map elements are key/value pairs; a key must be provided for '
43                     . 'value ' . $value
44                 );
45             }
46     
47             $this->data[$offset] = $value;
48         }
49     
50         /**
51          * Returns `true` if this map contains a mapping for the specified key.
52          *
53          * @param mixed $key The key to check in the map.
54          */
55         public function containsKey($key): bool
56         {
57             return array_key_exists($key, $this->data);
58         }
59     
60         /**
61          * Returns `true` if this map maps one or more keys to the specified value.
62          *
63          * This performs a strict type check on the value.
64          *
65          * @param mixed $value The value to check in the map.
66          */
67         public function containsValue($value): bool
68         {
69             return in_array($value, $this->data, true);
70         }
71     
72         /**
73          * Return an array of the keys contained in this map.
74          *
75          * @return mixed[]
76          */
77         public function keys(): array
78         {
79             return array_keys($this->data);
80         }
81     
82         /**
83          * Returns the value to which the specified key is mapped, `null` if this
84          * map contains no mapping for the key, or (optionally) `$defaultValue` if
85          * this map contains no mapping for the key.
86          *
87          * @param mixed $key The key to return from the map.
88          * @param mixed $defaultValue The default value to use if `$key` is not found.
89          *
90          * @return mixed|null the value or `null` if the key could not be found.
91          */
92         public function get($key, $defaultValue = null)
93         {
94             if (!$this->containsKey($key)) {
95                 return $defaultValue;
96             }
97     
98             return $this[$key];
99         }
100     
101         /**
102          * Associates the specified value with the specified key in this map.
103          *
104          * If the map previously contained a mapping for the key, the old value is
105          * replaced by the specified value.
106          *
107          * @param mixed $key The key to put or replace in the map.
108          * @param mixed $value The value to store at `$key`.
109          *
110          * @return mixed|null the previous value associated with key, or `null` if
111          *     there was no mapping for `$key`.
112          */
113         public function put($key, $value)
114         {
115             $previousValue = $this->get($key);
116             $this[$key] = $value;
117     
118             return $previousValue;
119         }
120     
121         /**
122          * Associates the specified value with the specified key in this map only if
123          * it is not already set.
124          *
125          * If there is already a value associated with `$key`, this returns that
126          * value without replacing it.
127          *
128          * @param mixed $key The key to put in the map.
129          * @param mixed $value The value to store at `$key`.
130          *
131          * @return mixed|null the previous value associated with key, or `null` if
132          *     there was no mapping for `$key`.
133          */
134         public function putIfAbsent($key, $value)
135         {
136             $currentValue = $this->get($key);
137     
138             if ($currentValue === null) {
139                 $this[$key] = $value;
140             }
141     
142             return $currentValue;
143         }
144     
145         /**
146          * Removes the mapping for a key from this map if it is present.
147          *
148          * @param mixed $key The key to remove from the map.
149          *
150          * @return mixed|null the previous value associated with key, or `null` if
151          *     there was no mapping for `$key`.
152          */
153         public function remove($key)
154         {
155             $previousValue = $this->get($key);
156             unset($this[$key]);
157     
158             return $previousValue;
159         }
160     
161         /**
162          * Removes the entry for the specified key only if it is currently mapped to
163          * the specified value.
164          *
165          * This performs a strict type check on the value.
166          *
167          * @param mixed $key The key to remove from the map.
168          * @param mixed $value The value to match.
169          *
170          * @return bool true if the value was removed.
171          */
172         public function removeIf($key, $value): bool
173         {
174             if ($this->get($key) === $value) {
175                 unset($this[$key]);
176     
177                 return true;
178             }
179     
180             return false;
181         }
182     
183         /**
184          * Replaces the entry for the specified key only if it is currently mapped
185          * to some value.
186          *
187          * @param mixed $key The key to replace.
188          * @param mixed $value The value to set at `$key`.
189          *
190          * @return mixed|null the previous value associated with key, or `null` if
191          *     there was no mapping for `$key`.
192          */
193         public function replace($key, $value)
194         {
195             $currentValue = $this->get($key);
196     
197             if ($this->containsKey($key)) {
198                 $this[$key] = $value;
199             }
200     
201             return $currentValue;
202         }
203     
204         /**
205          * Replaces the entry for the specified key only if currently mapped to the
206          * specified value.
207          *
208          * This performs a strict type check on the value.
209          *
210          * @param mixed $key The key to remove from the map.
211          * @param mixed $oldValue The value to match.
212          * @param mixed $newValue The value to use as a replacement.
213          *
214          * @return bool true if the value was replaced.
215          */
216         public function replaceIf($key, $oldValue, $newValue): bool
217         {
218             if ($this->get($key) === $oldValue) {
219                 $this[$key] = $newValue;
220     
221                 return true;
222             }
223     
224             return false;
225         }
226     }
