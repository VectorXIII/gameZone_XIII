21     namespace Mockery\Matcher;
22     
23     class Ducktype extends MatcherAbstract
24     {
25         /**
26          * Check if the actual value matches the expected.
27          *
28          * @param mixed $actual
29          * @return bool
30          */
31         public function match(&$actual)
32         {
33             if (!is_object($actual)) {
34                 return false;
35             }
36             foreach ($this->_expected as $method) {
37                 if (!method_exists($actual, $method)) {
38                     return false;
39                 }
40             }
41             return true;
42         }
43     
44         /**
45          * Return a string representation of this Matcher
46          *
47          * @return string
48          */

50         {
51             return '<Ducktype[' . implode(', ', $this->_expected) . ']>';
52         }
