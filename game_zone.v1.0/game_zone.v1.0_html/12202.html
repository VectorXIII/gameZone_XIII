21     namespace Mockery\Generator;
22     
23     class CachingGenerator implements Generator
24     {
25         protected $generator;
26         protected $cache = array();
27     
28         public function __construct(Generator $generator)
29         {
30             $this->generator = $generator;
31         }
32     
33         public function generate(MockConfiguration $config)
34         {
35             $hash = $config->getHash();
36             if (isset($this->cache[$hash])) {
37                 return $this->cache[$hash];
38             }
39     
40             $definition = $this->generator->generate($config);
41             $this->cache[$hash] = $definition;
42     
43             return $definition;
44         }
45     }
