67         {
68             $k = Caster::PREFIX_PROTECTED.'code';
69             if (isset($a[$k], self::$errorCodes[$a[$k]])) {
70                 $a[$k] = new ConstStub(self::$errorCodes[$a[$k]], $a[$k]);
71             }
